قرار أميري رقم (34) لسنة 2022 بالهيكل التنظيمي لوزارة المـالـيـة 34 / 2022
عدد المواد: 24
فهرس الموضوعات

المواد (1-24)



نحن تميم بن حمد آل ثاني أمير دولة قطر،
بعد الاطلاع على الدستـور،
وعـلـى قـانـون الموارد البشـريـة المـدنيـة الصـادر بالـقـانـون رقـم (15) لسـنة 2016، المعدل بالـقـانـون رقم (??) لسنة 2020،
وعلـى القـرار الأميــري رقم (10) لسـنة 2019 بالهيكل التنظيــمي لوزارة الماليـة،
وعلـى القـرار الأميــري رقـم (57) لسنة 2021 بتعييـن اخـتـصـاصـات الوزارات،
وعلى اقتراح مجلس الوزراء،
قررنـا مـا يلي:

المواد

المادة 1


     يتـكـون الهيكل التنظيمـي لـوزارة الماليـة مـن الوحدات الإداريـة المبينة بالخريـطـة التنظيميـة المرفقة بهذا القرار، وهي:
أولاً: الوحدات الإداريـة التـابـعـة للـوزيـر:
١- مـكـتـب الـوزيـر.
٢- المـكـتـب الفـنـي.
٣- إدارة التـدقـيـق الـداخـلـي.
 
 
 
ثانياً: الـوحـدات الإداريـة التـابـعـة لـوكـيـل الوزارة:
۱- مكتب وكيـل الـوزارة.
ثالثاً: الوحـدات الإداريـة الـتـابـعـة لوكيـل الـوزارة المـسـاعـد للشـؤون الاقـتـصـاديـة:
۱- مكتـب وكيـل الـوزارة المسـاعـد.
٢- إدارة البحوث الاقـتـصـاديـة.
٣- إدارة الـتـنـافـسـيـة الاقـتـصـاديـة.
٤- إدارة التـعـاون الدولي والاتـفـاقـيـات المـاليـة.
رابعاً: الوحـدات الإداريـة التـابـعـة لوكيـل الـوزارة المـسـاعـد لـشـؤون الـسـيـاسـات المـاليـة:
۱- مكتـب وكيـل الـوزارة المسـاعـد.
٢- إدارة الـسيـاسـات الائتـمـانـيـة والـديـن الـعـام.
٣- إدارة السيـاسـات المـاليـة.
٤- إدارة تنظيم المشـروعـات الـعـامـة.
٥- إدارة تنظيم المشـتـريـات الحكوميـة.
خامساً: الوحـدات الإداريـة التـابـعـة لوكيـل الـوزارة المـسـاعـد لـشـؤون الخـزانـة الـعـامـة:
۱- مكتـب وكيـل الـوزارة المسـاعـد.
٢- إدارة الـرقـابـة المـاليـة.
٣- إدارة الموازنـة الـعـامـة.
٤- إدارة الحسابات والخـزانـة الـعـامـة.
 
 
سادساً: الوحـدات الإداريـة التـابـعـة لوكيـل الـوزارة المـسـاعـد لشؤون الخـدمـات المـشـتـركـة:
۱- مكتـب وكيـل الـوزارة المسـاعـد.
٢- إدارة التخطيط والجـودة والابتكار.
٣- إدارة الشـؤون الـقـانـونيـة.
٤- إدارة العـلاقـات الـعـامـة والاتـصـال.
٥- إدارة الموارد البـشـريـة.
٦- إدارة الشـؤون المـالـيـة والإداريـة.
٧- إدارة نـظـم المـعـلـومـات.
 

 

المادة 2


يختـص المكتـب الفـنـي بمـا يـلـي:
١- إبـداء الـرأي الفني في المـوضـوعـات التي تُحـال إليـه مـن الوزيـر.
۲- دراسـة وتقييم التقاريـر والمشـروعـات والبـرامـج المتعلقة بأنشـطـة الوزارة، التي تُحـال إليـه مـن الوزيـر.
۳- بحث ودراسـة ومـتـابـعـة المـوضـوعـات الـتـي تُحـال إليـه مـن الوزيـر، بالتنسيق مع الوحدات الإداريـة المعـنيـة.
٤- إعـداد ملخص للمـواضيـع والتقاريـر والـدراسـات الـتـي يـلـزم عـرضـها على الـوزيـر، وإبـداء الـرأي بشـأنـها.
 

 

المادة 3


تختص إدارة التـدقيـق الـداخـلـي بمـا يـلـي:
١- إعـداد مشروع خـطـة التـدقـيـق السنـويـة المبنيـة علـى المخـاطـر لـلـوزارة، ورفعها للوزيـر لاعتمادها، وإعداد تقاريـر بـنـتـائـج التـدقيـق.
۲- مـراقبـة مـدى الالتزام بتنفيـذ القـوانـين واللوائح والقرارات المنظمـة للعمـل.
۳- مـراجعـة التعليمات الماليـة، واقـتـراح مـا يـلـزم لـهـا مـن تـعـديـلات، ومـراقبـة تنفيـذهـا.
٤- مـراقبـة الأعمـال الإداريـة وإجـراءات تعيين الموظفين، وإجازاتـهم، وتـرقـيـاتـهم، وغير ذلـك ممـا يـتـعـلـق بـشـؤون الموظفين، والتحقـق مـن الالتزام بالقوانين واللوائح والقـرارات المنظمة لذلك.
٥- مـراقبـة الأعمـال المـاليـة والمحاسبيـة، وسندات الصرف، وإجـراءات القيـد في السجـلات المحاسبيـة، والتـأكـد مـن تطبيق القـواعـد المعتمدة في هـذا الشـأن.
٦- تقييـم المخـاطـر المتعلقة بالتدقيـق المالي والإداري، واقتراح إجـراءات الحـد منـها.
۷- متـابعـة الإجـراءات التـصـحـيـحـيـة المطـلـوبـة مـن الوحـدات الإداريـة المعنية، وفـق نتائج التدقيـق، وإعداد التقارير الدوريـة بشـأنـها.
٨- التـأكـد مـن سـلامـة الأمـوال والعـهـد والأثاث والأدوات والمبـانـي والمـوجـودات الأخرى التي تملكها الوزارة أو تخضع لإشـرافـها.
٩- الاشتـراك في أعمـال الجـرد الـدوري والمفاجئ علـى المخـازن والعهـد والتـأكـد مـن دقـة التسجيـل بـالـدفـاتـر وسـلامـة الإجـراءات المطبقـة.
۱۰- متـابعـة نتائج تقاريـر الجهـات الـرقـابيـة، وإبداء التـوصـيـات المنـاسبـة بشأنها.
 

 

المادة 4


تختص إدارة البحـوث الاقـتـصـاديـة بمـا يـلـي:
١- إعـداد البحوث والدراسات والتقارير الماليـة والاقـتـصـاديـة عـن سـيـاسـة الاقـتـصـاد الـكـلي والقـطـاعـات الاقـتـصـاديـة المختـلفـة، وغيـرهـا مـن الأمـور الاقـتـصـاديـة، وإعـداد المقـتـرحـات والـتـوصـيـات بشـأنـها.
۲- دراسـة وتحـليـل المتغيرات المالية والاقتصاديـة المحـليـة والدوليـة، وبيـان مـدى تـأثـيـرهـا على السياسات والبـرامـج الاقـتـصـاديـة.
٣- دراسة وتحـلـيـل نـتـائج تنفيـذ السياسات والخطط الاقـتـصـاديـة والقوانين المتصـلـة بـها، واقـتـراح سبـل تـذلـيـل مـعـوقـاتـها، بالتنسيـق مـع الوحدات الإداريـة المعـنـيـة والجـهـات المختصة.
٤- تحـديـد وتصنيـف قـطـاعـات اقـتـصـاد الدولـة، وجـمـع البيـانـات المتعلقة بتـلك القطـاعـات مـن مصـادرهـا لإعـداد قـاعـدة بيـانـات اقتصـاديـة شـامـلـة، وتحـلـيـل بـيـانـاتـها، ووضـع مـؤشـرات اقتصـاديـة عـامـة لـكـل قـطـاع اقـتـصـادي في الـدولـة.
٥- بنـاء قـاعـدة لبيـانـات المؤشرات الاقـتـصـاديـة وتحـديـثـها بانتـظام، وتنـسـيـق وجـمـع البـيـانـات مـن الجـهـات الحـكـوميـة الأخـرى.
٦- إنشاء وتطـويـر تـطـبـيـق إلكتروني أو أكثـر لعـرض تقاريـر أداء الاقتصاد الوطني والتـحـلـيـلات المصـاحـبـة وبيـان أثرها، بالتنسيـق مـع الـوحـدات الإداريـة المعـنيـة.
 

 

المادة 5


تختص إدارة التنافسية الاقتصادية بما يلي:
١- اقتراح المبادرات الاقتصادية المتعلقة بالاقتصاد الوطني.
٢- اقتراح التشريعات فيما يخص زيادة القدرة التنافسية للاقتصاد الوطني.
٣- اقتراح السياسات والبرامج والخطط بهدف تعزيز التنوع الاقتصادي وخلق فرص اقتصادية جديدة، بالتنسيق مع الجهات المعنية بالدولة.
٤- دراسة تنويع مصادر الدخل العام، ودعم التنمية المستدامة في الدولة، بالتنسيق مع الوحدات الإدارية والجهات ذات العلاقة.
٥- دراسـة الأبـعـاد الاقـتـصـاديـة لمشـروعـات القـوانـين والتشـريـعـات والسيـاسـات والبـرامـج الـتـي تـم اقـتـراحـها، بالتنسيـق مـع الجهـات المعنية في الـدولـة.
٦- التنسيـق مـع الـوحـدة الإداريـة المعـنـيـة والجـهـات المختصـة بشـأن مـا يجب اتخاذه مـن إجـراءات لتـطـويـر وتحسين أداء الاقـتـصـاد بما ينعكس إيجاباً على ترتيب وتصنيـف الـدولـة فـي هـذا المجـال.

 

المادة 6


تختص إدارة التعاون الدولي والاتـفـاقـيـات المـاليـة بمـا يـلـي:
١- إعداد أوراق العمل المتعلقة بالمؤتمرات والندوات والاجتماعـات المحـليـة والإقـلـيـمـيـة والدوليـة فـيـمـا يـخـص نشـاط الـوزارة، بالتنسيـق مـع الوحـدات الإداريـة المعنيـة والجـهـات المخـتـصـة.
٢- الإعـداد لحضـور المؤتمرات والاجتمـاعـات الـتـي تـعـقـدهـا المنظمـات والهيئـات الإقليميـة والدوليـة المخـتـصـة بقضـايـا الـوزارة، الـتي تشـارك الدولـة في عضـويـتـها.
٣- إعـداد التقاريـر والـردود التي تطلبهـا المنظمات والهيئات الإقليميـة والدوليـة فيمـا يـخـص نشـاط الـوزارة، بالتنسـيـق مـع الوحـدات الإدارية المخـتـصـة.
٤- دراسـة ومتـابـعـة تنفيـذ تـوصـيـات المنظمـات والهيئـات الإقليميـة والدوليـة، فيمـا يـخـص نشـاط الـوزارة، بالتنسيق مـع الجهـات المخـتـصـة بـالـدولـة.
٥- إعداد التقارير اللازمـة بشـأن إسـهـامـات المنظمات والهيئات الإقليميـة والدوليـة في تنميـة وتـدعـيـم الأنشـطـة المختلفـة في مجـالات عمـل الوزارة، بالتنسيق مع الجهات المخـتـصـة.
٦- متـابـعـة سـداد المساهمات الماليـة للدولـة في المنظمـات المعنيـة بمـجـالات عـمـل الـوزارة الـتـي تكـون الـدولـة عـضـواً فـيـها، بالتنسيق مـع الجهات المخـتـصـة.
۷- دراسة السبل الكفيلة بتـفـعـيـل دور الدولـة في مجـال نـشـاط الـوزارة مع المنظـمـات العـربـيـة والإقـلـيـمـيـة والـدوليـة.
۸- دراسة وتحـلـيـل مـشـروعـات الاتفاقيـات المـاليـة، ومـتـابـعـة تـنـفـيـذهـا، وإعـداد تقاريـر عـن تـطـور العـلاقـات المـاليـة والاقـتـصـاديـة النـاشـئـة عنها، وذلك بمـا يـخـدم أغـراض الخطط الإنمائيـة للـدولـة، بالتنسيـق مـع الـوحـدات الإداريـة المعـنيـة.
٩- المشاركة في إعداد مشـروعـات الاتفاقيات الماليـة المتعلقة باختصـاصـات الوزارة، بالتنسيق مع الوحدات الإداريـة والجـهـات المخـتـصـة.
١٠- إعداد الدراسات اللازمـة مـن قبـل الـوزارة لأعـمـال الـفـريـق التفاوضـي للدولـة، ولمـواقـف الدولـة حـيـال المفاوضـات الماليـة، بالتنسيـق مع الوحدات الإداريـة المعـنـيـة والجـهـات المـخـتـصـة.
۱۱- جمـع المعلـومـات والبحـوث الصـادرة عـن المنظمات والهيئـات ذات الصلة بقضـايـا الـوزارة محـليـاً وإقليمياً ودولياً، ونشـرهـا بـغـرض الاسـتـفـادة مـنـها.
 

 

المادة 7


تختص إدارة السياسات الائتـمـانـيـة والـديـن الـعـام بمـا يـلـي:
۱- تـطـويـر استراتيجـيـة القـروض الـتـي تـرتـكـز عـلـى تـحـديـد أطـراف القروض ومصادر التمـويـل والمعـلـومـات التي يجـب تـوفـرهـا للمـمـولـين وشركات التقييم والمستثمرين ووكالات التصنيف الائتماني.
۲- تـطـويـر سيـاسـات وخطط التمـويـل بمـا يـحـقـق خـفـض تكاليف التمـويـل في الأسواق المحـلـيـة والـعـالميـة، بالتنسيـق مـع الجهات المختصة في الدولة.
٣- ضبط الالتـزامـات والضمـانـات على الوحدات المالية الحكومية والأجهـزة الحكوميـة الأخـرى والشـركـات التي تمتلك الدولـة حـصـة فـيـها، مـن خلال تحـلـيـل التـدفـقـات النـقـديـة والسيـولـة المتـوفـرة.
٤- العـمـل على تنـويـع مـصـادر التمـويـل مـن الأسـواق المحـليـة والعـالميـة لسد احتياجات الـدولـة التمويليـة، وفقاً لسـقـف الـديـن الـذي تحدده الـدولـة.
٥- التنسيق مع الوحدات الماليـة الحـكـوميـة والجـهـات المختصة في الدولة بشـأن تـطـويـر أسس العمـل لتحقيق أعلى تصنيف ائتماني للـدولـة، والمحافظة عليـه عـن طـريـق وكـالات التصنيف الائتمانيـة الـدوليـة بالاعـتـمـاد عـلى المعـايـيـر الـدوليـة ذات الصلة.
٦- اقـتـراح استراتيجيـات وخطط التمـويـل المتعلقة بتأمين القـروض والضمانات المطـلـوبـة بالتنسيق مع الجهة المختصة والجهات المعنيـة.
٧- إدارة المخاطـر المتعلقة بالعملات الأجنبيـة لسداد الديـون وفـوائـدهـا، بالتنسيق مع الوحدات الإداريـة المعـنيـة في الوزارة والجهات المختصة.
٨- إدارة القروض والسـنـدات والصكوك وأذون الخـزانـة الـدائـنـة والمـديـنـة والضمانات الحكومية الداخـلـيـة والخـارجـيـة، بالتنسيـق مـع الجـهـات المعنيـة.
٩- إعـداد التقارير الدوريـة بشـأن الالتزامات الماليـة والضـمـانـات القـائمـة على الحكومة والأجـهـزة الـتـابـعـة لـها
 

 

المادة 8


تختص إدارة السيـاسـات المـاليـة بمـا يـلـي:
۱- مـتـابـعـة التـطـورات الماليـة المحـلـيـة والخـارجـيـة، وقيـاس أثـرهـا علـى المـوازنـة الـعـامـة للـدولـة، ووضـع التصورات والتـوصـيـات بـشـأن السياسات المالية اللازمة، بالتنسيق مع الجهات المختصة في الدولة.
٢- إعـداد تـوقـعـات بالمؤشرات الماليـة الرئيسية، وبناء النماذج الخاصـة بـذلك.
٣- إعـداد الإطـار المـالي المتـوسـط والـطـويـل المـدى متضمـناً الأسـقـف الماليـة، واقـتـراح السيـاسـة المـاليـة العـامـة للدولـة بمـا يـتـوافـق مع تحـقـيـق الأهـداف التنمويـة للـدولـة.
٤- مـتـابـعـة تنفيذ السياسـة المـاليـة، وإعـداد التـقـاريـر الـدوريـة فـي هـذا الشـأن متضمنـة التـوصـيـات اللازمـة.
٥- إعداد الدراسات بشأن الآثار المالية والاقـتـصـاديـة المتـوقـعـة للسيـاسـات والقرارات، بالتنسيق مع الجهات المختصة في الـدولـة.
٦- إعداد الدراسات حول مستـوى الـكـفاءة في الإنفاق الـعـام للـدولـة على المستوى القطاعي، وانعكاسها على كفاءة أداء القـطـاعـات الرئيسيـة.
٧- إنـشـاء قـاعـدة بـيـانـات للمؤشرات المالية والاقـتـصـاديـة، بالتـعـاون مـع الوحـدات المعـنـيـة وبالتنسيـق مـع الجـهـات المختصة في الدولـة، وتـوفـيـر بـيـانـات ومـعـلـومـات مـحـدثـة ودراسـات تحـلـيـلـيـة في المجـالات الماليـة والاقـتـصـاديـة.
۸- تـوفـيـر البـيـانـات والإحصاءات والمعـلـومـات المـاليـة الخاصـة بعمليـات الوزارة للمسـتـخـدمـين مـن داخـل الـوزارة أو مـن الجـهـات الحكوميـة الأخـرى.
۹- اقتراح مشـروعـات الأدوات التشريعية ذات الصلة بالسياسات الماليـة.
 

 

المادة 9


تختص إدارة تنـظيـم المشـروعـات الـعـامـة بمـا يـلـي:
۱- اقتــراح السيـاسـات والآليات اللازمـة لتنـظـيـم وإدارة المشـروعـات الـعـامـة، بالتنسيق مع الجهات المختصة.
۲- تقييم المشـروعـات العـامـة القـائمـة والمقـتـرحـة مـن الجـهـات الـتـي تمـول مـوازنـاتـهـا مـن الدولـة، بالتنسيـق مـع الـوحـدات الماليـة الحـكـوميـة، وذلك تجـنـبـاً لـتـضـارب القرارات.
۳- متـابـعـة المشـروعـات الـعـامـة بمـا يـحـقـق الاسـتـخـدام الأمثـل لمـوارد الدولة، وتـوجـيـهـها بمـا يـتـوافـق مـع سـيـاسـة التنميـة الشـامـلـة للـدولـة في هـذا الـشـأن.
٤- إنشاء قاعـدة بيـانـات مـتـكـامـلـة للمشـروعـات الـعـامـة، والعمـل على تحديثها بالبيـانـات اللازمـة مـن خـلال الـربـط مـع الجـهـات المختصة في الدولة، بالتنسيق مع الوحدة الإداريـة المعنية في الوزارة.
 

 

المادة 10


تختص إدارة تنـظيـم المشـتـريـات الحكوميـة بمـا يـلـي:
١- إعـداد استراتيجيـة المشتـريـات الحكومية في الدولة، والإشراف على تنفيـذهـا بـعـد اعتمـادهـا مـن الجهات المختصة، والعمل على تطـويـرهـا وفقاً لأفـضـل الممارسات الـعـالميـة.
٢- إعـداد وتـطـويـر الآليـات والقـواعـد والإجـراءات القـانـونيـة والإداريـة والفـنـيـة، وإصـدار الـتـعـامـيـم الـلازمـة لتنفيذ أحكام القـانـون المنـظم للمنـاقـصـات والمزايــــدات والمشتـريـات الحكوميـة، وإبـداء الـرأي في الاستفسارات الخاصـة فـي هـذا الـشـأن.
٣- إعـداد السيـاسـات والإجـراءات والضمـانـات والحـوافـز اللازمـة لـدعـم القـدرات التنافسيـة بين الموردين للمنتجـات، ومقـدمـي الخـدمـات الوطنيـة والـصـديـقـة للبيئـة، والمشـاريع الصغيرة والمتـوسـطـة، بالتنسيق مع الجهات المخـتـصـة بـالـدولـة.
٤- إعـداد نمـاذج مـوحـدة للمشتـريـات، وتصنيـف وتقييم أداء المقاولين والموردين، وفقاً لمعاييـر الشـفـافـيـة والعـدالـة المتبعـة عـالميـاً، وتطبيقها وتحديثها على النظام المـوحـد للمشتـريـات الحـكـوميـة، بالتنسيـق مـع الوحدات الإداريـة المـعـنـيـة في الوزارة والجـهـات المخـتـصـة بـالـدولـة. 
٥- إعداد وتـطـويـر قـاعـدة بيانات إحـصـائيـة للمشـتـريـات الحـكـومـيـة وأداء المقاولين والموردين ومقدمي الخـدمـات، وتنظيم السـجـلات والتقاريـر الخـاصـة بها، وفقاً لأفـضـل الممارسـات الـعـالميـة.
٦- مـراقـبـة أعـمـال لجـان المنـاقـصـات والمـزايـدات، وإصـدار الآليـات والقـواعـد والإجراءات القـانـونيـة والإداريـة والفـنـيـة، وإصـدار التعـامـيـم اللازمـة لـتـطـويـر أعـمـالـها.
٧- إعـداد وتـطـويـر الآليـات والـقـواعـد والإجـراءات القـانـونيـة والإداريـة والفـنـيـة وإصـدار التعـامـيـم اللازمة لتنفيـذ القيمـة المحـليـة، وإعـداد وتـطـويـر قـاعـدة بيـانـات إحصائيـة، وتنـظـيـم السـجـلات والتـقـاريـر الخاصـة بـها.
 

 

 

المادة 11


تختص إدارة الـرقـابـة المـالـيـة بمـا يـلـي:
١- إعـداد وإصـدار الـقـواعـد واللوائح والإجـراءات والتـعـامـيـم المـالـيـة والمحـاسـبـيـة فيما يخص الوحدات المالية الحكومية، والـرقـابـة المـالـيـة عليها، وإصـدار تـقـاريـر المتـابـعـة بشـأنـها.
۲- مـراقـبـة تنفيـذ القـواعـد واللوائح والإجـراءات والتـعـامـيـم المـاليـة والمحـاسـبيـة، وإصـدار تـقـاريـر المتـابـعـة بشـأنـها.
٣- إعـداد وتـطـويـر النـمـاذج والمسـتـنـدات الماليـة المسـتـخـدمـة وتعميمها، بالتنسيق مع الوحدات الإدارية ذات الـعـلاقـة.
٤- وضـع مـعـايـيـر وإجـراءات التـدقـيـق الداخـلي على مستـوى الوحـدات الماليـة الحـكـوميـة، وإصـدار تـقـاريـر المتـابـعـة بشـأنـها.
٥- إصـدار الـقـواعـد واللوائح والإجراءات المتعلقة بالـرقـابـة علـى البـرامـج والأنظمـة المـاليـة والمحاسبيـة، ومـراقـبـة تـنـفـيـذهـا.
٦- مـتـابـعـة إجـراءات إعـداد الحـسـاب الختامي والإغـلاق الـرسمي، والحصول على شهادة التأييد للحساب الختامي بالتنسيـق مـع ديـوان المحاسبة.
۷- اقـتـراح وإبـداء الـرأي في مشـروعـات الأدوات التشريعيـة ذات الصلة بالانعكاسات المـالـيـة.
٨- إعـداد وتـطـويـر الآليـات والـقـواعـد والإجـراءات القـانـونيـة اللازمـة لتنفيذ أحـكام قـانـون النـظـام المـالي.
 

 

 

المادة 12


تختص إدارة الموازنـة الـعـامـة بمـا يـلـي:
١- إعـداد المنـشـور السنوي للموازنـة الـعـامـة، والذي يتضمـن الـقـواعـد التنفيـذيـة والمـوجـهـات العـامـة لتقييم أداء الوحدات الماليـة الحـكـوميـة الـواجـب اتبـاعـها لإعـداد وتنفيذ الموازنـة الـعـامـة.
۲- مـنـاقـشـة تقـديـرات الموازنـة العـامـة للدولـة في ضـوء مـا تـضـعـه الـوحـدات المـالـيـة الحكوميـة لمؤشرات ومعـايـيـر أداء نتـائـجـها ومـخـرجـاتـهـا وجـدوى عـمـلـيـاتـها المتعددة، وإعـداد المشـروع النهـائي للمـوازنـة الـعـامـة للـدولـة.
۳- مـتـابـعـة تـنـفـيـذ المـوازنـة العـامـة للدولة مع الوحدات المالية الحكوميـة والأجـهـزة الأخرى، وتقييـم مـخـرجـاتـهـا ومـدى تحقيقها لمبـدأ القيمـة مقـابـل التـكـلفـة، وإعـداد التـقـاريـر اللازمـة بشـأنـها.
٤- مـراجـعـة وتقييـم طلبـات الاعتمـاد الإضـافي والمنـاقـلات، وتحـديـد إمكانـيـات ووسـائـل تمـويـلـها.
٥- إبداء الرأي فـيـمـا تـعـده الوحدات الماليـة الحـكـومـيـة مـن تـقـاريـر بـنـتـائج أدائها، وتـقـديـم الـدعـم الفـنـي اللازم لتـطـويـرهـا.
٦- المتـابـعـة الدوريـة لمؤشرات قيـاس أداء الوحـدات الماليـة الحـكـوميـة فيما يخص الموازنـة العـامـة، وإعـداد تـقـريـر سـنـوي عـنـهـا مـتـضـمنـاً مـسـتـويـات الأداء والمـعـوقـات والحلول المقـتـرحـة.
۷- اقـتـراح مشـروعـات القـوانـين واللوائح والقرارات والبـرامـج الـتـي لهـا انـعـكـاسـات مـالـيـة، وتحـديـد كـيـفـيـة تـمويـلـهـا مـن خـلال الموازنـة العـامـة للـدولـة.
۸- تقييـم وتحـديـث وإدارة البيـانـات المـاليـة الحـكـوميـة، بالتنسيـق مـع الوحدات الإداريـة المعنية في الوزارة والجـهـات المختصة في الدولة.
 

 

 

المادة 13


تختص إدارة الحـسـابـات والخـزانـة الـعـامـة بمـا يـلـي:
۱- تـرحـيـل جميع العمليات الماليـة مـن مـصـروفـات وإيرادات الوحـدات الماليـة الحكومية والأجـهـزة الأخـرى.
٢- إعـداد المـراكـز الماليـة لعمـلـيـات الصـرف والقبـض، ومـسـاهـمـات ومـسـاعـدات الدولة المحـليـة والخارجيـة، وإجـراء مـا يـلـزم بشـأنـهـا مـن مـطـابـقـات دورية مع الجهات المختصـة.
٣- التدقيـق على أوجـه الصـرف فـي طـلـبـات الـدفـع والـتـسـويـات الخـاصـة بالوحدات الماليـة الحكوميـة.
٤- إدارة عمليات الصرف للوحدات الماليـة الحكومية والأجـهـزة الأخـرى، ومـتـابـعـة وتحصيل الإيرادات الـعـامـة للـدولـة.
٥- إدارة ومـحـاسـبـة الأصـول الثـابتـة بالجـهـات الحكوميـة.
٦- تنفيـذ استراتيجيات وخطط التمـويـل المتـعـلـقـة بتـأمـين القروض والضـمـانـات المـطـلـوبـة للجـهـات المعنيـة.
۷- سداد أقـسـاط وفـوائـد القروض والسندات والصكوك وأذون الخزانـة الـدائـنـة والمـديـنـة والضـمـانـات الحـكـوميـة الـداخـلـيـة والخارجيـة، وإجـراء التـسـويـات، بالتنسيق مع الجـهـات المخـتـصـة.
۸ - مـتـابـعـة الإغلاق المبدئي للموازنـة الـعـامـة، وتحويـل أرصدة حسـابـات الوحدات المالية الحكومية إلى حساب الدولة المـوحـد، وفق المـواعـيـد المقررة، بالتنسيق مع الوحدات الإداريـة المعنيـة في الوزارة.
٩- مـتـابـعـة المخاطر المتعلقة بالعملات الأجنبيـة لسداد الـديـون وفـوائـدهـا، بالتنسيق مع الوحدات الإدارية المعنية في الوزارة والأجهزة الأخـرى.
١٠- إعـداد كشوف التدفقات النقـديـة الدورية للوحدات الماليـة الحـكـوميـة، وفتـح الاعتمـادات المستنـديـة وقـيـدهـا بالحسـابـات المتعلقـة بـها، ومـراقبـة تـنـفـيـذهـا.
۱۱- فـتـح وإدارة حساب الخـزانـة المـوحـد، ومتـابـعـة بـاقـي حـسـابـات الخـزانـة، وتغذيتها بالسيـولـة اللازمة بالتنسيق مع الجهات المعنيـة.
۱۲- المـراجـعـة الدورية للحسابات المفتـوحـة لـدى المصارف، واقـتـراح إلـغـاء غير الضروري منها.
۱۳ - إبـداء الـرأي بشـأن سـبـل تمـويـل العـجـز المالي الحكومي، بالتنسيـق مع الإدارات المعـنيـة.
١٤- اقـتـراح مشـروعـات القـوانين واللوائح المتعلقة بالمصـروفـات والإيرادات الـعـامـة وسـبـل التـمـويـل.
 

 

المادة 14


تختص إدارة التخطيط والجـودة والابتكار بمـا يـلـي:
١- إعـداد الخطـة الاستراتيجيـة للوزارة، بالتنسيـق مـع الجهات المختصة والوحـدات الإدارية المعنيـة، واتخاذ إجراءات اعتمـادهـا.
۲- حصر التحديات القائمة، وتحـديـد وتقييم المخاطر المؤسسيـة المتـوقـع حدوثـها، والـتـي قـد تـؤثـر علـى تحقيـق أهـداف الوزارة، وتقـديـم المقترحات بشأنها، بالتنسيق مع الوحدات الإداريـة المعـنيـة.
٣- إدارة وتجهيز خطط بديلـة تضمن استمراريـة تشغيـل الأعمـال بالـوزارة في حـالات الـطـوارئ والأزمـات، بالتنسيـق مـع الـوحـدات الإداريـة المختـلفـة كـل فـيـما يخـصـها.
٤- رفع مشـروعـات الخطط التنفيذية والمشـاريـع والبـرامـج الخـاصـة بالوحدات الإدارية المختـلفـة، للاعتمـاد مـن قبـل الـوزيـر.
٥- متـابـعـة تنفيذ الخطط المتعلقة بأنشـطـة الـوزارة، وتقييـم نـتـائـجـها، ورفـع تـقـاريـر دوريـة بشـأنـها.
٦- دراسـة المشـاكـل والمعـوقـات الـتـي تـواجـه تنفيـذ الخطـة الاستراتيجيـة والخطط التنفيـذيـة، واقتراح الحلول المناسبة لها.
۷- تقييـم جـودة الخـدمـات المقـدمـة مـن الوحـدات الإداريـة، بالتنسيـق مع الجهـة المختـصـة.
۸ - مـراجـعـة وتقييـم جـودة الأداء علـى مـسـتـوى الوحـدات الإداريـة، ووضـع مـقـتـرحـات لـرفـع الـكفـاءة العـامـة بـها.
٩- التحـقـق مـن انسجام نـظـم وأساليب العمـل مـع اختصـاصـات وأهـداف الوزارة، واقتــراح خطـط تـطـويـر تـلـك النـظـم والأساليـب، وتبسيط الإجراءات، بالتنسيق مع الوحدات الإدارية المعنية والجهة المختصة.
۱۰- دراسـة مـشـاكـل ومعـوقـات العمـل بالــــوزارة، وبحـث أسبـابـها، واقـتـراح الحلول المنـاسبـة لـها.
۱۱- تلقي المبادرات والمقترحات والأفكار الإبـداعـيـة، ودراستها بالتنسيـق مع الوحدات الإداريـة المعنيـة والجـهـة المختـصـة.
۱۲- رصد وتجميـع وتحـلـيـل البيانات الإحصائيـة الخـاصـة بأنشـطـة الـوزارة، والتنسيق مع الوحدات الإدارية المختصة بشأن التقاريـر الإحـصـائيـة المطـلـوبـة مـن الـوزارة.
۱۳- التنسيـق مـع الوحدات الإداريـة المعنيـة والجهـة المختصة فيما يتعلـق بالمؤشـرات الـدوليـة المتـعـلـقـة بأنشـطـة واخـتـصـاصـات الوزارة
 

 

المادة 15


تختص إدارة الشـؤون الـقـانـونيـة بمـا يـلـي:
١- بحـث ودراسـة ومتـابعـة المسـائـل القـانـونيـة الخـاصـة بنشـاط الـوزارة، بالتنسيق مع الوحدات الإداريـة المعنيـة.
٢- إعـداد مشـروعـات الأدوات التشريعيـة الخـاصـة بالوزارة، وإبـداء الـرأي في مشـروعـات القـوانـين التي تُحـال إليـها.
٣- إبـداء الـرأي الـقـانـوني في المـوضـوعـات التي تُحـال إليـها.
٤- إعـداد مشـروعـات العقـود والاتفاقيـات ومـذكـرات التفـاهـم المتعلقة باخـتـصـاصـات الوزارة، بالتنسيق مع الوحدات الإداريـة المعنيـة.
٥- التحقيـق في الوقـائـع والمخالفـات المنسوبـة لموظفي الـوزارة، وإعـداد المذكرات اللازمة بنتائج التحقيـق مـشـفـوعـة بالـرأي القـانـوني والتـوصـيـات، وعـرضـها على السلطة المختصة، ومتابعة تنفيـذ ما يتخذ مـن قـرارات بشأنها
٦- متـابـعـة المنـازعـات والـقـضـايـا الـتـي تـكـون الـوزارة طـرفـاً فيـهـا، بالتنسيق مع الجـهـات المختـصـة
 

 

المادة 16


تختص إدارة العـلاقـات الـعـامـة والاتـصـال بمـا يـلـي:
١- إصـدار النشرات والبـرامـج الإعـلاميـة التي تهدف إلى التعـريـف بـدور الوزارة وأنشطتها واختصاصاتها، بالتنسيق مع الوحدات الإدارية المعنيـة.
۲- رصـد ومـتـابـعـة مـا ينشـر في الصحـف ووسـائـل الإعـلام ومـواقـع التـواصـل الاجتماعي من مـوضـوعـات تتعلق بالوزارة واخـتـصـاصـاتـها، وعـرضـها على المسؤولين، وإعـداد الـردود المنـاسبـة، بالتنسيـق مـع الـوحـدات الإداريـة المعـنـيـة والجـهـات المخـتـصـة.
٣- التنسـيـق مـع الوحدات الإداريـة المختصـة بشـأن إعـداد خطط وبـرامـج التـوعـيـة والتثقيـف، وتطبيقهـا مـن خـلال الفعاليـات والأنشـطـة المختـلفـة بالتنسيق مع الجهات ذات الـعـلاقـة.
٤- تلقـي طـلـبـات وشكاوى المـراجـعـين، وإحالتها إلى الوحدات الإداريـة المعنيـة، والرد على اسـتـفـسـاراتـهـم.
٥- القيام بترتيبات السـفـر والضيـافـة والإقـامـة لضيوف الوزارة وللـوفـود الـزائـرة والمغادرة، بالتنسيـق مـع الوحدات الإداريـة المعـنيـة.
٦- تنظيـم الأنشطـة الثقـافيـة والاجتماعيـة والرياضيـة لموظفي الوزارة، والإشراف على الاحتفالات التي تقيمها أو تشارك فيـهـا الـوزارة.
۷- تنظيـم المـؤتمـرات والندوات والمعارض التي تقيمهـا الـوزارة، وإعـداد الموازنة اللازمة لذلك، بالتنسيق مع الوحدات الإداريـة المعنيـة.
۸- تـرجـمـة الوثـائـق والكتـب والبحوث العلميـة والمكاتبـات المتعلقة باخـتـصـاصـات الوزارة.
۹- اسـتـقـبـال طلبـات كبار السن والأشخاص ذوي الإعـاقـة، بالتنسيـق مـع الوحـــدات الإداريـة المخـتـصـة، لإنجـاز مـعـامـلاتـهـم وتـذليـل أي مشكلات أو صـعـوبـات قـد تـواجـهـهـم.
 

 

المادة 17


تختص إدارة الموارد البـشـريـة بمـا يـلـي:
١- تنفيذ القوانين واللوائح والنـظـم المتـعـلـقـة بشـؤون الموظفين.
٢- إعـداد الدراسـات الخـاصـة بتخطيـط المـوارد البشـريـة بـالـوزارة، وتحـديـد احتياجـات الـوزارة مـن الوظـائـف والموظفين، بالتنسيـق مع الجهـة المخـتـصـة والوحـدات الإداريـة المختـلفـة.
٣- إعداد مشروع وصـف وتصنيـف وترتيب الوظائف، ومتـابعـة تنفيـذه، وتطـويـره.
٤- إعداد مشروع موازنة الباب الأول، بالتنسيق مع الوحدات الإدارية المعـنـيـة.
٥- تنفيذ إجـراءات تقييـم الأداء للمـوظـفـين.
٦- القيام بإجـراءات التـعـيـين والنـقـل والنـدب والإعـارة.
۷- دراسة استحـقـاق الموظفين للإجـازات، وفقاً للـقـانـون.
٨- إعـداد الدراسـات الخـاصـة بالهيكل الوظيـفـي لـلـوزارة، وتقـديـم المقتـرحـات اللازمة لتطـويـر التنظيـم الإداري بـها.
۹- تحـديـد الاحتياجات التـدريبيـة لمـوظـفـي الـوزارة، بالتنسـيـق مـع الـوحـدات الإدارية المختـلـفـة، ومتابعة تنفيذها، وتقييـم مـدى الاسـتـفـادة مـنـها.
۱۰- اتخاذ الإجـراءات اللازمـة لتخصيص السـكـن الحـكـومي وصـرف بـدل الأثـاث لموظفي الوزارة، بالتنسيـق مـع الجـهـات المخـتـصـة.
۱۱- وضـع وتحـديـث قـواعـد البـيـانـات الخـاصـة بشـؤون موظفي الوزارة.

 

المادة 18


تختـص إدارة الشؤون المـاليـة والإداريـة بمـا يـلـي:
۱- تنفيـذ القـوانـيـن والـلـوائـح والنـظـم المـاليـة والإداريـة المتـعـلـقـة بـعـمـل الوزارة.
٢- إعـداد مشـروع الموازنـة السـنـويـة للـوزارة، وحـسـابـها الخـتـامي، بالتنسـيـق مـع الـوحـدات الإداريـة المعـنـيـة.
۳- توفير احتياجـات الـوزارة ووحداتها الإدارية المختلفـة مـن المـسـتـلـزمـات والأجهـزة اللازمة لأداء مـهـامـها، بالتنسيق مع الوحـدات الإداريـة المعنية.
٤- القيام بأعمـال المشـتـريـات والمنـاقـصـات والمزايـدات، وفـق الأنظمـة والقـواعـد المعمـول بـها في الـدولـة.
٥- الإشراف على تنفيـذ الاعتمـادات، وتـدقـيـق حـسـابـات الإيرادات والمصـروفـات.
٦- تـسـلـم وتســليـم وتصنيف وقـيـد البـريـد الـوارد والصـادر.
٧- تنـظـيـم أرشـيـف الوزارة، وحـفـظ الوثـائـق، وفقاً لأحـدث الطرق.
٨- إعـداد مستـنـدات الصـرف، وسـائـر المـعـامـلات الماليـة الأخـرى.
۹- تحـصـيـل الـرسـوم ومقـابـل الخـدمـات التي تؤديها الوزارة.
١٠- القيـام بـجـمـيـع أعـمـال الخـدمـات الإداريـة.
١١- الإشـراف على مـخـازن الوزارة.
 

 

المادة 19


تختص إدارة نـظـم المـعـلـومـات بمـا يـلـي:
١- إعـداد خطط وسيـاسـات اسـتـخـدام الحاسب الآلي في أنشـطـة الـوزارة، ومتـابـعـة تـنـفـيـذهـا.
۲- برمجـة وحـفـظ واستـرجـاع وتـطـويـر نـظـم المـعـلـومـات والبيانات اللازمـة لأنشـطـة الـوزارة.
۳- تـأمـين الأنظمـة والبرمجيـات والشبكات والبنية التحتيـة لنـظم المعـلـومـات مـن مـخـاطـر الحـوادث الإلكتـرونيـة.
٤- إعـداد خطط التعـافـي مـن آثـار الحـوادث الإلـكـتـرونيـة حـال وقـوعـها أو تقليـل حـدوثها أو تصعيـدهـا، وفـق الـسيـاسـات والمبادئ التوجيهيـة لأمـن المعـلـومـات، بالتنسيق مع الجهـة المخـتـصـة.
٥- تـوفـيـر وصيـانـة الأجهـزة والبـرامـج والشبكات الإلـكـتـرونيــة اللازمـة لنـظم العمـل الآليـة بالـوزارة، بالتنسـيـق مـع الوحـدات الإداريـة المعـنيـة.
٦- تصـمـيـم وتشـغـيـل وإدارة قـواعـد البـيـانـات والمعلـومـات بأنـواعـها المختلفة.
۷- تقـديـم الـدعـم الفني اللازم لمسـتـخـدمي الشبكات الإلـكـتـرونيـة بالوزارة، وتدريبهم على استخـدام الأنظمـة والبـرامـج وأجـهـزة الحاسب الآلي وملحـقـاتـها.
٨- إنشاء ومتـابـعـة وتحديث موقع الوزارة على شبكة المعلـومـات الـدوليـة.
 

 

المادة 20


يصـدر بـتـحـديـد اختصـاصـات كـل مـن مـكـتـب الوزير، ومكتـب وكيـل الوزارة، ومكتـب وكيـل الـوزارة المسـاعـد، قـرار مـن الـوزيـر.
 

 

المادة 21


يجـوز بقـرار مـن مـجـلـس الـوزراء، بناءً علـى اقـتـراح الـوزيـر، تعـديـل تنظيـم الـوحـدات الإداريـة الـتـي تـتـألـف منهـا الـوزارة بالإضـافـة أو الحـذف أو الدمج، وتعيين اختصـاصـاتـها وتعـديـلـها.
 

 

المادة 22


يجـوز بقـرار مـن الوزير إنشاء ما تقتضيه المصلحـة الـعـامـة مـن أقسـام في الوحدات الإداريـة التي تتـألـف منهـا الـوزارة، أو إلغاؤهـا أو دمجها، وتعيين اخـتـصـاصـاتـهـا وتـعـديـلها، ولا يـكـون قـرار الوزيـر نـافـذاً إلا بعـد اعتماده مـن مـجـلـس الـوزراء.

 

المادة 23


يُلغى الـقـرار الأمـيـري رقم (۱۰) لسنة ٢٠١٩ المشـار إليـه.
 

 

المادة 24


على جميع الجهات المختصة، كـل فيما يخصه، تنفيذ هذا القـرار. ويُعـمـل بـه مـن تـاريـخ صـدوره. ويُنشـر في الجـريـدة الـرسـميـة.
