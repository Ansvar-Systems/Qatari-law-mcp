مرسوم رقم (39) لسنة 2025 بالتصديق على اتفاقية المساهمة الأساسية بين صندوق قطر للتنمية في دولة قطر ومنظمة الأمم المتحدة للطفولة (اليونيسيف) لدعم الموارد العادية لليونيسيف للعام 2025 و 2026 39 / 2025
عدد المواد: 2
فهرس الموضوعات

المواد (1-2)



نـحـن تمـيـم بـن حـمــد آل ثـانـي              أمـيـر دولـة قـطــر ،
بـعـد الاطـلاع عـلـى الـدستـور،
وعــلى وثـيـقـة التـصـديـق الـصـادرة فـي الـتـاسـع مــن شــهــر رمـضـان عـــام 1446 هـجـريـة ، المـوافـق للـتـاسـع مــن شــهــر مـارس عـــام 2025 ميـلاديـة ،
وعـلى اقـتــراح رئيـس مجـلــس الــوزراء ووزيـــر الخـــارجــية ،
وعـلى مـشـــروع المــرســـوم المـقـــدم مـن مجـلـس الــوزراء ،
 
رسـمنــا بمـا هــو آت :
 

المواد

المادة 1


صُـودق عـلـى اتـفـاقـيـة المـسـاهـمـة الأسـاسـيـة بـين صنـدوق قـطـر للـتـنـمـيـة فـي دولـة قـطـر ومـنـظـمـة الأمـم المـتـحـدة للـطـفـولـة (الـيـونـيـسـيـف) لـدعـم المـوارد الـعـاديـة للـيـونيـسـيـف للـعـام 2025 و 2026 ، المـوقـعـة بمــدينــة نـيـويـورك بتــاريـخ 07/02/2025 ، المـرفـق نـصــهـا بـهـذا المـرسـوم ، وتــكــون لــهـا قــوة الـقـانـون ، وفـقـاً لـلـمـادة (68) مـن الدسـتـور .
 

 

المادة 2


عـلـى جـمـيـع الجـهـات المـخـتـصـة ، كـل فـيـمـا يـخـصـه ، تـنـفـيـذ هـذا المـرســوم . ويُـعـمـل بـه مـن تـاريـخ صــدوره . ويُـنـشــر فـي  الجـريــدة الـرسـمـيـة .
