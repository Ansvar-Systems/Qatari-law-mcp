قرار أميري رقم (33) لسنة 2022 بالهيـكـل التنظيمي لوزارة الخـارجـيـة 33 / 2022
عدد المواد: 38
فهرس الموضوعات

المواد (1-38)



نحن تميم بن حمد آل ثاني أمير دولة قطـر،
بعد الاطلاع على الدستـور،
وعلى قـانـون المـوارد البشـريـة المـدنيـة الصـادر بالـقـانـون رقـم (15) لسنـة 2016، المعدل بالـقـانـون رقم (23) لسنة 2020،
وعلـى الـقـانـون رقـم (17) لسنـة 2021 بشـأن السـلـكـين الـدبـلـومـاسي والقنصلي،
وعلـى القـرار الأميـــــري رقـم (27) لسنـة 2016 بالهيكل التنظيمـي لوزارة الخارجية،
وعلـى القـرار الأميـري رقـم (57) لسنـة 2021 بتعيــن اخـتـصـاصـات الوزارات،
وعلى اقتراح مجلس الوزراء،
قررنا ما يلي:
 

المواد

المادة 1


في تطبيـق أحـكـام هـذا القـرار، تكـون للكلمتـــين والعبـــارات التاليـة المعـاني الموضحـة قـريـن كـل منهـا، مـا لـم يقتـض السيـاق معنی آخـر:
الوزارة: وزارة الخارجيـة.
الـوزيـر: وزير الخارجيـة.
وزيـر الـدولـة: وزيـر الـدولـة للشـؤون الخـارجـيــة.
الأمـين الـعـام: الأمـين الـعـام لـوزارة الخـارجـيـة.
البعثة التمثيـلـيـة: إحـدى بـعـثـات التمثيـل الدبلـومـاسـي أو القنـصـلي للدولة في الخـارج.
البعثات الدبلوماسية المعتمدة لدى الدولة: البعـثـات الـدبـلـومـاسـيـة الأجنبيـة المعتمدة لدى الدولة.
الموظفين المحليين: الموظفين في البعـثـات التمثيليـة أو البعـثـات الدبلوماسية المعتـمـدة لـدى الدولـة مـن غيـر أعضاء السلكين.
 

 

المادة 2


يتكـون الهيكـل التنظيمـي لـوزارة الخارجيـة مـن الوحـدات الإداريـة المبينة بالخريـطـة التنظيميـة المـرفـقـة بهذا القـرار، وهي:
أولاً: الوحـدات الإداريـة التـابـعـة للـوزيـر:
۱- مـكـتـب الـوزيـر.
۲- المـكـتـب الفـنـي.
٣- إدارة تخطيط السيـاسـات.
٤- إدارة التـخـطـيـط والجـودة والابتكار.
٥- إدارة الـرقـابـة والتـدقـيـق.
٦- إدارة الشـؤون القـانـونيـة.
ثانياً: الوحدات الإداريـة التـابـعـة لـوزيـر الـدولـة:
۱- مـكـتـب وزيـر الـدولـة.
 
 
 
ثالثاً: الوحدات الإداريـة التـابـعـة للأمـين الـعـام:
۱- مـكـتـب الأمـين الـعـام.
٢- إدارة شؤون مجـلـس الـتـعـاون.
٣- إدارة الشـؤون الـعـربيـة.
٤- إدارة الشـؤون الآسـيـويـة.
٥- إدارة الشؤون الأفريقية.
٦- إدارة الشؤون الأوروبية.
٧- إدارة الشؤون الأمـريـكـيـة.
٨- إدارة المنـظـمـات الـدوليـة.
٩- إدارة التـعـاون الدولي.
۱۰- إدارة حقوق الإنـسـان.
١١- إدارة الشـؤون الـقـنـصـلـيـة.
۱۲- إدارة المـراســم.
١٣- إدارة الموارد البـشـريـة.
١٤- إدارة الشؤون المـالـيـة والإداريـة.
١٥- إدارة نـظـم المـعـلـومـات.
١٦- إدارة الشـؤون الـهـنـدسيـة والخـدمـات الـعـامـة.
١٧- إدارة الـوثـائـق والمراسلات.
١٨- إدارة الإعـلام والاتـصـال.
۱۹- المعـهـد الـدبـلـومـاسـي.
٢٠- إدارة الشؤون الأمـنيـة.
٢١- إدارة الـعـمـلـيـات.
۲۲- اللجـنـة الـدائـمـة لتنـظيـم المـؤتمـرات.
 

 

المادة 3


يتولى وزير الدولـة، تحت إشـراف الـوزيـر، تسـيـيـر كـافـة أعمـال الوزارة، وذلك في إطـار السيـاسـة العـامـة للـوزارة.

 

المادة 4


يـكـون لـلـوزيـر مـسـاعـد أو أكثـر، يصدر بتعييـنـهـم قـرار مـن الأميـر، وتحدد اخـتـصـاصـاتـهـم بـقـرار مـن الـوزيـر، ويكـون لـكـل مـسـاعـد وزير، مـكـتـب يتبـع لـه

 

المادة 5


يـكـون لـلـوزارة متـحـدث رسمي، يصدر بتعييـنـه وتحـديـد اخـتـصـاصـاتـه قـرار مـن الـوزيـر.

 

المادة 6


يـكـون لـلـوزيـر مبـعـوث أو أكثـر، يصـدر بتعيينـهم وتحـديـد اخـتـصـاصـاتـهـم قـرار مـن الـوزيـر.

 

المادة 7


يختـص المكتـب الفـنـي بمـا يـلـي:
١- إبـداء الـرأي الفـني في الموضـوعـات التي تـُحـال إليـه مـن الـوزيـر.
۲- دراسـة وتقييم التقاريـر والمشـروعـات والبـرامـج المتعلقـة بأنشـطة الوزارة، التي تُحـال إليـه مـن الوزيـر.
۳- بحث ودراسـة ومـتـابـعـة الموضـوعـات الـتـي تـُحـال إليـه مـن الوزيـر، بالتنـسـيـق مـع الـوحـدات الإداريـة المعـنيـة.
٤- إعـداد ملخـص للمـواضـيـع والـتـقـاريـر والدراسـات الـتـي يـلـزم عـرضـها على الـوزيـر، وإبداء الرأي بشـأنـها.

 

المادة 8


تختـص إدارة تخطيـط الـسـيـاسـات بمـا يـلـي:
۱- اقتـــراح السيـاسـات الخارجيـة المنـاسبـة للدولـة مـن خـلال متـابعـة ومـراجـعـة وتحـديـث الـتـوجـهات الاستراتيجيـة للدولـة، ووضـع تصـور لـهـا، بالتنسيق مـع الوحدات الإداريـة والجـهـات المخـتـصـة ومـراكـز الأبحـاث والـدراسـات.
۲- تنسيـق الـسيـاسـات الخـارجيـة للدولـة مـن خـلال الاجـتـمـاعـات التنسيقيـة والحـوارات الاستراتيجيـة واللجـان المشتـركة، وتقـديـم التوصيـات، والمشاركة في الاجتماعـات بـهـذا الشـأن.
۳- بحـث ودراسـة وتحليـل المسائل السيـاسـيـة والاقتصـاديـة والعـلاقـات الدوليـة الخـاصـة بـاخـتـصـاصـات الـوزارة، بالتنسيق مـع الـوحـدات الإداريـة المعنيـة.
٤- تحليـل الأحـداث المحلية والدوليـة ذات الأهميـة، واستخـلاص النتائج اللازمة، وتقـديـم المقترحات بشـأنـهـا وعـرضـها على الـوزيـر.
٥- تحـديـد وتقييـم المخـاطـر المتـوقـع حـدوثها في مجـالات الأنشـطة الـدوليـة المتعلقة بالسيـاسـة والاقـتـصـاد.
٦- متـابـعـة تـنـفـيـذ الـقـرارات السياسيـة والاتفاقيـات الدوليـة الصـادرة، بالتنـسيـق مـع الجـهـات المعنيـة.
۷- تلخيـص وتقييـم التقارير والدراسـات السيـاسـيـة الـتـي تـرد لـلـوزيـر، وإبـداء الـرأي بشـأنـها.
٨- إنشـاء قـاعـدة بـيـانـات خـاصـة بالشـؤون السيـاسـيـة والـعـلاقـات الدوليـة بالتنسيق مع الوحـدات الإداريـة المختصة.
 

 

المادة 9


تختص إدارة التخطيط والجـودة والابتـكـار بـمـا يـلـي:
١- إعداد الخطة الاستراتيجيـة للوزارة، بالتنسيق مع الجهات المختصة والوحدات الإداريـة المعـنيـة، واتخاذ إجراءات اعـتـمـادهـا.
۲- حـصــر التحـديـات القـائمـة، وتحـديـد وتقييم المخاطـر المؤسسيـة المتـوقـع حـدوثـها، والتي قد تؤثـر علـى تـحـقـيـق أهـداف الوزارة، وتقـديـم المقترحـات بـشـأنها، بالتنسيـق مـع الـوحـدات الإداريـة
المعنيـة.
٣- إدارة وتجـهـيـز خـطـط بـديـلـة تضمن استمراريـة تشـغيـل الأعمـال بالــــوزارة في حـالات الطوارئ والأزمـات، بالتنسيـق مـع الـوحـدات الإداريـة المختـلفـة كـل فيـما يخـصـها.
٤- رفـع مـشـروعـات الخطط التنفيـذيـة والمشـاريـع والبـرامـج الخـاصـة بالـوحـدات الإدارية المختـلفـة، للاعتمـاد مـن قبـل الـوزيـر.
٥- متـابعـة تنفيذ الخطط المتعلقة بأنشـطـة الـوزارة، وتقييـم نـتـائـجـها، ورفـع تـقـاريـر دوريـة بشـأنـها.
٦- دراسـة المشـاكـل والمعـوقـات الـتـي تـواجـه تنفيذ الخطـة الاستراتيجيـة والخطط التنفيـذيـة، واقتراح الحلول المناسبة لها.
۷- تقييـم جـودة الخـدمـات المقـدمـة مـن الـوحـدات الإداريـة، بالتنسيـق مع الجهـة المختصة.
۸- مـراجـعـة وتقييم جودة الأداء على مستـوى الوحدات الإداريـة، ووضـع مـقـتـرحـات لـرفـع الـكفـاءة العـامـة بـها.
 
 
 
 
٩- التـحـقـق مـن انسجـام نـظـم وأساليـب العـمـل مـع اختصـاصـات وأهـداف الوزارة، واقتراح خطـط تـطـويـر تـلـك النـظـم والأساليب، وتبسيط الإجراءات، بالتنسيق مع الوحدات الإدارية المعنية والجهة المختصة.
۱۰- دراسـة مـشـاكل ومـعـوقـات العـمـل بالـوزارة، وبحـث أسبـابـها، واقـتـراح الحلول المنـاسبـة لـها.
۱۱- تلقـي المبــادرات والمقترحـات والأفكار الإبـداعـيـة، ودراستها بالتنسيق مع الوحدات الإدارية المعنيـة والجـهـة المختصة.
۱۲- رصـد وتجميـع وتحليـل البيـانـات الإحصائيـة الخـاصـة بأنشـطـة الوزارة، والتنسيق مع الوحدات الإداريـة المختصـة بشـأن التقاريـر الإحصائيـة المطـلـوبـة مـن الـوزارة.
۱۳- التنسيق مع الوحدات الإداريـة المعنيـة والجهـة المختصة فيما يتعلـق بالمؤشـرات الـدوليـة المتعلقة بأنشـطـة واختصـاصـات الوزارة.
 

 

المادة 10


تختص إدارة الـرقـابـة والتـدقـيـق بمـا يـلـي:
۱- وضـع مـشـروع خـطـة التـدقيـق السنـويـة علـى الوحدات الإداريـة بالوزارة والبعثات التمثيـليـة، ورفعـهـا للوزير لاعتمـادهـا، وإعـداد تـقـاريـر بـنـتـائـج التـدقـيـق.
۲- مـراقـبـة مدى الالتزام بتنفيذ القـوانـين واللوائح والقرارات المتعلقة بأنشـطـة الـوزارة.
۳- مـراقـبـة حـسـابـات الوحدات الإداريـة والبعثات التمثيليـة، والتحقق مـن أن التصـرفـات المـاليـة والقيود الحـسـابيـة الخـاصـة بهـا قـد تمـت وفقاً للقـوانـين والقـواعـد المعـمـول بـها.
٤- مـراقـبـة إجـراءات تعيين الموظفين، وإجازاتـهم، وتـرقـيـاتـهم، وغيـر ذلـك ممـا يـتـعـلـق بـشـؤون الموظفين، والتحـقـق مـن الالتـزام بالقوانين واللوائح والقرارات المنظمة لذلك.
٥- كشف حـالات الإهمال والمخالفات والتجـاوزات الماليـة والإداريـة، وبحـث الأسبـاب الـتي أدت إلى وقـوعـها، واقتـــراح الوسـائـل الكفيلة بعلاجها.
٦- مـراجـعـة عمليات المتحصلات القنصليـة، وإجـراء الجـرد النقـدي اللازم لهـا، وأيـة أوراق أخـرى ذات قيمـة مـاليـة، والتحقـق مـن مـدى مـطـابـقـتـهـا للقيـود والسجـلات.
۷- مـراجـعـة التعليمـات الماليـة، واقـتـراح مـا يـلـزم مـن تـعـديـلات عليها، ومـراقـبـة تنفيذها.
۸- مـراقـبـة انضبـاط سيـر المجـريـات المـاليـة والإداريـة، وإعـداد تقـاريـر عـن الزيـارات الـرقـابيـة الـتـي تقـوم بـهـا، ورفعها للسلطة المعنيـة في الوزارة.
٩- التـأكـد مـن سـلامـة الأمـوال والعـهـد الـتـي تمتـلـكـهـا الـوزارة أو تخضع لإشـرافـها.
۱۰- الاشتراك في جـرد مـحـتـويـات المخـازن والعـهـد الأخـرى
 

 

المادة 11


تختص إدارة الشؤون القـانـونيـة بمـا يـلـي:
١- إبـداء الـرأي القـانـوني في الأمور والمسـائـل المحـالـة إليـها.
٢- إعـداد وصيـاغـة مـشـروعـات الأدوات التشريعيـة الخـاصـة بالـوزارة، وإبـداء الـرأي في مشـروعـات الأدوات التشريعيـة المحـالـة إلى الوزارة.
۳- دراسـة مـشـروعـات المعـاهـدات والاتفـاقيـات الدوليـة الـتـي تـرغـب الدولة أن تكـون طـرفـاً فيهـا، وإبـداء الـرأي القـانـونـي بشـأن التـوقيـع عليهـا أو الانضـمـام إليـهـا، بالتنسيـق مـع الوحدات الإداريـة المعنيـة والجـهـات المختصة.
٤- إبـداء الـرأي في المعـاهـدات والاتفاقيـات الدوليـة مـن حيـث تجـديـدهـا، أو مـدهـا، أو إنـهـاؤها، ومـا يـطـرأ عليـهـا مـن تـعـديـلات، بالتنسيـق مع الوحدات الإدارية المعنيـة والجـهـات المختصة.
٥- تـولي ومتـابـعـة الأمـور المتعلقة بالمحاكم الإقليميـة والدوليـة، ومتابعـة أعمـال اللجـان القـانـونيـة المتخصصـة ومـا يـصـدر عنهـا مـن قـرارات في المجالات الإقليمية والدوليـة، وبحـث مـدى اتفـاقـهـا مـع قـوانـين الدولة ومصالحـها الـعـلـيـا.
٦- اتخاذ الإجراءات اللازمة لإبرام المعـاهـدات والاتفاقيـات الـتـي تـكـون الدولـة طـرفـاً فيهـا، والتصديق عليهـا ونشـرهـا وتسجيلها أو إيداعـها لدى الهيئات الإقليميـة والدوليـة المختصـة، بالتنسيق مـع الوحـدات الإداريـة المعـنيـة والجـهـات المختصة.
٧- إعـداد الملفـات المتعلقة بالاتفاقيـات والمعـاهـدات الـتـي تكـون الدولـة طرفاً فيها وحفظها، بالتنسيـق مـع الجـهـات المختصة.
۸- دراسـة القضـايـا والمـوضـوعـات ذات الطبيـعـة القـانـونيـة المتعلقـة بمصالح الدولة في الخارج وبعثـاتـها التمثيليـة، والتـوجـيـه بمـا يـلـزم اتبـاعـه لحمايتها، بالتنسيق مع الجهات المختصة في الدولة.
٩- إعـداد ومـراجـعـة وصيـاغـة مـشـروعـات العقـود الـتـي تبـرمـهـا الـوزارة وبعثـاتـها التمثيلية، وإحالتها إلى الجهـات المختصة لمـراجـعـتـها، ومتـابـعـة مـا يـطـرأ مـن مشكلات تعترض تنفيـذهـا.
۱۰- التحقيـق في الوقائع والمخالفـات المنسـوبـة لموظفي الوزارة المحـالـة إليهـا، وفقاً للأدوات التشريعيـة المعمـول بـهـا، وإعـداد المذكـرات اللازمـة بنتائج التحقيـق مـشـفـوعـة بـالـرأي الـقـانـوني، ومـتـابـعـة تنفيذ ما يُتخذ بشـأنـها مـن قـرارات.
۱۱- معالجـة الجـوانـب القـانـونيـة المتعلقة بجميع المعـاهـدات والاتفاقيـات ومذكرات التفاهم والبـروتـوكـولات واللوائح الدوليـة الـتـي تـربـط الدولـة بالخــارج، أو يمكن أن تلـزمـها، والحرص علـى استيــفاء الإجـراءات اللازمـة للموافقة عليهـا وتـصـديقـها وقبولها أو الانضمام إليها ونشرها.
۱۲- النظـر فـي النـزاعـات الدبلــومـاسـيـة والقنصليـة وجـمـيـع الجـوانـب القـانـونيـة المرتبطة بالأعـمـال الخـارجـيـة التي تقـوم بـهـا الـوزارة.
١٣- إعـداد مشـروعـات الاتفاقيـات الـتي يتقـرر عقـدهـا مـع الـدول العـربـيـة والأجنبيـة، بالتنسيق مع الوحدات الإدارية المعنيـة.
١٤- حفظ نسـخ مـن جميـع المعـاهـدات والوثائق الدبلوماسية المتعلقة بها، ووثـائـق التصـديـق أو القبول أو الانضمام المتعلقة بجميع المعاهدات الدوليـة التي تكون الدولـة طـرفـاً فيـها.
١٥- متـابـعـة المنـازعـات والقضـايـا الـتـي تـكـون الـوزارة طـرفـاً فيهـا، بالتنسيق مع الجهات المختصة .
 

 

المادة 12


تختص إدارة شؤون مجـلـس التـعـاون بمـا يـلـي:
١- العمـل عـلـى تـنـميـة وتطـويـر العـلاقـات الثنائيـة بـين الدولـة ودول مجلس التعاون لدول الخليـج الـعـربـيـة، والعمـل عـلـى تـحـقـيـق أهـداف المجلـس في المجـالات السيـاسـيـة والاقـتـصـاديـة والثقـافيـة والفـنـيـة وغيـرهـا، ومتـابـعـة مـا يـحـدث من تطورات في هـذه الـدول.
۲- دراسـة التقارير الواردة مـن الأمـانـة العـامـة لمجلس التعاون، وإبـداء الرأي بشـأنـها، وإحـالـتـها إلى جهة الاختصـاص.
۳- دراسة وتحليـل التـقـاريـر والأبحـاث الـواردة مـن البعثات التمثيليـة في دول المجلس، واتخاذ مـا يـلـزم بشـأنـها.
٤- دراسـة ومـتـابـعـة مـا يـُحـال إليـهـا مـن أمـور ومـوضـوعـات وتقاريـر وأبحـاث تتعلـق بالأوضـاع والتطــورات والعـلاقـات بــين الدولـة ودول مجلـس التـعـاون.
٥- دراسـة وتحليـل وتقيـيـم القـرارات والتـوصـيـات الـتي تصـدر عـن المجلـس وأجـهـزتـه ولجـانـه المختلفـة، واقتراح أسـلـوب تنفيـذهـا ومتابعتـها، بالتنسيق مـع الـوزارات والأجهـزة الحكوميـة الأخـرى المختصة.
٦- التنسيـق مـع الأمـانـة العـامـة لمجلـس التـعـاون والأجهـزة المعنيـة في الدولـة بشـأن اجتمـاعـات القمـة والمجـالـس الوزاريـة واللجـان، وتنظيــم مشـاركة الدولـة في هـذه الاجتمـاعـات، وإعـداد الدراسـات والموضـوعـات والتقارير التي تتطلبها هـذه المشـاركـة.
٧- التشـاور والتنسيق مع الجهـات المختصة في الدولـة حـول مـا يقـع في نطـاق اختصاص هـذه الجهـات مـن مـوضـوعـات ذات طبيعـة اقتصـاديـة أو ثقـافيـة أو فنيـة أو غيـرهـا، تُحـال إلى الـوزارة مـن قبـل دول المجلـس أو مـن الأمـانـة العـامـة أو المؤسسات أو اللجـان الخـلـيـجـيـة المشتركة، وذلك لدراستها وإبـداء الـرأي بشـأنـها.
۸- المشـاركـة في اجـتـمـاعـات القمـة والمجـالـس الوزاريـة واللجـان الخليجية المشتـركـة.
٩- إعداد الدراسات والأبحاث والتـقـاريـر الـلازمـة لـدعـم العـمـل المشترك وتطـويـر التعاون مع دول مجلـس التعـاون، بالتنسيق مع الجهـات المختصة بالـدولـة.
١٠- التنسيـق مـع إدارة الشـؤون الـقـانـونيـة بشـأن إعـداد مشـروعـات الاتفاقيات التي يتقرر عقـدهـا مع دول مجلـس التـعـاون.
۱۱- متـابعـة وتـوجـيـه نشـاط البعثات التمثيلية في دول مجلس التعـاون.
۱۲- تنسيق ومتـابـعـة وتطـويـر العـلاقـات مـع البعثات الدبلــومـاسيـة لدول مجلس التعاون المعتمدة لدى الـدولـة.
۱۳- متـابـعـة مـا يُنشر في صـحـف دول مجلـس التعـاون عـن كـل مـا يـهـم الدولـة، وتحـلـيـلـه وإبـداء الـرأي حـول مـا يـلـزم اتخاذه مـن ترتيبـات أو توصيات بشـأنـه، بالتنسيق مع إدارة الإعلام والاتصال بالوزارة.
١٤- متـابـعـة مـا يـصـدر عـن البعثات الدبلوماسيـة لـدول مجلس التعـاون المعتمدة لدى الدولـة مـن نـشـرات أو مـطـبـوعـات، وإبـداء مـا قـد يـلـزم بشـأنها مـن آراء أو تعليقات، بالتنسيق مع إدارة الإعـلام والاتصـال بالوزارة.
١٥- الإعـداد والمتـابـعـة والتنسيق لزيـارات وفـود الدولـة الرسميـة العليـا لدول مجلس التعاون، ومـا يـُطـلـب منـهـا مـن تقـاريـر بـشـأنـها.
١٦- الإعـداد والمتـابعـة والتنسيق للزيـارات الرسميـة الـتـي تقـوم بـهـا وفـود عـليـا مـن دول مجلـس التعـاون للدولـة، ومـا يـُطـلـب منهـا مـن تـقـاريـر بـشـأنـها.
۱۷- التحضيـر لأعمـال اللجـان المشتـركـة مـع دول مجلـس التعـاون والمشـاركـة فيها، ومتـابعـة تنفيـذ مـا يـصـدر عنـهـا مـن قـرارات.
١٨- إعداد التقارير اللازمـة حـول علاقات الدولة مع دول مجلس التعاون، وتحـديـث المـعـلـومـات الواردة بتلك التـقـاريـر.
١٩- إعـداد تقاريـر دوريـة عـن نـشـاط البعثات التمثيليـة في دول مجلـس التعـاون.
 

 

المادة 13


تختـص إدارة الـشـؤون الـعـربـيـة بما يلي:
١- العمـل على تنميـة وتطـويـر العـلاقـات الثنائيـة بين الدولـة والـدول العـربـيـة في المجالات السياسيـة والاقـتـصـاديـة والثقـافـيـة والفنيـة وغيـرهـا، ومتـابـعـة مـا يـحـدث مـن تـطـورات في هـذه الـدول.
۲- دراسـة التقارير الواردة مـن الأمـانـة العـامـة لجـامـعـة الـدول العـربـيـة، وإبـداء الـرأي بشـأنـها، وإحالتها إلى جـهـة الاخـتـصـاص.
۳- دراسـة وتحليـل التقاريـر والأبحـاث الـواردة مـن البعثات التمثيليـة في الدول العـربـيـة، واتخاذ مـا يـلـزم بشـأنـها.
٤- دراسـة ومـتـابـعـة مـا يـُحـال إليهـا مـن أمـور ومـوضـوعـات وتقاريـر وأبحـاث تتعلـق بالأوضاع والتطــورات والعـلاقـات بـين الدولـة والدول العـربـيـة الأخـرى.
٥- التنسيق مـع جـامـعـة الـدول العـربـيـة والأجـهـزة المعنيـة في الدولـة بشـأن اجتمـاعـات القمـة والمجـالس الوزاريـة المختلفـة والمنظمـات المخـتـصـة واللجـان، وإبـداء الـرأي في كـل مـا يـُحـال إليهـا مـن مـوضـوعـات تتعلق بتنميـة وتطـويـر التعاون مع جامعة الدول العربيـة.
٦- المشاركة في اجـتـمـاعـات القمـة والمجـالـس الـوزاريـة والمنظمات المختصة واللجان المشار إليها في البنـد الـسـابـق.
۷- متـابـعـة وتـوجـيـه نشـاط البعثات التمثيـليـة في الدول العربيــة.
۸- تنسيـق ومـتـابـعـة وتطـويـر العـلاقـات مـع البعثات الدبلـومـاسـيـة للدول العـربـيـة المعتمـدة لـدى الـدولـة.
۹- دراسـة ومـتـابـعـة مـا يُنشر في صـحـف الـدول العـربـيـة عـن كـل مـا يـهـم الدولـة، وتحـلـيـلـه وإبـداء الـرأي حـول مـا يـلـزم اتخاذه مـن ترتيبـات أو توصيات بشـأنـه، بالتنسيق مع إدارة الإعلام والاتصال بالوزارة.
۱۰- متـابعـة مـا يـصـدر عـن البعثات الدبلـومـاسـيـة للـدول العـربـيـة المعتمـدة لـدى الدولـة مـن نـشـرات أو مطبـوعـات، وإبـداء مـا قـد يـلـزم بشـأنها من آراء أو تعليقات، بالتنسيق مع إدارة الإعـلام والاتصال بالوزارة.
۱۱- متـابعـة تـنـفـيـذ مـا يـصـدر عـن جـامـعـة الـدول العـربـيـة ومنظماتـها المتخصصـة مـن قـرارات تـكـون الدولـة قـد أقـرتـها، وإعـداد التقاريـر بشأن مـدى التـقـدم في تنفيذ هذه القـرارات.
۱۲- الإعـداد والمتـابـعـة والتنسيق لزيـارات وفـود الدولـة الرسميـة العليـا للدول العـربـيـة، ومـا يـُطـلـب منـها مـن تقـاريـر بـشـأنـها.
١٣- الإعـداد والمتـابعـة والتنسيـق للزيارات الرسميـة الـتـي تقـوم بـهـا وفـود عليا من الدول العـربـيـة للدولة، وما يـُطـلـب منها من تقاريـر بـشـأنـها.
١٤- التحضير لأعمال اللجـان المشتركة مـع الـدول العـربـيـة، والمشـاركـة فيها، ومتـابـعـة تنفيـذ مـا يـصـدر عنـهـا مـن قـرارات.
١٥- التنسيق مـع إدارة الشـؤون الـقـانـونيـة بـشـأن إعـداد مشـروعـات الاتفاقيات التي يتـقـرر عـقـدهـا مع الدول العـربـيـة.
١٦- إعـداد التـقـاريـر اللازمـة حـول عـلاقـات الدولـة مـع الـدول العـربـيـة، وتحـديـث المـعـلـومـات الواردة بتلك التـقـاريـر.
١٧- إعـداد تقاريـر دوريــة عـن نـشـاط البعثـات التمثيلية في الـدول العـربيـة.
 

 

المادة 14


تختـص إدارة الشؤون الآسـيـويـة بمـا يـلـي:
١- العمـل علـى تنميـة وتطـويـر العـلاقـات الثنائيـة بين الدولـة والـدول الآسيـويـة في المجالات السياسيـة والاقـتـصـاديـة والثقـافـيـة والفنيـة وغيـرهـا، ومتـابـعـة مـا يـحـدث مـن تـطـورات في هـذه الـدول.
۲ - دراسة وتحليـل التـقـاريـر والأبحـاث الـواردة مـن البعثات التمثيليـة في الدول الآسيـويـة، واتخاذ مـا يـلـزم بشـأنـها.
۳- دراسـة ومـتـابـعـة مـا يُـحـال إليـهـا مـن أمـور ومـوضـوعـات وتقاريـر وأبحـاث تتعلق بالأوضاع والتطورات والعـلاقـات بـين الدولـة والـدول الآسيـويـة.
٤- متـابـعـة وتـوجـيـه نشـاط البعثات التمثيـليـة في الدول الآسيـويـة.
٥- تنسيـق ومـتـابـعـة وتطـويـر العلاقـات مـع البعثات الدبلوماسيـة للـدول الآسيـويـة المعـتـمـدة لـدى الدولة.
٦- متـابـعـة مـا يُنشـر في صحـف الـدول الآسيـويـة عـن كـل مـا يـهـم الدولـة، وتحـلـيـلـه وإبـداء الـرأي حـول مـا يـلـزم اتخاذه مـن ترتيبـات أو توصيات، بالتنسيق مع إدارة الإعـلام والاتصـال بالوزارة.
۷- متـابعـة مـا يـصـدر عـن البعثات الدبلـومـاسـيـة للـدول الآسيـويـة المعتمدة لدى الدولـة مـن نـشـرات أو مطبـوعـات، وإبـداء مـا قـد يـلـزم بشـأنها مـن آراء أو تعليقات، بالتنسيق مع إدارة الإعـلام والاتصـال بالوزارة.
٨- التنسيق مع المنظمات الإقليميـة الآسـيـويـة والجهات المعنية في الدولة، وإبـداء الـرأي في كـل مـا يُـحـال إليـهـا مـن مـوضـوعـات تتعلق بالتعـاون مع هذه المنـظـمـات، بالتنسيق مع إدارة المنظمات الدوليـة بالوزارة.

المادة 18


تختص إدارة المنـظـمـات الـدولـيـة بمـا يـلـي:
١- العمـل علـى تنميـة عـلاقـات الدولـة مـع منظمـة الأمـم المـتحـدة ووكالاتها المتخصصة والمنظمات الدوليـة الأخـرى، في ضوء الأهـداف والمصالح العـلـيـا للـدولـة، وذلك بالتنسيق والتـشـاور مـع الجـهـات المختصة.
۲- متـابـعـة نشـاط منظمــة الأمـم المتحـدة ووكالاتهـا المتخصصة والمؤتمرات والاجتمـاعـات الدوليـة، وذلـك مـن خـلال المشـاركة فيهـا أو متـابـعـة وقـائـع أعمـالـها، وتزويـد الوفـود الدائمـة أو الوفـود المشاركة فيهـا بالتعليمـات الصـادرة عـن الدولـة، وفقاً لما تقتضيـه مصلحة الدولـة.
۳- متـابعـة وتـوجـيـه نشـاط الـوفـود الدائمـة للدولة، ودراسـة مـا تـرفـعـه مـن تقـاريـر حـول أعمـالـها، واتخاذ مـا يـلـزم بـشـأنـها.
٤- التنسيق والتعـاون مـع وزارات الدولـة وأجهـزتـها المختصـة بشـأن المشاركة في المنظمات والمؤتمرات والاجتمـاعـات الـدوليـة.
٥- دراسـة وتحـلـيـل مـا يـصـدر عـن منظمـة الأمـم المتحـدة والمنظمـات والمؤتمرات الدوليـة مـن قـرارات أو تـوصـيـات تـكـون الدولـة قـد وافقـت عليـهـا أو أقـرتـها، وإعـداد التقارير اللازمـة بشـأن تنفيـذ هذه القرارات أو التـوصـيـات، بالتعاون مع الجهات المختصة بالـدولـة.
٦- تـقـديـم الـدعـم والإشراف عـلى مـقـر بـيـت الأمـم المـتـحـدة فـي الـدوحـة مـن النـاحـيـة الإدارية واللوجستية، بالتنسيق مع الجهات المعنية في الدولة.
۷- اقتراح الترشيحات لعضـويـة المنظمات الدوليـة واستـضـافـة مـقـارهـا ومتابعتـها، بالتنسيـق مـع الجـهـات المختصة.
۸- متـابعـة تـرشـيـحـات الـدول والأفراد للمنـاصـب الدوليـة، بالتنسيق مع الجهات المختصة في الـدولـة.
۹- تـقـديـم المقـتـرحـات والتوصيـات اللازمـة بشـأن الاستفـادة مـن بـرامـج وأنشـطـة المنظمات والهيئات العـربـيـة والإقليميـة والـدوليـة.
 

 

المادة 19


تختص إدارة الـتـعـاون الـدولـي بمـا يـلـي:
١- التنسيق والتعـاون مـع الجهـات المختصة في الدولـة بشـأن تعـزيـز كـافـة مجالات التعاون الفني، بين الـدولـة والـدول الأخـرى.
۲- تحـديـد احتيـاجـات الجـهـات الحـكـومـيـة في الدولـة مـن أنشـطة التعـاون الفـنـي المقـدمـة مـن الأمـم المـتحـدة ووكالاتهـا المتخصصة والمنظمـات العـربـيـة والإقليميـة والدوليـة الأخـرى، وفقاً للخطـط والبـرامـج المـوضـوعـة، واتخاذ إجـراءات تنفيـذهـا.
۳- دراسـة وتحليـل أنشـطـة التعـاون الفـنـي الـتـي تقـدمـها المنظمـات والهيئات الدوليـة في مجـال الخـبـرات والاستشـارات، والتأهيـل والتـدريـب، والمـعـلـومـات والندوات والمؤتمرات، والمطبـوعـات
والنشـرات والتقاريـر والإحصائيـات، والدراسات والبحـوث، ونظـم وبـرامـج المعـلـومـات، وتحـديـد سـبـل الاسـتـفـادة مـنـها.
٤- متـابعـة بــرامج وأنشـطة المنظمات والهيئـات العـربيــة والإقليميـة والدولية ذات الطابع الفني، ومـا يـصـدر عنهـا مـن قـرارات وتوصيـات، وإعـداد خطـط العمـل للاسـتـفـادة مـن هـذه البـرامـج والأنشـطة بالتعاون مع الجهات المختصة.
٥- إعـداد الدراسات والبحـوث ذات العـلاقـة بأنشـطـة التعـاون الفـني مع المنظمات والهيئـات العـربـيـة والإقليميـة والدوليـة، بالتنسيـق مع الجهات الحكومية في الـدولـة.
٦- تنظيم ومتـابـعـة سـبـل الاسـتـفـادة مـن المعـونـات الفنيـة الـتي تقـدمـها منظمـة الأمـم المـتحـدة والوكالات والمنظمات المتخصصـة بالتنسيق مع الجهات المعنية بالدولـة، وتنـظـيـم مسـاهـمـة الدولـة فيما يـُطـلـب مـن مـهـام أو ترتيبـات دوليـة.
٧- إعـداد التقـريـر السـنـوي الخـاص بأنشـطـة التعـاون الفـني للجهـات الحكومية في الدولة مع المنظمات والهيئـات الـدوليـة.
٨- المشاركة في إعـداد سـيـاسـة الدولـة في مجـال المسـاعـدات والمعـونـات الإنمائيـة والإنسـانيـة لـدعـم التنميـة المستـدامـة في الـدول الناميـة، بما يتـفـق مـع أولـويـات الدولـة والقرارات والتوصيات الدوليـة بـهـذا الشـأن، وذلك بالتنسيـق مـع الجـهـات المختصة.
۹- حـصـر وتسـجيـل المسـاعــدات الإنمائيـة المقـدمـة مـن الدولـة للخـارج، وتحـديـثـها سـنـويـاً، بالتنسيـق مـع الجـهـات المختصة.
١٠- إعـداد التـقـاريـر والدراسات والبحـوث الخاصـة بالمسـاعــدات الإنمائيـة والإنسانيـة المقـدمـة مـن الدولـة للـدول الأخـرى، وتوثيقها لدى المنظمات والهيئـات الـدوليـة.
۱۱- تقديم الاستشـارات الفنيـة لأجهـزة الدولـة والمنظمات غيـر الحـكـوميـة المعـنيـة بـتـقـديـم المسـاعـدات الإنمـائيـة والإنسـانيـة.
۱۲- متـابعـة أعمـال اللجـان المشتركة في مجـال التعـاون الـدولي، بالتنسـيـق مـع الوحـدات الإداريـة المعنيـة في الـوزارة، والجـهـات المختصة بالـدولـة.
 

 

المادة 20


تختص إدارة حـقـوق الإنـسـان بمـا يـلـي:
۱- تقـديـم الـرأي والمشورة في المسائل المتعلقة بحـقـوق الإنسـان الـتـي تُحـال إليـها.
٢- إبـداء الـرأي في مشـروعـات الاتفاقيـات الدوليـة والاتفاقيـات القـائـمـة المتعلقة بحقوق الإنسان، الـتـي تـرغـب الدولـة أن تكـون طرفاً فيهـا، وذلك بالتنسيق مع إدارة الشـؤون الـقـانـونيـة بـالـوزارة
والجهات المختصة بالـدولـة.
٣- المشاركة في إعـداد التقاريـر الـتـي تـعـدهـا الدولـة عـن حـقـوق الإنسـان بموجب الاتفاقيـات الدوليـة، وتقـديمـها لهيئـات الـرصــد الدوليـة المعنيـة، بالتنسيق مع إدارة الشـؤون الـقـانـونـيـة بالوزارة.
٤- إعداد الدراسات والبحوث والتقاريـر ذات الـعـلاقـة بحقوق الإنسـان بالتعاون مع الجهات المختصة في الـدولـة.
٥- إعـداد الـردود المنـاسـبـة علـى تـقـاريـر المنظمات الدوليـة الحـكـوميـة والمنظمات غيـر الحـكـوميـة حـول أوضاع حقوق الإنسان في الدولة، بالتنسيق مع الجهات المختصـة، وإرسالها لتلك المنـظـمـات.
٦- إعـداد الـردود علـى تـقـاريـر الحـكـومـات الأجنبيـة عـن أحـوال حقـوق الإنسان في الدولـة، بالتنسيق مـع الجـهـات المختصة، وإبـلاغـها لتلك الحـكـومـات.
۷- مـوافـاة البعثات التمثيـلـيـة بمسـتـجـدات حـقـوق الإنسان في الـدولـة.
۸- متـابـعـة القضـايـا المتعلقة بحقوق الإنسان محلياً ودولياً، ومتابعـة الاجتمـاعـات والأنشـطة التي تُعقد في إطـار المنظمـات الإقليميـة والعـربـيـة والدوليـة في مجـال حـقـوق الإنسـان، والتنسيـق لإشـراك الوحدات الإدارية المعنية في الوزارة والجهات المختصة في الدولة.
٩- إحـالـة الشكاوى الـتـي تـرد مـن الخـارج والمتعلقة بحقوق الإنسان إلى الجهات المختصة، ومتـابـعـتـها.
١٠- إعـداد الخطط والمقـتـرحـات اللازمـة للاستـفـادة مـن خـدمـات الاستشارات والمسـاعــدات الفنيـة الـتـي تقـدمـها المنظمات والهيئـات الدوليـة في مجـال حـقـوق الإنسـان.
۱۱- تمثيـل الوزارة في الجـهـات المعـنيـة بـحـقـوق الإنـسـان.
 

 

المادة 21


تختـص إدارة الـشـؤون القـنـصـلـيـة بمـا يـلـي:
١- حـمـايـة ورعاية مصالح المواطنين في الخارج، ومتـابعـة هـذه المصالح، بالتنسيـق مـع البـعـثـات التمثيلية، أو البعثات الدبلـومـاسـيـة المعـتـمـدة لـدى الـدولـة، والوزارات والجهات الحكوميـة الأخـرى.
٢- إعـداد القـواعـد والتعليمات المنظمـة للشـؤون القنصلية في ضوء القـوانـين واللوائح والقرارات المعمـول بـهـا، وإبلاغ البـعـثـات التمثيليـة بهـذه التـعـلـيـمـات، ومتـابـعـة تـنـفـيـذهـا.
٣- إعـداد التعليمات لإصدار وتجـديـد جـوازات السـفـر الـعـاديـة للمواطنين وتـذاكـر المـرور في الحـالات الـتـي تسـتـدعي ذلـك، وفقاً للقـواعـد المقررة، وإبلاغ البعثات التمثيـليـة بشـأنـها.
٤- التـصـديق على الشهـادات والوثائق والمستندات الصادرة عـن الـوزارات والأجـهـزة الحكومية والبعثات الدبلـومـاسيـة ومكاتب المنظمات الدوليـة المعتمدة لدى الدولة، وتحـصيـل الـرسـوم المقررة لذلك.
٥- التـأكـد مـن سـلامـة تحـصيـل الرسوم المقررة على التأشيرات الـتي تصـدرها البعثات التمثيـليـة.
٦- اتخاذ مـا يـلـزم مـن إجـراءات بشأن التأشيـرات اليـدويـة الـتـي تُمنح على جوازات السفر العـاديـة خـارج النظام الإلكتروني للتأشيرات.
۷- تسـلم وتســليم الأوراق القضائيـة وطـلـبـات الإنـابـة والمسـاعـدة مـن وإلى الجـهـات المختصة بالدولـة، والجـهـات الأجنبيـة عـن طـريـق البعثـات التمثيـليـة أو البـعـثـات الـدبـلـومـاسـيـة المعـتمـدة لـدى الـدولـة، وذلك بالتنسيق مع الجهات المختصة.
۸- التـنـسـيـق مـع الجـهـات المخـتـصـة بشـأن ممـارسـة الدولة لحـقـوق الرقـابـة والتفتيش على السـفـن التـي تحـمـل جـنـسـيـات أجنبيـة، أو على طـاقـم كل منها، وكذلك السفـن والـطـائـرات المسجلة في الدولـة وتحـمـل جنسيتـهـا عنـد تـواجـدهـا في الدول الأخـرى.
٩- التنسيق مع الجهات المختصة في تقـديـم المسـاعـدة للسـفـن والـطـائـرات الأجنبيـة وأطقمـها، وتلقـي البـلاغـات عـن دخـولـها أو سفـرهـا عـبـر المياه والأجواء القطـريـة، وإصـدار تـصـاريـح المرور للمـلاحـة البحـريـة والطيران غير التجاريين في المياه والأجـواء القـطـريـة.
۱۰- استـخـــراج تصاريح الإذن بالهبوط والإقلاع والعبـور للـطـائـرات المدنية والعسكرية، بالتنسيق مع الجهات المختصة.
 

 

المادة 22


تختـص إدارة المـراســم بمـا يـلـي:
١- التـعـاون والتنسيـق مـع الـديـوان الأميـري، في تنظيـم وتطبيـق مـراسـم الاسـتـقـبـال والـوداع، وغيـر ذلـك مـن التـرتـيـبـات الرسميـة المتعلقـة برؤسـاء الـدول وأولياء العهـود ونـواب الرؤسـاء ورؤسـاء الحـكـومـات والأمـراء وكبـار الشـخـصـيـات الـرسـميـة العـربـيـة والأجنبيـة.
۲- اتخاذ الترتيبات اللازمـة لمراسم استقبال رؤساء البعثات الدبلوماسيـة ورؤسـاء مـكـاتـب المنظمات الدولية المعتمدة لدى الدولـة عـنـد حـضـورهـم للبلاد لأول مـرة لتقديم أوراق اعتمـادهـم، وتسـلم وحـفـظ وثـائـق اعـتـمـادهـم.
۳- رعـايـة شـؤون موظفي البعثات الدبلــومـاسـيـة ومكاتـب المنظمـات الـدوليـة المعـتـمـدة لـدى الــدولـة.
٤- إصـدار البـطـاقـات الشخصيـة للمـوظـفـين الدبلـومـاسـيـين والإداريين في البعثات الدبلـومـاسـيـة ومكاتـب المنـظـمـات الدوليـة المعتمـدة لـدى الدولـة، ولعـائـلاتـهـم والأشخـاص الـتـابـعـيـن لـهـم وفقاً لـلأدوات التشريعيـة المعمـول بـهـا.
٥- التنسـيـق مـع وزارة الداخـلـيـة بشـأن إصـدار البطـاقـات الشخصيـة للموظفين المحليين والفنيين والمسـتـخـدمـين في البعثات الدبلــومـاسيـة ومـكـاتـب المنظمات الدوليـة المعتمـدة لـدى الدولـة، وللخـدم الخـاص، وفقاً للأدوات التشـريـعـيـة المعـمـول بـها.
٦- إعـداد وتحـديـث القـائـمـة الدبلــومـاسيـة للبعثات التمثيليـة بالتنسيق مـع الوحـدات الإداريـة المعنيـة، والبعثات الدبلــومـاسـيـة ومـكـاتـب المنظمات الدوليـة المعتـمـدة لـدى الدولة، والبيـانـات الخـاصـة بموظفيـها مـن دبـلـومـاسـيـين وغيـرهـم.
٧- التعاون والتنسيق مع الجهات المختصة في ترتيب وتحـديـد مـواعـيـد ومقـابـلات مـوظـفـي السـلـكـين الدبلـومـاسـي والقنـصـلي في البعثات الدبلــومـاسيـة ومـكـاتـب المنـظـمـات الدوليــة المعتمدة لـدى الدولـة، مع المسؤولين في الدولة، وتنسيـق بـرامـج الزيارات الرسمية للوفود.
٨- إعـداد الترتيبات اللازمة لتنظيم الاحتفالات والاستقبالات الرسمية.
۹- الاحتفاظ بأعـلام الـدول العـربـيـة والأجنبيـة والأنـاشـيـد الوطنيـة الخـاصـة بها، وتزويـد البـعـثـات الدبلـومـاسـيـة المعـتـمـدة لـدى الدولـة بـعـلم دولـة قـطـر ونـشـيـدهـا الوطني.
۱۰- منـح الحـصـانـات والامتيازات والإعـفـاءات للبعثات الدبلـومـاسـيـة ومـكـاتـب المنظمات الدوليـة المعتمـدة لـدى الدولـة، ولمـوظـفـيـها وعـائـلاتـهم، في ضـوء الاتفاقيات الدوليـة ومـبـدأ المعاملة بالمثـل.
۱۱- تسـلم وتسليـم الأوراق القضائيـة والشـكاوى، المتعلقة بالبعثـات الدبلــومـاسيـة ومـكـاتـب المنظمات الدوليـة المعتمـدة لـدى الدولـة، والأجـانـب المتمتـعـين بالحـصـانـة الدبلـومـاسـيـة بصفتهم الشخصيـة أو المتعلقة بعملهم.
۱۲- جـمـع وتحـديـث البيـانـات الرئيسيـة عـن المنـاسبـات الرسمية في الـدول الأخرى، وذلك لاتخاذ إجراءات المجـامـلـة الـدوليـة والـرسـميـة بشـأنهـا.
١٣- إصـدار وتجـديـد جـوازات السفـر القـطـريـة الدبلــومـاسـيـة والخـاصـة والمهمة.
١٤- منـح تأشيـرات الـدخـول والمـرور لرؤسـاء الـدول الأجنبيـة والوفـود الرسميـة وكبـار المسؤولــين وضيـــوف الدولـة، وأفراد أسـرهــم ومـرافـقـيـهم، ومـوظـفي البعثات الدبلـومـاسـيـة ومكاتب المنظمـات الدولية المعتمدة لدى الدولة، وعـائـلاتـهم والأشخـاص الـتـابـعـيـن لـهـم، بالتنسيـق مـع الجـهـات المختصـة.
١٥- الإشراف على استخـدام الـنـقـلـيـات الخـاصـة بـالـوزارة.
 

 

المادة 23


تختص إدارة الموارد البـشـريـة بمـا يـلـي:
١- تنفيذ القوانين واللوائح والنـظـم المتعلـقـة بشـؤون الموظفين.
٢- إعداد الدراسات الخـاصـة بتخطيط الموارد البشـريـة بالـوزارة، وتحـديـد احـتـيـاجـات الوزارة من الوظائف والموظفين، في ضوء المقرر الوظيـفي المعتمـد، بالتنسيق مع الجهة المختصة والوحدات الإدارية المختـلفـة.
۳- تخطيط وتحـديـد احتيـاجـات البعثات التمثيليـة مـن الـوظـائـف والموارد البشرية المحلية أو من خارج دول المقـر بالتنسيق مع البعثات التمثيلية.
٤- إعـداد مشـروع وصـف وتصنيف وترتيب الوظائف، ومتابعـة تنفيـذه، وتطـويـره.
٥- إعداد مشروع موازنـة الباب الأول، بالتنسيق مـع الـوحـدات الإداريـة المعـنـيـة والبعثات التمثيـليـة.
٦- القيام بإجـراءات التـعـيـين، والنـقـل، والنـدب، والإعـارة، وإنهـاء الخـدمـة.
۷- اتخاذ الإجراءات اللازمة لتخصيص السكن الحكومي وصـرف بـدل الأثـاث لموظفي الوزارة، بالتنسيق مع الجهات المختصة.
٨- إعـداد مـلـف خـدمـة لـكـل مـوظـف يحـتـوي عـلـى جـمـيـع مـسـوغـات ومستندات التعيين، وجميع القرارات التي تصـدر بشـأنـه مـن بـدء خـدمـتـه حـتى انتـهـائـها، والاحتفاظ بنسخـة إلـكـتـرونيـة مـن المـلـف.
٩- إعـداد وإصـدار البطـاقـات الـوظـيـفـيـة لجـمـيـع مـوظـفـي الـوزارة، وللموظفين المحليين في البعثات التمثيـليـة.
١٠- إدخال البيانات المتعلقة بصرف الرواتب الشهـريـة والمستحقات الماليـة لموظـفـي الـوزارة والموظفين المحليين في البعثات التمثيـليـة، بالتنسيق مع الجهات المعنيـة.
۱۱- دراسـة وتـنـفـيـذ اسـتـحـقـاقـات الموظفين للإجازات وفقاً للقـانـون.
۱۲- تحـديـد الاحتـيـاجـات التـدريبيـة لموظفي الوزارة والبعثات التمثيليـة، وتـوفـيـر فـرص التـدريـب المنـاسبـة، ومتـابـعـة تـنـفـيـذهـا، وتقييـم مـدى الاستفـادة مـنـها.
۱۳- التنسيق مع المعـهـد الـدبـلـومـاسي، والجـهـة المختصة، والمؤسـسـات والمعـاهـد التعليميـة وكـافـة الجـهـات المتخصصة في الداخـل والخـارج، بشـأن تـدريـب وتـعـلـيـم اللغـات الأجنبيـة لمـوظـفـي الـوزارة والبعثات التمثيـليـة.
١٤- متـابعـة أحـوال الـدارسين والمبتعثـين مـن مـوظـفـي الـوزارة والبعثات التمثيـليـة إلى الجامعات والمؤسسات التـدريبيـة في الداخـل والخارج، بالتنسيـق مـع الـوحـدات الإداريـة والجـهـات المختصـة، ودراسـة أوضـاعـهم، وإعـداد التـقـاريـر عـن أدائـهـم فيـها.
١٥- اقتراح السيـاسـات والإجراءات والأنظمـة المنـاسبـة لإدارة أداء المـوارد البـشـريـة بـالـوزارة، بالتنسيـق مـع الـوحـدات الإداريـة المعـنيـة.
١٦- تنفيذ ومتـابعـة إجـراءات تقييم الأداء الوظيفي.
١٧- إعـداد الدراسـات الخـاصـة بالهيـكل التنظيمـي والـوظيـفـي للـوزارة والبـعـثـات التمثيليـة، وتقديم المقترحات اللازمـة لتطـويـر التنظيم الإداري بـها.
۱۸- وضـع وتحـديـث قـواعـد البيـانـات الخـاصـة لموظفي الوزارة والموظفين المحليين في البعثات التمثيـليـة.
١٩- إعـداد ومـراجـعـة وتـدقـيـق مستنـدات الـصـرف، والمعـامـلات الماليـة المتعلقـة بالمـوظـفـين.
 

 

المادة 24


تختـص إدارة الـشـؤون الماليـة والإداريـة بما يلي:
۱- تنفيـذ القـوانـيـن واللوائح والنـظم المالية والإداريـة المتعلقـة بـعـمـل الوزارة.
٢- إعداد مشروع الموازنة السنـويـة للوزارة، وحسابها الختامي، بالتنسيق مـع الوحدات الإداريـة المعـنيـة.
۳- تـوفـيـر احـتـيـاجـات الـوزارة والبعـثـات التمثيـليـة مـن المـسـتـلـزمـات والأجهزة اللازمة لأداء مهامها، بالتنسيق مع الوحدات الإداريـة المعـنيـة.
٤- القيام بأعـمـال المشـتـريـات والمنـاقـصـات والمـزايـدات، وفـق الأنظمـة والقـواعـد المعـمـول بـها فـي الـدولـة.
٥- الإشـراف على تنفيـذ الاعتمـادات، وتدقيـق حـسـابـات الإيرادات والمـصـروفـات.
٦- إعـداد مـسـتـنـدات الصـرف، وسـائـر المعـامـلات الماليـة الأخـرى.
۷- اتخاذ الإجـراءات اللازمـة لتخصيـص المباني الحـكـومـيـة للـوزارة، وصيـانـتـها، بالتنـسـيـق مـع الجـهـات المختصة.
۸- تـوفـيـر مـقـر للبعثـة التمثيليـة، وسـكن لأعـضـائـها، والخـدمـات اللازمة، بالتنسيـق مـع الـوحـدات الإداريـة المختصة.
۹- تحـصيـل الـرسـوم ومـقـابـل الخـدمـات التي تؤديها الوزارة.
۱۰- القـيـام بأعـمـال الخـدمـات الإداريـة.
١١- الإشراف على مـخـازن الوزارة.
 

 

المادة 25


تختص إدارة نـظـم المـعـلـومـات بمـا يـلـي:
۱- تـوفـيـر وتـطـويـر البيانات والمعلـومـات السياسية والاقتصـاديـة والثقـافيـة والفـنـيـة وغيـرهـا، بمـا يـوفـر قـاعـدة مـتـكـامـلـة مـن المعـلـومـات للوحـدات الإداريـة بالوزارة وبعـثـاتـها التمثيـليـة.
۲- دراسـة وتصميـم وتنفيـذ وصيـانـة نـظم المعلـومـات الآليـة بمـا يـسـاعـد على تطـويـر أعمال الوزارة، وتقـديـم المقترحات الفـنـيـة بشـأنـها.
٣- المعالجـة الآليـة لبيـانـات ومـعـلـومـات الـوزارة، والإشراف على إدخـالـها واستـرجـاعـها، وإصـدار المعـلـومـات والـتـقـاريـر اللازمـة عنـها.
٤- تـوفـيـر الأجهـزة والـبـرامـج والـشـبـكـات الإلـكـتـرونيـة اللازمـة لنظـم العمـل الآليـة بالوزارة، وصيانتهـا، بالتنـسـيـق مـع الوحـدات الإداريـة المعنية.
٥- تـأمـين الأنظمـة والبـرمـجـيـات والشبكات والبنية التحتيـة لنـظم المعـلـومـات مـن مـخـاطـر الحـوادث الإلكتـرونيـة، ووضـع النـظم والإجراءات الكفيلة بالمحافظة على سـريـة نظم المعلومات.
٦- إعـداد خطط التعـافـي مـن آثـار الحـوادث الإلكتـرونيـة حـال وقـوعـها أو تقليـل حـدوثها أو تصعيـدهـا، وفـق السياسات والمبادئ التـوجـيـهيـة لأمـن المعـلـومـات، بالتنسـيـق مـع الجـهـة المخـتـصـة.
۷- ربط الأنظمة الإلكتـرونيـة المستخـدمـة بالوزارة مع البعثات التمثيلية.
۸- تقـديـم الـدعـم الفـنـي الـلازم لمسـتـخـدمي الشبكات الإلكترونيـة في الوزارة والبعثات التمثيليـة، وتدريبـهـم علـى اسـتـخـدام الأنظمـة والبـرامـج وأجـهـزة الحـاسـب الآلي ومـلـحـقـاتـها.
۹- تحـديـد الاحتيـاجـات مـن الأنظمـة الآليـة ومـواصـفـات الأجـهـزة والمعـدات المطـلـوبـة.
۱۰- برمجـة وحـفـظ واستـرجـاع وتعـديــل نـظـم المـعـلـومـات والبيـانـات اللازمـة لأنشـطـة الـوزارة.
۱۱- تـصـمـيـم وتشـغيـل وإدارة قـواعـد البـيـانـات والمـعـلـومـات بـأنـواعـها المختلفة.
۱۲- اقـتـراح البـرامـج التـدريبيـة فـي مـجـال تـقـنـيـة المـعـلـومـات.
١٣- إنشاء ومتـابعـة وتحـديـث مـوقـع الوزارة على شبكة المعلـومـات الـدوليـة.
 

 

المادة 26


تختص إدارة الـشـؤون الـهـنـدسيـة والخـدمـات الـعـامـة بمـا يـلـي:
١- الإشراف على مشـاريـع بنـاء وتـأثـيـث وتجـهـيـز مـبـانـي الـوزارة.
۲- مـتـابـعـة تـطـويـر المنـطـقـة الـدبـلـومـاسـيـة، بالتنسيـق مـع الجـهـات المختصة بالدولة.
۳- مـتـابـعـة إجـراءات عـقـود إنشـاء وشــراء واستئجـار مـبـاني الـوزارة والبعثــات التمثيـليـة بالتنسيـق مـع الوحـدات الإداريـة والجهـات المختصة.
٤- الإشراف علـى جـمـيـع الأعـمـال المتعلقـة بـصـيـانـة المبـاني والأثـاث والآلات والتوصيـلات الـكـهـربـائـيـة والمـائيـة وخـطـوط الهاتف والتلكس والفـاكـس وغـيـرهـا الخـاصـة بالوزارة.
٥- الإشراف على التـأمـين عـلـى جـمـيـع المباني وأثـاث الوزارة، وجـمـيـع ممتلكات الدولة مـن مـبـان وأثـاث في البعثات التمثيـليـة، ومتابعـتـه، بالتنسيـق مـع الجـهـات المختصة.
٦- فـتـح سـجـلات لجـمـيـع المباني المملـوكـة للوزارة والبعثات التمثيـليـة وتجـهـيـزاتـها، متضمنـة كـل البيـانـات المتعلقة بالمبـنـى مـن النـاحـيـة الفنيـة أو المـالـيـة كالخــرائـط وسندات الملكيـة والفواتير والإيصالات وغـيـرهـا مـن المسـتـنـدات، وتحـديـثـها.
۷- اتخاذ الإجـراءات اللازمـة بشـأن الطلبـات الواردة مـن إدارة المـراســم الخاصـة بالبعثات الدبـلـومـاسـيـة ومكاتب المنظمات الدوليـة المعتمـدة لدى الدولة المتعلقة بشـراء واستئـجـار المبـاني والخـدمـات الحكوميـة الخـاصـة بـهم، بالتنسيـق مـع الجـهـة المختصة بالـدولـة.
۸- اتخاذ الإجـراءات اللازمـة بشـأن الطلبـات الـواردة مـن البعثـات التمثيـليـة، ومتـابـعـة ودراسـة متطلبـات المبـاني الممـلـوكـة للـوزارة خـارج الدولة، بالتنسيـق مـع الجـهـات المختصة بالدولـة.
٩- إجـراء التقييـم الفـنـي لجـمـيـع المشـاريع والمبـاني الممـلـوكـة للوزارة في الداخـل والخـارج، وضمـان مـواءمـة وتـكـامـل كـل مشـروع بتفـرعـاتـه.
١٠- الإشراف والمتـابـعـة للمشـاريـع، مع وضع الخطط والبرامج التنفيـذيـة المنـاسبـة، والتقيـد بجـمـيـع معايير الجـودة والاستـدامـة.
 

 

المادة 27


تختـص إدارة الـوثـائـق والمـراسـلات بمـا يـلـي:
۱- تسـلـم وتـسـلـيـم وتصنيف وقـيـد البـريـد الـوارد والصـادر.
۲- تـوزيـع البـريـد الـوارد على الوحدات الإداريـة المعـنـيـة، وفقاً للنظام المتـبـع فـي هـذا الشأن.
٣- تأمين إيصال البريـد الصـادر خـارج الوزارة، بعـد تـرميـزه وتسجيلـه وفقاً للنظام المتبـع، بالتنسيق مع الوحدات الإداريـة المـعـنـيـة والجـهـات المختصة، بـعـد التـأكـد مـن اسـتيـفـائـه لجـمـيـع المـرفـقـات.
٤- إعداد وتجـهـيـز الحقيبة الدبلوماسيـة، وتأمين إرسالها إلى البعثـات التمثيـليـة بالتنسيـق مـع الجـهـات المختصة.
٥- تسـلـم الحـقـيبـة الـدبـلـومـاسـيـة الـواردة مـن البعثات التمثيليـة، وتسـلـيـم مـحـتـويـاتـها إلى الوحدات الإداريـة المعنيـة والجهات المختصة.
٦- تـنـظـيـم أرشـيـف الـوزارة وحـفـظ الـوثـائـق، وفقاً لأحـدث الطـرق.
۷- توصيـف وتصنيف الوثـائـق والمستندات وصـفـاً مـاديـاً يبـين عـنـاصـرهـا ومـحـتـويـاتـهـا وعـمـل فـهـارس مـوضـوعـيـة لـها وحفظها.
٨- أرشـفـة سـجـلات التسليـم للـوثـائـق والمـراسـلات.
۹- اسـتـرجـاع البيـانـات والمعـلـومـات عـنـد طـلـبـها.
۱۰- الاحتفاظ بالبحوث والدراسات التي تـخـص عـمـل الـوزارة.
۱۱- الاحـتـفـاظ بصــورة مــن وثـائـق الندوات والمؤتمرات واللقـاءات التي تخص عـمـل الوحـدات الإداريـة بالوزارة.
۱۲- حـفـظ أصـول جـمـيـع المعـاهـدات والوثائق الدبلومـاسـيـة المتعلقة بهـا، ووثـائـق الـتـصـديـق أو القبـول أو الانضـمـام المتعلقـة بـجـمـيـع المعـاهـدات الدوليـة الـتـي تـكون الدولـة طـرفـاً فيهـا، بالتنسيـق مـع الوحدات الإداريـة المختصة.
۱۳- الاحـتـفـاظ بأصـول سـنـدات الملكيـة والعـقـود الخـاصـة بالوزارة.
 

 

المادة 28


تختص إدارة الإعـلام والاتـصـال بـمـا يـلـي:
١- إعـداد وإصـدار البيـانـات الـتـي تـوضـح الصـورة الإيجابيـة للدولـة إلى الـرأي العـام العـالمي، والـرد عـلـى مـا مـن شـأنـه أن ينـال مـن سـيـادتـها ومكانتها الدوليـة، بالتنسيق مع الجهات المعنيـة.
٢- التنسيـق مـع مـراسـلي الصحـف ووكالات الأنباء ووسائل الإعـلام المحليـة والعـربـيـة والأجنبيـة المختـلفـة، وتزويـدهـا بمـا يـصـدر عـن الـــوزارة مـن بـيـانـات صحـفـيـة حـول الأحــداث داخـل الدولـة وخارجها.
۳- رصـد ومـتـابـعـة وتحـلـيـل الأحـداث اليـوميـة داخـل الدولـة وخـارجـها مـن خـلال وكـالات الأنبـاء ووسائل الإعـلام المحـلـيـة والعـربـيـة والدوليـة، وإعـداد التقاريـر بـشـأنـها.
٤- إعـداد مـشـاريـع البيـانـات والـتـصـريـحـات الصـحـفـيـة الـتي تصـدر عـن الـوزارة.
٥- تزويـد البـعـثـات التمثيـليـة بتقـاريـر إخـبـاريـة عـن مـجـرى الأحـداث الوطنيـة في مختلف المجالات.
٦- إصـدار المجـلات والنـشـرات الإعـلاميـة الدوريـة، وتـوزيـعـهـا داخـل الدولـة وخـارجـها.
۷- تنـسـيـق وإقـامـة المؤتمرات الصحفية التي تخص نشاط الإدارة.
٨- إعداد الفـيـديـوهـات والصـور ووسائل الإيضاح المتعلقة باخـتـصـاصـات الإدارة.
٩- القيام بأعـمـال التـرجـمـة الـفـوريـة والتـحـريـريـة للمـوضـوعـات والوثـائـق والكتب والبحوث العلميـة والمكاتبات الأجنبية المتعلقة باختصاصات الوزارة
 

 

المادة 29


يختص المعـهـد الـدبـلـومـاسـي بمـا يـلـي:
١- إعـداد وتـأهـيـل وتـدريـب العـامـلـيـن بـالـوزارة للـعـمـل في السـلـكـين الدبـلـومـاسـي والقنـصـلـي ومـجـال التـعـاون الدولي، بالتنسيق مع إدارة المـوارد البـشـريـة، وعـقـد الـدورات الدراسـيـة والتـدريبيـة وبـرامـج التـأهـيـل الخـاصـة لتنميـة قـدراتهـم العـلـميـة والنـظـريـة.
٢- إعـداد وتنفيـذ البـرامـج التـدريبيـة للعـامـلـين في الوزارة والبـعـثـات التمثيـليـة، وإكـسـابـهـم المهـارات اللازمـة.
٣- إعـداد الـدراسـات والبحـوث على المستـويـيـن الـدولي والإقـليـمي في المجالات الدبلـومـاسـيـة والعـلاقـات الدوليـة السيـاسـيـة والاقتصـاديـة، وغيـرهـا مـن المـجـالات الأخـرى.
٤- دراسـة مـا يـصــدر عـن التجـمـعـات الإقليميـة والـدوليـة ومـراكـز الأبحـاث والدراسـات، وتحـلـيـلـهـا وتـزويـد المسؤولين بالـرأي حـولـها بقـدر ارتبـاطـها بتنفيذ السيـاسـة الخـارجـيـة للـدولـة.
٥- التعاون مع الوزارات والأجـهـزة الحكومية والهيئات والمؤسسات العـامـة والمؤسسات الخاصة ذات النفع العام بما يحـقـق أداء المعـهـد لمهـامـه.
٦- تـقـديـم الاستشـارات للوزارات والأجـهـزة الحكوميـة والهيئـات والمؤسـسـات العـامـة والمـؤسـسـات الخـاصـة ذات النـفـع العـام في الدولة، وذلك وفقاً لاخـتـصـاصـات المـعـهـد.
٧- التـعـاون والتنسيـق مـع الجـامـعـات والمـعـاهـد المماثلـة داخـل الدولـة وخـارجـهـا فـي كـافـة المجـالات التي تحقق أهـداف المعـهـد.
٨- إقـامـة النـدوات واللقاءات العلميـة والحـلـقـات الدراسيـة، والمشـاركـة في الاجـتـمـاعـات والمؤتمرات والندوات، ذات الصلة بأهـداف المعـهـد داخـل وخارج الـدولـة.
۹- تـوفـيـر الخـدمـات المتـطـورة لمكتبة الوزارة وتزويـدهـا بمـصـادر المعـرفـة المتخصصة المرتبـطـة بمـجـالات الـعـمـل الـدبـلـومـاسي والقنصـلي.
 

 

المادة 30


تختص إدارة الـشـؤون الأمنيـة بمـا يـلـي:
۱- تـنـظـيـم ومـراقـبـة إجـراءات دخول وخروج جـمـيـع الأشخـاص لمبنى الوزارة.
۲- مـتـابـعـة إجراءات السلامة لمبنى الوزارة وجـمـيـع مـحـتـويـاتـه واقـتـراح السبل الكفيـلـة بـعـلاج مـا قـد يـوجـد في هذه الإجـراءات مـن قـصـور، وذلك بالتنسيق مع الجهات المختصة بالدولة.
٣- التـأكـد مـن كفاءة عمـل ترتيبات الدفاع المدني بمبنى الوزارة وملحقاته.
٤- إعـداد وإصـدار بـطـاقـات هـويـة مـنـاسـبـة للـعـمـال وغيـرهـم مـمن يـؤدون أعـمـالاً مـؤقـتـة في مبنى الوزارة وملحـقـاتـه.
٥- إعـداد الإرشـادات والتـوجـيـهـات الأمـنـيـة وتعميمـها علـى العـامـلـين في الوزارة.
٦- الإشـراف علـى أعـمـال حـراسـة مـبـنى الوزارة وملحـقـاتـه، بالتنسيـق مع الجـهـات المختصة.
۷- تـنـظـيـم الإشراف علـى أمـاكـن وقـوف الـسيـارات وحـركـتـهـا داخـل مبنى الوزارة، وإصـدار تـصـاريـح دخـولـها.
۸- الاحـتـفـاظ بسـجـل مـفـاتيـح لجـمـيـع مـكـاتـب الـوزارة، واتخاذ الإجراءات الأمنيـة اللازمـة عـنـد فـقـدانـها.
٩- وضع نظام أمني خاص للبعثات التمثيلية، والإشراف عليـه ومتابعته.
۱۰- مـتـابـعـة الأوضاع الأمنية في دول مـقـر البعثات التمثيـليـة بالتنسيـق مـع الوحـدات الإداريـة المعـنـيـة والجـهـات المختصة.
 

 

المادة 31


تختص إدارة الـعـمـلـيـات بمـا يـلـي:
١- تلقي الاتـصـالات والمراسلات والمعـلـومـات الواردة للوزارة مـن الجـهـات المختـلفـة على مدار الساعـة، وإحالتها إلى الوحدات الإداريـة المعـنـيـة والجـهـات المختصة، والتنـسـيـق مـعـها.
۲- تلقي اتـصـالات المواطنين القطريين بالخارج وتذليـل الصعـوبـات الـتي تـواجـهـهـم علـى مـدار السـاعـة لمسـاعـدتـهم في طـلـبـاتـهـم، بالتنسيـق مـع البـعـثـات التمثيـليـة وإدارة الشؤون القـنـصـلـيـة.
۳- تسـلـم وتـسـلـيـم الـرسـائـل الـرسـميـة المستـعـجلـة باليـد إلى الإدارات بالــوزارة أو البعثـات الـدبـلـومـاسـيـة ومـكـاتـب المنظمات الدوليـة المعتمدة لدى الـدولـة.
٤- تسلـم نـسـخ مـن المـراسـلات الـعـاجـلـة بـعـد الـدوام الـرسـمـي وخـلال الإجـازات الرسميـة والـواردة مـن البـعـثـات التـمـثيـليـة إلى الإدارات بالوزارة، وإخـطـارهـم بأهميتها، لاتخاذ اللازم مـن قبـلـهم.
٥- مـتـابـعـة استكمال استـخـراج تـصـاريح الإذن بالهـبـوط والإقلاع والعـبـور للـطـائـرات المـدنـيـة والعـسـكـريـة بـعـد الدوام الرسمي وخـلال الإجـازات، بالتنسيق مع إدارة الشؤون القنـصـلـيـة والجهـات المختصة بالـدولـة.
٦- القيام بإجـراءات فـتـح صـالات المطار في الحالات العـاجـلـة بـعـد الـدوام الرسمي وخلال فترة الإجـازات، بالتنسيـق مـع إدارة المـراســم.
٧- تـلـقـي طـلـبـات التـأشـيـرات مـن البـعـثـات التمثيـليـة، والبـعـثـات الدبـلـومـاسـيـة ومكاتب المنظمات الـدوليـة المعـتـمـدة لدى الدولة، بعـد الدوام الرسمي وخلال فترة الإجازات، وإحالتها إلى الإدارات المعـنيـة.
٨- إصـدار وتحـديـث دليـل أرقـام هـواتـف كـبـار المسؤولين ومدراء الإدارات والمكاتب والموظفين بوزارة الخـارجـيـة والبعـثـات التمثيـليـة، والبعثات الدبـلـومـاسـيـة ومـكـاتـب المنـظـمـات الـدوليـة المعـتـمـدة لـدى الدولة، بالتنـسـيـق مـع إدارة المـراسـم.
 

 

المادة 32


تختـص اللـجـنـة الـدائـمـة لـتـنـظـيـم المـؤتمـرات بمـا يـلـي:
۱- تـنـظـيـم المـؤتمـرات المتعلقـة بـعـمـل الـوزارة أو التي تُكلـف بـهـا، وإدارة أعـمـالـها، بالتنسيـق مـع الوحدات الإداريـة بالوزارات والجـهـات الحكوميـة الأخـرى ذات الصلة، ووفقاً لأهـداف المؤتمرات والأغـراض التـي تُـعـقـد مـن أجـلها.
۲- اقـتـراح مـوازنـات تـلك المؤتمرات، بالتنسيق مـع الجـهـات الحكومية الأخرى المختصة، ووفقاً للقرارات العليـا التي تقضي بإقـامـة تـلك المؤتمرات.
۳- وضـع وتـطـويـر آلـيـات المـراجـعـة بغرض تقييـم أعـمـال المـؤتمـرات، والتـقـدم بـتـوصـيـات مـحـددة حـول تحسين وسـائـل تـنـظيـمـها وإدارتها.
٤- التنسيق بين الجـهـات الحكوميـة في الدولـة وإخـطـارهـا بالمـواعـيـد المقـتـرحـة للاستـضـافـة، وإعداد مخطط سنـوي وجـدول لـها.
٥- أي اخـتـصـاصـات أخـرى تـتـعـلـق بـتنـظـيـم المـؤتمرات تكلفـهـا بهـا السـلطـة المختصة في الوزارة.

 

المادة 33


يصدر بتحـديـد اخـتـصـاصـات كـل مـن مـكـتـب الـوزيـر، ومـكـتـب وزيـر الـدولـة، ومـكـتـب مـسـاعـد الـوزيـر، ومكتب الأمين العـام، قـرار مـن الوزيـر.

 

المادة 34


يجـوز بقـرار مـن مـجـلـس الوزراء، بناءً على اقتراح الـوزيـر، تعـديـل تنظيـم الـوحـدات الإداريـة الـتي تتـألـف منهـا الـوزارة بالإضـافـة أو الحـذف أو الـدمـج، وتعيين اختصـاصـاتـهـا وتـعـديـلـها.

 

المادة 35


يجـوز بقـرار مـن الوزير إنشاء ما تقتضيه المصلحـة العـامـة مـن أقسام في الوحدات الإداريـة الـتـي تـتـألـف منهـا الـوزارة، أو إلغاؤهـا أو دمجها، وتعيين اختصـاصـاتـهـا وتـعـديـلها، ولا يـكـون قـرار الوزيـر نـافـذاً إلا بعـد اعتماده من مجلس الوزراء

 

المادة 36


للـوزيـر في إطـار السياسـة العـامـة للـوزارة، تشـكـيـل لجـان عـامـة ومتخصصة، وتعيين اخـتـصـاصـاتـهـا وتـعـديـلـها، وتحـديـد أهـدافـها ومـدتها، وإعـادة تنظيمها أو إلـغـاؤهـا.

 

المادة 37


يُلغى القرار الأميـري رقم (۲۷) لسنة ٢٠١٦ المشـار إليـه.

 

المادة 38


على جميع الجهات المختصة، كـل فيما يخصه، تنفيذ هذا القـرار. ويُعـمـل بـه مـن تـاريـخ صـدوره. ويُنشر في الجـريـدة الـرسـميـة.
