مـرسـوم رقـم (94) لـسـنـة 2025 بـالـتـصـديـق عـلـى مـذكـرة تـفـاهـم بـشـأن الاعـتـراف المـتـبادل بـشـهـادات الـبـحـارة بـين حـكـومـة دولـة قـطـر وحـكـومـة جـمـهـوريـة بـنـين 94 / 2025
عدد المواد: 2
فهرس الموضوعات

المواد (1-2)



نـحـن تمـيـم بـن حـمـد آل ثـانـي                أمـيـر دولـة قـطـر ،
بـعـد الاطـلاع عـلـى الـدسـتـور،
وعـلـى وثـيـقـة الـتـصـديـق الـصـادرة فـي الـسـادس والـعـشـريـن مـن شـهـر ذي الـحـجـة عـام 1446 هـجـريـة ، المـوافـق لـلـثـانـي والـعـشـريـن مـن شـهـر يـونـيـو عـام 2025 مـيـلاديـة ،
وعـلـى اقـتـراح رئـيـس مـجـلـس الـوزراء ووزيـر الـخـارجـيـة ،
وعـلـى مـشـروع المـرسـوم المـقـدم مـن مـجـلـس الـوزراء ،
رسـمـنـا بمـا هـو آت :
 

المواد

المادة 1


صُـودق عـلـى مـذكـرة تـفـاهـم بـشـأن الاعـتـراف المـتـبادل بـشـهـادات الـبـحـارة بـين حـكـومـة دولـة قـطـر وحـكـومـة جـمـهـوريـة بـنـين ، المـوقـعـة بمـدينـة الـدوحـة بـتـاريـخ 20 /05/ 2025 ، المـرفـق نـصـهـا بـهـذا المـرسـوم ، وتـكـون لـهـا قـوة الـقـانـون ، وفـقـاً لـلـمـادة (68) مـن الـدسـتـور .

 

المادة 2


عـلـى جـمـيـع الـجـهـات المـخـتـصـة ، كـل فـيـمـا يـخـصـه ، تـنـفـيـذ هـذا المـرسـوم . ويُـعـمـل بـه مـن تـاريـخ صـدوره . ويُـنـشـر فـي  الـجـريـدة الـرسـمـيـة .
