بـعـد الاطـلاع عـلـى الـدسـتـور،

وعـلـى الـقـانـون رقـم (15) لـسـنـة 2002 بـشـأن الـطـيـران المـدنـي ، والـقـوانـين المـعـدلـة لـه ،
وعـلـى الـقـرار الأمـيـري رقـم (66) لـسـنـة 2018 بإعـادة تـنـظـيـم الـهـيـئـة الـعـامـة للـطـيـران المـدنـي ،

الـفـصـل الأول - تـعـاريـف وأحـكـام عـامـة

المادة 1


فـي تـطـبـيـق أحـكـام هـذا الـقـانـون ، تـكـون للـكـلمـات والـعـبارات الـتـالـيـة ، المـعـانـي المـوضـحـة قـرين كـل مـنـهـا ، مـا لم يـقـتـض الـسـيـاق مـعـنـى آخــر :
الـوزيـر : وزيـر المـواصـلات .
الـهـيـئـة : الـهـيـئـة الـعـامـة للـطـيـران المـدنـي .
الـرئـيـس : رئـيـس الـهـيـئـة .
الإدارة : الـوحـدة الإداريـة المـخـتـصـة بالـهـيـئـة .
المـنـظـمـة : المـنـظـمـة الـعـالمـيـة للأرصـاد الـجـويـة .
أنـشـطـة الأرصـاد الـجـويـة : المـمـارسـات الـعـلمـيـة أو الـتـقـنـيـة المـنـفـردة
أو المـجـتـمـعـة الـتـي يـقـوم بـهـا الـفـرد أو الـجـهـات والمـؤسـسـات المـعـنـيـة فـي مـجـال الـرصـد أو الـتـنـبـؤ الـجـوي والـبـحـري أو الـعـمـلـيـات الـتـقـنـيـة المـسـانـدة كـالمـراقـبـة أو الـقـيـاس أو الـتـحـلـيـل أو الـتـشـغـيـل
أو الـصـيـانـة أو الـدعـم الـفـنـي أو المـعـايـرة الـدوريـة لأجـهـزة الأرصـاد الـجـويـة أو الاتـصـال أو الـتـدريـب
أو الـبـحـث كإجـراءات عـمـل وفـق مـعـاييـر فـنـيـة مـحـددة مـن أجـل الـحـصـول عـلـى إحـدى مـخــرجـات أو خـدمـات الأرصـاد الـجـويـة .
خـدمـات الأرصـاد الـجـويـة : خـدمـات إعـداد وتـقـديـم مـعـلـومـات الأرصـاد الـجـويـة والمـنـاخ ، بمـا فـيـهـا الـرصـد والـتـنـبـؤ والـتـحـذيـرات الـجـويـة والـبـحـريـة والـتـوقـعـات المـنـاخـيـة ومـخــرجـاتـهـا وتـبادلـهـا مـع الـجـهـات والمـؤسـسـات المـخـتـصـة بالأرصـاد الـجـويـة داخـل الـدولـة وخـارجـهـا .
الـرصـد : عـمـلـيـة عـلمـيـة وتـقـنـيـة تـطـبـيـقـيـة مـتـكـامـلـة تـهـدف إلـى المـراقـبـة المـنـتـظـمـة للـحـالات والـظـواهـر الـجـويـة وقـيـاس الـعـنـاصـر الـجـويـة والـبـحـريـة بـشـكـل يـومـي فـي أوقـات مـحـدده لـجـمـع المـعـلـومـات وتـشـفـيـرهـا وفـقـاً للـقـوانـين الـدولـيـة وإحـالـتـهـا لـكـافـة المـسـتـفـيـدين .
الـتـنـبـؤ الـجـوي : عـمـلـيـة عـلمـيـة وتـقـنـيـة تـطـبـيـقـيـة مـتـكـامـلـة لـتـحـلـيـل حـالـة الـغـلاف الـجـوي بـهـدف الـتـوقـع الـدقـيـق للـعـنـاصـر والمـتـغـيـرات والـظـواهـر الـجـويـة الـتـي سـتـسـود خـلال فـتـرة زمـنـيـة قـادمـة ، فـوق مـنـطـقـة جـغـرافـيـة مـحـددة وإعـداد الـتـقـاريـر والـتـحـذيـرات المـنـاسـبـة وفـقـاً للـقـوانـين الـدولـيـة وإحـالـتـهـا لـكـافـة المـسـتـفـيـدين .
المـعـايـرة الـدوريـة : عـمـلـيـة تُجـرى مـن قـبـل جـهـة مـتـخـصـصـة ومـعـتـمـدة ومـرخـص لـهـا بإجـراء عـمـلـيـة فـحـص ومـنـح اعـتـمـاد لـجـهـاز مـعـين للـرصـد الـجـوي ، مـن حـيـث دقـة الـقـراءة أو الـقـيـاس المـطـلـوب وفـق مـعـاييـر دولـيـة مـعـتـمـدة .
أجـهـزة الـرصـد الـجـوي : مـعـدات الـرصـد والمـراقـبـة والاسـتـشـعـار الـتـي تـسـتـخـدم فـي خـدمـات الأرصـاد الـجـويـة .
الـهـواة : الأشـخـاص الـطـبـيـعـيـون الـذين يـمـارسـون أنـشـطـة الأرصـاد الـجـويـة لأغـراض غـيـر ربـحـيـة .
الـنـشـرة الـجـويـة : مـجـمـوعـة المـعـلـومـات الـتـي تـبـين حـالـة الـطـقـس الـسـائـد والمـتـوقـع فـي الـدولـة .
الـتـحـذيـرات : إعـلان الإدارة عـن حـدوث ظـواهـر جـويـة أو بـحـريـة خـطـرة تـم رصـدهـا أو الـتـنـبـؤ بـهـا وتـؤثـر سـلـباً عـلـى الـسـلامـة الـعـامـة .
مـحـطـة الـرصـد الـجـوي : المـكـان أو الـفـضـاء أو المـنـصـات أو الـعـوامـات الـبـحـريـة أو المـسـطـح المـائـي المـخـصـص للـقـيـام بـعـمـلـيـة رصـد الـعـنـاصـر المـخـتـلـفـة والـتـي تـحـتـوي عـلـى الأجـهـزة والمـعـدات لإجـراء عـمـلـيـات الـرصـد والمـراقـبـة للأحـوال الـجـويـة والـبـحـريـة .  

 

المادة 2


 مـع مُراعـاة أحـكـام الاتـفـاقـيـات المـتـعـلـقـة بالأرصـاد والـتـنـبـؤات الـجـويـة الـتـي أصـبـحـت الـدولـة طـرفـاً فـيـهـا ، تـكـون مـمـارسـة أنـشـطـة وتـقـديـم خـدمـات الأرصـاد الـجـويـة داخـل إقـلـيـم الـدولـة وعـلـى المـنـصـات الـبـحـريـة ذات المـنـفـعـة الاقـتـصـاديـة ، وفـقـاً لأحـكـام هـذا الـقـانـون .
ولا تـسـري أحـكـام هـذا الـقـانـون عـلـى الـجـهـات الـعـسـكـريـة .
 

 

المادة 3


تتـولـى الإدارة دون غـيـرهـا إعـداد وإصـدار الـنـشـرات الـجـويـة والـتـحـذيـرات الـعـامـة المـتـعـلـقـة بأحـوال الـطـقـس والمـنـاخ .
 كـمـا تتـولـى تـقـديـم خـدمـات الأرصـاد والـتـحـذيـرات الـخـاصـة بـشـؤون الـطـيـران والمـلاحـة الـجـويـة والـبـحـريـة .
 

 

الـفـصـل الـثـانـي - الـتـراخـيـص

المادة 4


لا يـجـوز ممـارسـة أنـشـطـة الأرصـاد الـجـويـة لأغـراض ربـحـيـة ، إلا بـعـد الـحـصـول عـلـى تـرخـيـص مـن الإدارة .

 

المادة 5


يـجـب أن تتـوفـر فـي طـالـب الـتـرخـيـص بمـزاولـة أنـشـطـة الأرصـاد الـجـويـة الـشـروط والـضـوابـط الـتـي يـصـدر بـهـا قـرار مـن الـوزيـر ، بـنـاءً عـلـى اقـتـراح الـرئـيـس .

 

المادة 6


يُقـدم طـلـب الـحـصـول عـلـى تـرخـيـص مـزاولـة أنـشـطـة الأرصـاد الـجـويـة أو تـجـديـده إلـى الإدارة عـلـى الـنـمـوذج المـعـد لـهـذا الـغـرض ، مـرفـقـاً بـه المـسـتـنـدات الـتـي تـحـددهـا الإدارة . 
وتتـولـى الإدارة الـبـت فـي الـطـلـب وإخـطـار صـاحـب الـشـأن بـقـرارهـا فـيـه عـلـى عـنـوانـه الـوطـنـي أو بأي وسـيـلـة أخــرى تـفـيـد الـعـلم ، خـلال ثـلاثـين يـومـاً مـن تـاريـخ تـقـديـمـه ، وإذا صـدر الـقـرار بالـرفـض وجـب أن يـكـون مـسـبـباً ، ويُعـتـبـر انـقـضـاء هـذه المـدة دون رد عـلـى الـطـلـب رفـضـاً  ضـمـنـيـاً لـه . 
ويـجـوز لمـن رُفـض طـلـبـه أن يـتـظـلم إلـى الـرئـيـس خـلال سـتـين يـومـاً مـن تـاريـخ إخـطـاره بالـقـرار أو مـن تـاريـخ اعـتـبار طـلـبـه مـرفـوضـاً ضـمـنـيـاً .
ويـبـت الـرئـيـس فـي الـتـظـلم خـلال ثـلاثـين يـومـاً مـن تـاريـخ تـقـديـمـه ، ويـعـتـبـر انـقـضـاء هـذه المـدة دون رد عـلـى الـتـظـلم رفـضـاً ضـمـنـيـاً لـه ، ويـكـون قـرار الـرئـيـس فـي الـتـظـلم نـهـائـيـاً .
 

 

المادة 7


تُصـدر الإدارة بـعـد اسـتـيـفـاء الـرسـوم المـقـررة ، الـتـرخـيـص بمـزاولـة أنـشـطـة الأرصـاد الـجـويـة وتـكـون مـدة الـتـرخـيـص سـنـة قـابـلـة للـتـجـديـد لمـدة أو مـدد أخــرى مـمـاثـلـة .
ويـجـب أن يـقـدم طـلـب الـتـجـديـد خـلال ثـلاثـين يـومـاً عـلـى الأقـل قـبـل تـاريـخ انـتـهـاء الـتـرخـيـص . 
ويـجـوز للإدارة ، قـبـول الـطـلـب إذا تـم تـقـديـمـه خـلال الـثـلاثـة أشـهـر الـتـالـيـة لانـتـهـاء مـدة الـتـرخـيـص ، مـتـى قـدم الـطـالـب عـذراً  تـقـبـلـه الإدارة .
ويـجـب لـتـجـديـد الـتـرخـيـص أن تتـوفـر جـمـيـع الـشـروط المـطـلـوبـة للـتـرخـيـص لأول مـرة .
وتتـولـى الإدارة قـيـد الـتـراخـيـص الـصـادرة ، فـي الـسـجـل الـذي تـعـده لـهـذا الـغـرض ، عـلـى أن يُعـتـمـد مـن الـرئـيـس .
 

 

المادة 8


يُعـد الـتـرخـيـص الـصـادر بمـوجـب أحـكـام هـذا الـقـانـون شـخـصـيـاً ، ولا يـجـوز الـتـنـازل عـنـه أو الـتـصـرف فـيـه بأي وجـه دون مـوافـقـة الإدارة  وإلا اعـتُبـر مـلـغـيـاً .
ويـجـب أن تتـوفـر فـي المـتـنـازل إلـيـه الـشـروط الـواجـب تـوفـرهـا فـي المـرخـص لـه .
 

 

الـفـصـل الـثـالـث - مـحـطـات الـرصـد الـجـوي

المادة 9


تـعـد جـمـيـع المـنـشـآت وأجـهـزة الـرصـد الـجـوي والمـعـدات والمـحـطـات الـسـلـكـيـة والـلاسـلـكـيـة والمـنـارات والأسـلاك ومـحـطـات الأرصـاد الـجـويـة الـتـابـعـة للإدارة مـن الأمـلاك الـعـامـة للـدولـة ، ويـتـم تـخـصـيـص الـعـقـارات الـلازمـة لإنـشـاء مـحـطـات الأرصـاد الـجـويـة وفـقـاً للـقـانـون ، ولا يـجـوز نـقـل أو تـغـييـر أو اسـتـبـدال تـلـك المـحـطـات إلا بمـوافـقـة الـهـيـئـة . 

 

المادة 10


تُنـشـأ حـقـوق ارتـفـاق لمـحـطـات الأرصـاد الـجـويـة لـتـأمـين حـسـن عـمـلـهـا وعـمـل الأجـهـزة الـخـاصـة والمـتـعـلـقـة بـهـا .
 وللـوزيـر ، وفـقـاً لمـقـتـضـيـات المـصـلـحـة الـعـامـة ، تـحـديـد نـطـاق ومـدى حـقـوق الارتـفـاق والمـنـاطـق الـتـي تـفـرض فـيـهـا ، ومـقـابـل الـتـعـويـض الـعـادل لـهـا .
ولا يـجـوز إقـامـة أي بـنـاء أو عـوائـق فـي المـنـاطـق المـشـمـولـة بـحـقـوق الارتـفـاق ، أو إجـراء أي تـحـويـر فـي طـبـيـعـة أو جـهـة اسـتـعـمـال الأراضـي الـخـاضـعـة لـلارتـفـاق ، إلا بمـوجـب مـوافـقـة مـسـبـقـة مـن الـهـيـئـة ، وطـبـقـاً للـشـروط الـتـي يـصـدر بـهـا قـرار مـن الـرئـيـس .
 

 

الـفـصـل الـرابـع - أجـهـزة الـرصـد الـجـوي

المادة 11


تُحـدد بـقـرار مـن الـرئـيـس ، أنـواع أجـهـزة الـرصـد الـجـوي ومـحـطـات وأنـظـمـة الـرصـد الـجـوي ومـواصـفـاتـهـا ، وشـروط الـتـرخـيـص لاسـتـيـرادهـا أو اقـتـنـائـهـا ، والأجـهـزة الـتـي يـحـظـر إدخـالـهـا إلـى الـدولـة .

 

المادة 12


يُحـظـر اسـتـعـمـال أجـهـزة الـرصـد الـجـوي ومـعـلـومـات الـطـقـس والمـنـاخ لـغـيـر الـغـرض المـرخـص بـه .

 

المادة 13


تُصـدر الإدارة تـراخـيـص اسـتـيـراد أو اقـتـنـاء أجـهـزة الـرصـد الـجـوي وتـشـغـيـلـهـا ، بـعـد الـتـثـبـت مـن اسـتـيـفـائـهـا لـشـروط الـتـرخـيـص المـشـار إلـيـهـا بالمـادة (11) مـن هـذا الـقـانـون ، ويـجـب عـلـى المـرخـص لـه إجـراء المـعـايـرة الـدوريـة عـلـى الأجـهـزة وفـقـاً لمـعـاييـر المـنـظـمـة .
 

 

المادة 14


تتـولـى الإدارة وضـع ضـوابـط لـتـحـديـد المـواقـع المـرخـص باسـتـخـدام أجـهـزة الـرصـد الـجـوي فـيـهـا داخـل الـدولـة بالـتـنـسـيـق مـع الـجـهـات المـخـتـصـة ، ويـجـوز للإدارة إعـادة تـحـديـد تـلـك المـواقـع وفـقـاً للـضـوابـط المـشـار إلـيـهـا ، مـتـى اقـتـضـت الـضـرورة ذلـك .

 

المادة 15


يُحـظـر اسـتـخـدام أجـهـزة مـن شـأنـهـا الـتـشـويـش عـلـى مـحـطـات الـرصـد الـجـوي أو تـعـطـيـلـهـا .
ويـتـم تـوجـيـه إنـذار للـجـهـة أو الـشـخـص صـاحـب الـجـهـاز بإجـراء تـعـديـلات عـلـيـه خـلال المـدة الـتـي تـحـددهـا لـه الإدارة .
 

 

المادة 16


لا يـجـوز نـقـل مـلـكـيـة أجـهـزة الـرصـد الـجـوي المـرخـص بـهـا أو رهـنـهـا أو تـأجـيـرهـا أو الـتـصـرف فـيـهـا بأي وجـه ، قـبـل الـحـصـول عـلـى إذن مـسـبـق مـن الإدارة .
 

 

الـفـصـل الـخـامـس - ضـوابـط مـمـارسـة أنـشـطـة الأرصـاد الـجـويـة

المادة 17


عـلـى مـراكـز الـتـدريـب الـخـاصـة الـعـامـلـة فـي الـدولـة أو الـتـي تـرغـب فـي تـدريـس بـرامـج الأرصـاد الـجـويـة وفـروعـهـا ، والمـدربـين الـعـامـلـين فـي تـلـك المـراكـز ، الالـتـزام بالـشـروط والـضـوابـط الـتـي يـصـدر بـهـا قـرار مـن الـرئـيـس .

 

المادة 18


يـجـب عـلـى جـمـيـع الـجـهـات والمـؤسـسـات الإعـلامـيـة الـعـامـة والـخـاصـة اسـتـقـصـاء مـعـلـومـات الـنـشـرات الـجـويـة المـتـعـلـقـة بإقـلـيـم الـدولـة مـن الإدارة ، وتتـحـمـل تـلـك الـجـهـات والمـؤسـسـات مـسـؤولـيـة أي أخـطـاء نـاتـجـة عـن عـدم الـدقـة فـي الـنـقـل أو الـتـصـرف فـي المـعـلـومـات المـقـدمـة لـهـم مـن الإدارة .

 

المادة 19


تتـولـى الإدارة تـدريـب وتـأهـيـل الأشـخـاص الـذين يـتـم اخـتـيـارهـم وتـعـيينـهـم مـن قـبـل وسـائـل الإعـلام أو أي مـؤسـسـات أخــرى بـغـرض إذاعـة الـنـشـرات والـتـنـبـؤات الـجـويـة والـتـحـذيـرات والـتـحـلـيـلات .

 

المادة 20


 عـلـى الـهـواة الـذين يـرغـبـون فـي ممـارسـة أنـشـطـة الأرصـاد الـجـويـة ، قـيـد أسـمـائـهـم فـي سـجـلات خـاصـة تـنـشـأ لـهـذا الـغـرض لـدى الإدارة .

 

المادة 21


للهـواة الـقـيـام بمـتـابـعـة الأحـوال الـجـويـة وأحـوال الـطـقـس ونـشـر وتـداول بـيـانـات الـتـنـبـؤات والـتـقـاريـر والمـعـلـومـات الـخـاصـة بالأرصـاد الـجـويـة دون هـدف ربـحـي ، بـشـرط الـحـصـول عـلـى تـلـك الـبـيـانـات المـراد نـشـرهـا وتـداولـهـا مـن الإدارة . 

 

المادة 22


يُحـظـر اسـتـعـمـال أجـهـزة الـرصـد الـجـوي ومـعـلـومـات الـطـقـس والمـنـاخ لأي مـن الأغـراض الـتـالـيـة : 
1- تـبادل المـعـلـومـات مـع الـجـهـات الأخــرى الإقـلـيـمـيـة أو الـعـالمـيـة أو تـزويـد أي طـرف بأيـة مـعـلـومـات خـاصـة بالإدارة عـلـى خـلاف أحـكـام الـقـانـون.
2- إعـطـاء مـعـلـومـات سـريـة أو خـدمـات خـاصـة بالأرصـاد الـجـويـة إلـى أي جـهـة بـقـصـد مـسـاعـدتـهـا عـلـى المـسـاس بأمـن الـدولـة.
3- تـوجـيـه أو بـث أو تـداول أيـة مـعـلـومـات سـريـة أو بـيـانـات أرصـاد جـويـة تـكـون خـادعـة أو مـضـللـة مـع الـعـلم بأن تـلـك المـعـلـومـات أو الـبـيـانـات مـن المـحـتـمـل أن تـؤدي إلـى الإضـرار بـنـشـاط أي جـهـة عـامـة أو خـاصـة  يـتـصـل نـشـاطـهـا بالمـحـافـظـة عـلـى سـلامـة الأشـخـاص أو إنـقـاذهـم ، أو تـعـرض ذلـك الـنـشـاط أو أي شـخـص أو أي وسـيـلـة نـقـل للـخـطـر .
 

 

المادة 23


تتـولـى الإدارة الـتـنـسـيـق مـع الـجـهـات المـخـتـصـة بـشـأن الـذبـذبات الـراديـويـة والمـحـددة بمـوجـب تـعـلـيـمـات المـنـظـمـة ، وذبـذبات الاتـصـال بالأقـمـار الـصـنـاعـيـة ، والـذبـذبات الـتـي يـتـم حـجـزهـا بمـوجـب اتـفـاقـيـات مـع الـشـركـات الـوطـنـيـة والأجـنـبـيـة ، وكـذلـك ذبـذبات الأجـهـزة الـتـي يـرخـص بـدخـولـهـا مـن قـبـل الـجـهـات المـخـتـصـة لاسـتـخـدامـات الأرصـاد الـجـويـة .       
ولا يـجـوز لأي شـخـص الاسـتـيـلاء أو الـتـبـديـل أو الـتـغـييـر عـلـى الـذبـذبات المـشـار إلـيـهـا فـي الـفـقـرة الـسـابـقـة .
 

 

المادة 24


تُعـتـبـر مـعـلـومـات وبـيـانـات الأرصـاد الـجـويـة فـي حـالات الـحـرب  أو الـطـوارئ ، فـي حـكـم أسـرار الـدفـاع ، ويُحـظـر عـلـى أي شـخـص تـبادلـهـا أو إفـشـاؤهـا بأي وجـه . 
 

 

الـفـصـل الـسـادس - الـجـزاءات والـعـقـوبات

المادة 25


مـع عـدم الإخـلال بالـعـقـوبات الـجـنـائـيـة المـنـصـوص عـلـيـهـا فـي هـذا الـقـانـون ، للإدارة فـي حـالـة مـخـالـفـة أحـكـام المـواد (15) ، (16) ، (23/ فـقـرة ثـانـيـة) مـن هـذا الـقـانـون أو الـقـرارات الـصـادرة تـنـفـيـذاً لـه ، تـوقـيـع أي مـن الـجـزاءات الـتـالـيـة عـلـى المـخـالـف :
1- الإنـذار .
2- وقـف الـتـرخـيـص لمـدة لا تـزيـد عـلـى سـتـة أشـهـر .
3- حـجـز الأجـهـزة الـتـي تـسـبـب الـتـشـويـش .
4- إلـغـاء الـتـرخـيـص نـهـائـيـاً .
وفـي جـمـيـع الأحـوال يـجـوز إزالـة أيـة تـعـديـات بالمـخـالـفـة لـهـذا الـقـانـون إداريـاً ، وذلـك عـلـى نـفـقـة المـخـالـف . 
ولـصـاحـب الـشـأن أن يـتـظـلم إلـى الـرئـيـس مـن قـرارات الإدارة ، ويـسـري عـلـى هـذا الـتـظـلم الإجـراءات الـخـاصـة بـنـظـر الـتـظـلم والـبـت فـيـه المـنـصـوص عـلـيـهـا فـي المـادة (6) مـن هـذا الـقـانـون . 
 

 

المادة 26


يُـلـغـى الـتـرخـيـص فـي الـحـالات الـتـالـيـة :
1- انـقـضـاء أو زوال الـشـخـص المـعـنـوي وفـقـاً للـقـانـون . 
2- إذا تـم الـحـصـول عـلـيـه بـنـاءً عـلـى مـعـلـومـات أو مـسـتـنـدات مـزورة  أو مـضـللـة .
3- مـزاولـة الـعـمـل المـرخـص لـه بالمـخـالـفـة لأحـكـام هـذا الـقـانـون والـقـرارات المـنـفـذة لـه .
4- إذا فـقـد المـرخـص لـه شـرطـاً مـن شـروط مـنـح الـتـرخـيـص .
5- إذا ارتـأت الـهـيـئـة أن المـصـلـحـة الـعـامـة أو مـقـتـضـيـات الـنـظـام الـعـام تـسـتـلـزم ذلـك ، عـنـد وجـود خـطـر يـهـدد الأمـن ، أو يـخـل بالمـصـلـحـة الـعـامـة .
6- إذا لم يـتـقـدم المـرخـص لـه بـطـلـب لـتـجـديـد الـتـرخـيـص فـي الـوقـت المـحـدد بـهـذا الـقـانـون .
7- الـتـنـازل عـن الـتـرخـيـص دون الـحـصـول عـلـى مـوافـقـة كـتـابـيـة مـسـبـقـة مـن الإدارة .
 

 

المادة 27


مـع عـدم الإخـلال بأي عـقـوبـة أشـد ينـص عـلـيـهـا قـانـون آخــر ، يـعـاقـب بالـعـقـوبات المـنـصـوص عـلـيـهـا فـي المـواد الـتـالـيـة ، عـلـى ارتـكـاب الـجـرائـم المـنـصـوص عـلـيـهـا فـي كـل مـنـهـا .

 

المادة 28


يُعـاقـب بالـغـرامـة الـتـي لا تـزيـد عـلـى (30,000) ثـلاثـين ألـف ريـال ، كـل هـاوٍ قـام بـنـشـر أو تـداول بـيـانـات الـتـنـبـؤات والـتـقـاريـر والمـعـلـومـات الـخـاصـة بالأرصـاد الـجـويـة دون الـحـصـول عـلـيـهـا مـن الإدارة .

 

المادة 29


يُعـاقـب بالـغـرامـة الـتـي لا تـزيـد عـلـى(200,000) مـائـتـي ألـف ريـال ، كـل مـن خـالـف أيـاً مـن أحـكـام المـادتـين (10/فـقـرة ثـالـثـة) ، (23/فـقـرة ثـانـيـة) مـن هـذا الـقـانـون .

 

المادة 30


 يُعـاقـب بالـغـرامـة الـتـي لا تـزيـد عـلـى (300,000) ثـلاثـمـائـة ألـف ريـال ، كـل مـن قـام بمـمـارسـة أحـد أنـشـطـة الأرصـاد الـجـويـة بـغـرض ربـحـي دون الـحـصـول عـلـى تـرخـيـص مـن الإدارة .
 

 

المادة 31


يُعـاقـب بالـحـبـس مـدة لا تـجـاوز سـتـة أشـهـر وبالـغـرامـة الـتـي لا تـزيـد عـلـى (200,000) مـائـتـي ألـف ريـال ، أو بإحـدى هـاتـين الـعـقـوبـتـين ، كـل مـن :
1- قـام بـتـعـيين شـخـص لمـمـارسـة أعـمـال الـرصـد الـجـوي ، دون أن يـكـون مـرخـصـاً لـه قـانـونـاً بـذلـك . 
2- قـام باسـتـخـدام أجـهـزة مـن شـأنـهـا الـتـشـويـش عـلـى مـحـطـات الـرصـد الـجـوي أو مـخــرجـاتـهـا .
3- قـام بـنـقـل أو تـغـييـر أو اسـتـبـدال مـحـطـات الـرصـد الـجـوي دون الـحـصـول عـلـى مـوافـقـة الـهـيـئـة .
4- اسـتـعـمـل الأجـهـزة ومـعـلـومـات الـطـقـس والمـنـاخ لأغـراض تـخـالـف مـا تـم الـتـرخـيـص بـه .
 

 

المادة 32


يُعـاقـب بالـحـبـس مـدة لا تـجـاوز خـمـس سـنـوات وبالـغـرامـة الـتـي لا تـزيـد عـلـى (1,000,000) مـلـيـون ريـال ، أو بإحـدى هـاتـين الـعـقـوبـتـين ، كـل مـن :
1- قـام بـنـشـر تـحـذيـرات بأحـوال الـطـقـس أو المـنـاخ أو تـحـذيـرات مـن شـأنـهـا الـتـأثـيـر عـلـى شـؤون الـطـيـران والمـلاحـة الـجـويـة والـبـحـريـة ، بالمـخـالـفـة لأحـكـام هـذا الـقـانـون .
2- سـرق مـعـدات أو أجـهـزة الـرصـد الـجـوي أو أي مـن مـمـتـلـكـات الأرصـاد الـجـويـة .
3- أتـلـف أو عـطـل أو عـرض للـخـطـر المـنـشـآت أو المـبانـي أو مـحـطـات الأرصـاد أو الأسـلاك أو الأجـهـزة أو المـعـدات الـلازمـة لـتـشـغـيـل مـحـطـات الأرصـاد الـجـويـة .
4- خـالـف أيـاً مـن الـواجـبات أو المـحـظـورات الـواردة بـهـذا الـقـانـون خـلال فـتـرة الـحـرب أو الـطـوارئ .
5- قـام بـتـبادل المـعـلـومـات مـع الـجـهـات الأخــرى الإقـلـيـمـيـة أو الـعـالمـيـة أو تـزويـد أي طـرف بأيـة مـعـلـومـات خـاصـة بالإدارة عـلـى خـلاف أحـكـام الـقـانـون .
6- أعـطـى مـعـلـومـات سـريـة أو خـدمـات خـاصـة بالأرصـاد الـجـويـة إلـى أي جـهـة بـقـصـد مـسـاعـدتـهـا عـلـى المـسـاس بأمـن الـدولـة .
7- وجـه أو بـث أو تـداول أيـة مـعـلـومـات سـريـة أو بـيـانـات غـيـر صـحـيـحـة عـن الأرصـاد الـجـويـة .
 

 

المادة 33


يُعـاقـب المـسـؤول عـن الإدارة الـفـعـلـيـة للـشـخـص المـعـنـوي الـخـاص بـذات الـعـقـوبات المـقـررة عـن الأفـعـال الـتـي تـرتـكـب بالمـخـالـفـة لأحـكـام هـذا الـقـانـون ، إذا ثـبـت عـلمـه بـهـا أو كـان إخـلالـه بالـواجـبات الـتـي تـفـرضـهـا عـلـيـه تـلـك الإدارة قـد أسـهـم فـي وقـوع الـجـريـمـة .
ويـعـاقـب الـشـخـص المـعـنـوي إذا ارتـكـبـت باسـمـه أو لـحـسـابـه إحـدى الـجـرائـم المـنـصـوص عـلـيـهـا فـي هـذا الـقـانـون ، بـعـقـوبـة الـغـرامـة المـقـررة لـتـلـك الـجـريـمـة ، وذلـك مـع عـدم الإخـلال بالمـسـؤولـيـة الـجـنـائـيـة للـشـخـص الـطـبـيـعـي الـتـابـع لـه .
 

 

المادة 34


 يـكـون لمـوظـفـي الـهـيـئـة ، الـذين يـصـدر بـتـخـويـلـهـم صـفـة مـأمـوري الـضـبـط الـقـضـائـي ، قـرار مـن الـنـائـب الـعـام ، بالاتـفـاق مـع الـرئـيـس ، ضـبـط وإثـبات الـجـرائـم الـتـي تـرتـكـب بالمـخـالـفـة لأحـكـام هـذا الـقـانـون ، ولـهـم فـي سـبـيـل ذلـك دخـول المـبانـي والمـنـشـآت وغـيـرهـا مـن الأمـاكـن ذات الـصـلـة والاطـلاع عـلـى الـسـجـلات والـدفـاتـر والمـسـتـنـدات وفـحـص المـعـدات وأجـهـزة الـرصـد الـجـوي ، وأيـة أدوات أخــرى ذات صـلـة بالـرصـد والـتـنـبـؤات الـجـويـة ، وطـلـب الـبـيـانـات والإيـضـاحـات الـتـي يـرونـهـا ضـروريـة .

 

المادة 35


يـجـوز للـرئـيـس أو مـن يـفـوضـه الـصـلـح فـي الـجـرائـم المـنـصـوص عـلـيـهـا فـي هـذا الـقـانـون ، قـبـل تـحـريـك الـدعـوى الـجـنـائـيـة أو أثـنـاء نـظـرهـا وقـبـل الـفـصـل فـيـهـا بـحـكـم نـهـائـي .
ويـكـون الـصـلـح فـي المـخـالـفـات المـحـددة بالـجـدول المـرفـق بـهـذا الـقـانـون ، مـقـابـل سـداد المـبالـغ المـبـينـة قـرين كـل مـنـهـا ، وإزالـة أسـباب المـخـالـفـة .
ويـكـون الـصـلـح فـي المـخـالـفـات غـيـر المـنـصـوص عـلـيـهـا فـي الـجـدول المـرفـق بـهـذا الـقـانـون ، مـقـابـل سـداد نـصـف الـحـد الأقـصـى لمـبـلـغ الـغـرامـة المـقـررة لـكـل مـنـهـا ، وإزالـة أسـباب المـخـالـفـة .
ويـتـرتـب عـلـى الـصـلـح عـدم جـواز تـحـريـك الـدعـوى الـجـنـائـيـة  أو انـقـضـاؤهـا بـحـسـب الأحـوال .
 

 

الـفـصـل الـسـابـع - أحـكـام خـتـامـيـة

المادة 36


لأغـراض تـطـبـيـق أحـكـام هـذا الـقـانـون ، يـجـوز للهـيـئـة الـتـعـاقـد مـع أي طـرف مـحـلـي أو أجـنـبـي لأي مـن الأغـراض الـتـجـاريـة ذات المـنـفـعـة المـشـتـركـة ، كـمـا يـجـوز لـهـا إنـشـاء شـركـات تـجـاريـة ، مـنـفـردة أو بالمـسـاهـمـة مـع الـغـيـر ، وذلـك بـعـد مـوافـقـة مـجـلـس الـوزراء ، بـنـاءً عـلـى اقـتـراح الـوزيـر .
 

 

المادة 37


يـجـب عـلـى الـجـهـات والمـؤسـسـات الـتـي تـمـارس أنـشـطـة أو تـقـدم خـدمـات تتـعـلـق بالأرصـاد الـجـويـة ، تـوفـيـق أوضـاعـهـا خـلال سـتـة أشـهـر مـن تـاريـخ الـعـمـل بـهـذا الـقـانـون ، ويـجـوز بـقـرار مـن مـجـلـس الـوزراء ، وفـقـاً لمـقـتـضـيـات المـصـلـحـة الـعـامـة ، مـد هـذه المـهـلـة لمـدة أو مـدد أخــرى ممـاثـلـة .
 

 

المادة 38


يُـصـدر الـوزيـر أو الـرئـيـس ، بـحـسـب الأحـوال ، الـقـرارات الـلازمـة لـتـنـفـيـذ أحـكـام هـذا الـقـانـون ، وإلـى حـين صـدور تـلـك الـقـرارات ، يـسـتـمـر الـعـمـل بالـقـرارات واللـوائـح والـنـظـم المـعـمـول بـهـا حـالـيـاً ، بمـا لا يـتـعـارض مـع أحـكـام هـذا الـقـانـون .
 

 

المادة 39


عـلـى جـمـيـع الـجـهـات المـخـتـصـة ، كـل فـيـمـا يـخـصـه ، تـنـفـيـذ هـذا الـقـانـون . ويُـنـشـر فـي الـجـريـدة الـرسـمـيـة .
