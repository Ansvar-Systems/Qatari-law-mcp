مـــرســـوم رقـــم (96) لـسـنـــة 2025 بـالـتـصـديــق عـلـى مـذكـرة تـفـاهـم بـيـن حـكـومـة دولـة قـطـر وحـكـومـة دولـة الـكـويـت فـي مـجـال الـعـمـل الخـيـري والإنـسـانـي 96 / 2025
عدد المواد: 2
فهرس الموضوعات

المواد (1-2)



نـحـن تمـيـم بـن حـمــد آل ثـانـي           أمـيـر دولـة قـطـــر ،
بـعـد الاطــلاع عـلـى الـدسـتـور،
وعـلـى وثـيـقـة الـتـصـديـق الـصــادرة فـي الــسـابـع عـشـر مـن شـهــر صـفـر عـام 1447 هــجـريـة ، المـوافــق لـلـحـادي عـشـر من شـهـر أغـسـطـس عـام 2025 مـيـلاديـة ،
وعـلـى اقـتـراح رئـيـس مـجـلـس الـوزراء ووزيـر الخـارجـيـة ،
وعـلـى مـشـروع المـرسـوم المـقـدم مـن مـجـلـس الـوزراء ،
رسمنـا بمـا هـو آت :
 

المواد

المادة 1


صُـودق عـلـى مـذكـرة تـفـاهـم بـيـن حـكـومـة دولـة قـطـر وحـكـومـة دولـة الـكـويـت فـي مـجـال الـعـمـل الخـيـري والإنـسـانـي ، المـوقـعـة بمـديـنـة الـدوحـة بــتـاريـخ 04/ 09/ 2024 ، المـرفـق نـصـهـا بـهـذا المـرسـوم ، وتـكـون لـهـا قـوة الـقـانـون ، وفـقـاً لـلـمـادة (68) مـن الـدسـتـور .

 

المادة 2


عـلـى جـمـيـع الجـهـات المـخـتـصـة ، كـل فـيـمـا يـخـصـه ، تـنـفـيـذ هـذا المـرســوم . ويُـعـمـل بـه مـن تـاريـخ صــدوره . ويُـنـشــر فـي الجـريــدة الـرسـمـيـة .
