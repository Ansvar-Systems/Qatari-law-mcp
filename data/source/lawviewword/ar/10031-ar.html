مرسوم رقم (41) لسنة 2025 بالتصديق على الاتفاق التكميلي لاتفاقية بين حكومة دولة قطر وبرنامج الأمم المتحدة الإنمائي حول إنشاء مكتب لبرنامج الأمم المتحدة الإنمائي في الدوحة - قطر 41 / 2025
عدد المواد: 2
فهرس الموضوعات

المواد (1-2)



نـحـن تمـيـم بـن حـمــد آل ثـانـي           أمـيـر دولـة قـطـــر ،
بـعـد الاطــلاع عـلـى الـدسـتـور ،
وعـلـى وثـيـقـة الـتـصـديـق الـصــادرة فـي الــتـاسـع مـن شـهــر رمـضـان عـام 1446 هــجـريـة ، المـوافــق للـتــاسـع من شـهـر مـــارس عـام 2025 مـيـلاديـة ،
وعـلـى اقـتـراح رئـيـس مـجـلـس الـوزراء ووزيـر الخـارجـيـة ،
وعـلـى مـشـروع المـرسـوم المـقـدم مـن مـجـلـس الـوزراء ،
رسمنـا بمـا هـو آت :
 

المواد

المادة 1


صُـودق عـلـى الاتـفـاق التـكـمـيـلـي لاتـفـاقـيـة بـين حـكـومـة دولـة قـطـر وبـرنـامـج الأمـم المـتـحـدة الإنمـائـي حـول إنـشـاء مـكـتـب لـبـرنـامـج الأمـم المـتـحـدة الإنمـائـي في الدوحـة – قــطــر ، المـوقـع بمـديـنـة الـدوحـة بـتـاريـخ 12/12/2024 ، المـرفـق نـصـه بـهـذا المـرسـوم ، وتـكـون لـه قـوة الـقـانـون ، وفـقـاً لـلـمـادة (68) مـن الـدسـتـور .

 

المادة 2


عـلـى جـمـيـع الجـهـات المـخـتـصـة ، كـل فـيـمـا يـخـصـه ، تـنـفـيـذ هـذا المـرســوم . ويُـعـمـل بـه مـن تـاريـخ صــدوره . ويُـنـشــر فـي  الجـريــدة الـرسـمـيـة .
