قرار أميري رقم (44) لسنة 2022 بالهيكل التنظيمي لوزارة الثقـافـة 44 / 2022
عدد المواد: 23
فهرس الموضوعات

المواد (1-23)



نحن تميم بن حمد آل ثاني أمير دولة قطر،
بعد الاطلاع على الدستور،
وعلى قانون الموارد البشرية المدنية الصادر بالقانون رقم (15) لسنة 2016، المعدل بالقانون رقم (23) لسنة 2022،
وعلـى القـرار الأميــري رقـم (7) لسنـة 2016 بالهيكل التنظيمـي لوزارة الثقـافـة والـريـاضـة،
وعلـى القـرار الأميــري رقـم (57) لسنـة 2021 بتعيين اختصـاصـات الوزارات،
وعلى اقتراح مجلس الوزراء،
قررنا ما يلي

المواد

المادة 1


يتـكـون الهيكل التنظيمـي لـوزارة الثقـافـة مـن الـوحـدات الإداريـة المبينة بالخريطة التنظيميـة المـرفـقـة بهذا القـرار، وهي:
أولاً: الوحدات الإداريـة التـابـعـة للـوزيـر:
۱- مـكـتـب الـوزيـر.
٢- المـكـتـب الفـنـي.
٣- إدارة التدقيـق الـداخـلي.
٤- إدارة التخطيط والجـودة والابتـكار.
٥- إدارة الشـؤون الـقـانـونيـة.
 
ثانياً: الوحـدات الإداريـة التـابـعـة لوكيـل الوزارة:
۱- مكتـب وكيـل الـوزارة.
٢- إدارة الـتـعـاون الدولي.
٣- إدارة العـلاقـات الـعـامـة والاتـصـال.
٤- إدارة الموارد البشـريـة.
٥- إدارة الشـؤون المـاليـة والإداريـة.
٦- إدارة نـظـم المـعـلـومـات.
ثالثاً: الوحدات الإدارية التـابـعـة لوكيل الوزارة المسـاعـد للشؤون الثقـافيـة:
۱- مكتـب وكيـل الـوزارة المسـاعـد.
٢ - إدارة المكتبـات.
٣- إدارة الإصـدارات والتـرجـمـة.
٤- إدارة الـثـقـافـة والـفـنـون.
٥- إدارة الـتـراث والـهـويـة.
رابعـاً: الوحـدات الإداريـة التـابـعـة لوكيـل الـوزارة المسـاعـد للشـؤون الإعـلاميـة:
۱- مكتـب وكيـل الـوزارة المسـاعـد.
٢- إدارة الـشـؤون الإعـلاميـة.
٣- إدارة التـطـويـر الإعـلامـي.
٤- إدارة البـث الإذاعـي.
٥- إدارة المـطـبـوعـات والمـصـنـفـات الـفـنـيـة.

 

المادة 2


يختص المكتـب الفـني بما يلي:
١- إبداء الرأي الفـني في المـوضـوعـات التي تحـال إليـه مـن الوزيـر.
۲- دراسـة وتقييم التقـاريـر والمشـروعـات والبـرامـج المتعلقة بأنشـطـة الوزارة، التي تُحـال إليـه مـن الوزيـر.
۳- بحـث ودراسـة ومـتـابـعـة المـوضـوعـات الـتـي تُحـال إليـه مـن الوزيـر، بالتنسيق مع الوحـدات الإداريـة المعنيـة.
٤- إعـداد ملخص للمـواضـيـع والتقارير والـدراسـات الـتـي يـلـزم عـرضـها على الـوزيـر، وإبـداء الـرأي بشـأنـها.

 

المادة 3


تختص إدارة التـدقيـق الـداخـلـي بـمـا يـلـي:
١- إعداد مشروع خـطـة التـدقيـق السنـويـة المبنيـة علـى المخـاطـر للوزارة، ورفعـها للوزيـر لاعتمـادهـا، وإعـداد تقاريـر بـنـتـائـج التـدقـيـق.
۲- مـراقبـة مـدى الالتزام بتنفيذ القـوانـين واللوائح والقرارات المنظمـة للعمـل.
۳- مـراجعـة التعليمات الماليـة، واقتــراح مـا يـلـزم لـهـا مـن تعـديـلات، ومـراقبـة تنفيـذهـا.
٤- مـراقبـة الأعـمـال الإداريـة وإجـراءات تعيين الموظفين، وإجـازاتـهم، وتـرقـيـاتـهم، وغير ذلـك ممـا يتعلـق بشـؤون الموظفين، والتحقـق مـن الالتزام بالقوانين واللوائح والقرارات المنظمة لذلك.
٥- مـراقبـة الأعمـال المـالـيـة والمحاسبيـة، وسنـدات الصـرف، وإجـراءات القيـد في السجـلات المحاسبيـة، والتـأكـد مـن تطبيـق القـواعـد المعتمـدة في هـذا الشـأن.
٦- تقييم المخاطـر المتعلقة بالتدقيـق المالي والإداري، واقتراح إجـراءات الحـد منـها.
۷- متـابعـة الإجـراءات التـصـحـيـحـيـة المطـلـوبـة مـن الوحدات الإداريـة المعـنـيـة، وفـق نـتـائـج التـدقيـق، وإعـداد التقارير الدوريـة بشـأنـها.
۸- التـأكـد مـن سـلامـة الأمـوال والعـهـد والأثاث والأدوات والمبـاني والموجودات الأخرى التي تملكها الوزارة أو تخضع لإشـرافـها.
٩- الاشتـراك في أعمـال الجـرد الـدوري والمفاجئ على المخـازن والعهـد والتـأكـد مـن دقـة التسجيـل بـالـدفـاتـر وسـلامـة الإجـراءات المطبقـة.
۱۰- متـابعـة نتائج تقاريـر الجهـات الـرقـابيـة، وإبـداء التوصيـات المنـاسبـة بشأنها.

 

المادة 4


تختص إدارة التخطيط والجـودة والابتكار بمـا يـلـي:
١- إعـداد الخطـة الاستراتيجيـة للوزارة، بالتنسيق مع الجهات المختصة والوحـدات الإدارية المعنيـة، واتخاذ إجـراءات اعتمـادهـا.
۲- حـصـر الـتـحـديـات القـائمـة، وتحـديـد وتقييـم المخاطـر المؤسسيـة المتـوقـع حـدوثـها، والتي قد تؤثر على تحقيـق أهـداف الـوزارة، وتقديم المقترحات بشأنها، بالتنسيق مع الوحدات الإداريـة المعنيـة.
٣- إدارة وتجـهـيـز خـطط بديلة تضمن استمراريـة تشغيـل الأعمـال بالـوزارة في حـالات الطوارئ والأزمـات، بالتنسيـق مـع الـوحـدات الإداريـة المختـلفـة كـل فيما يخـصـها.
٤- رفـع مـشـروعـات الخطط التنفيـذيـة والمشـاريـع والبـرامـج الخـاصـة بالوحدات الإدارية المختـلفـة، للاعتمـاد مـن قبـل الـوزيـر.
٥- متـابعـة تنفيذ الخطط المتعلقة بأنشـطـة الـوزارة، وتقييـم نـتـائـجـها، ورفـع تـقـاريـر دوريـة بشـأنـها.
٦- دراسة المشـاكـل والمعـوقـات الـتـي تـواجـه تنفيذ الخطة الاستراتيجيـة والخطط التنفيـذيـة، واقتراح الحلول المنـاسبـة لـهـا.
۷- تقييـم جـودة الخـدمـات المقـدمـة مـن الـوحـدات الإداريـة، بالتنسيـق مع الجهـة المختـصـة.
۸- مـراجـعـة وتقييـم جـودة الأداء علـى مـسـتـوى الوحدات الإداريـة، ووضـع مـقـتـرحـات لـرفـع الـكفـاءة العـامـة بـها.
٩- التحـقـق مـن انسجام نـظـم وأساليـب العـمـل مـع اختصـاصـات وأهـداف الوزارة، واقتراح خطـط تـطـويـر تـلـك النـظـم والأساليـب، وتبسيط الإجراءات، بالتنسيق مع الوحدات الإداريـة المعنيـة والجهـة المختصة.
۱۰- دراسة مشـاكل ومعـوقـات العمـل بالــوزارة، وبحـث أسبـابـها، واقتراح الحلول المنـاسبـة لـها.
۱۱- تلقـي المبادرات والمقترحـات والأفكار الإبـداعـيـة، ودراستها بالتنسيق مع الوحدات الإدارية المعنيـة والجـهـة المختصة.
۱۲- رصد وتجميع وتحـلـيـل البيانات الإحصائيـة الخـاصـة بأنشـطـة الـوزارة، والتنسيق مع الوحدات الإدارية المختـصـة بشـأن التقارير الإحصائيـة المطـلـوبـة مـن الـوزارة.
۱۳- التنسيـق مـع الوحدات الإداريـة المعنيـة والجهـة المختصة فيما يتعلـق بالمؤشـرات الـدوليـة المتعلقة بأنشـطـة واخـتـصـاصـات الـوزارة.

 

المادة 5


تختص إدارة الشـؤون الـقـانـونيـة بمـا يـلـي:
١- بحـث ودراسـة ومـتـابـعـة المسـائل القـانـونيـة الخـاصـة بنشـاط الـوزارة، بالتنسيق مع الوحدات الإداريـة المعنيـة.
٢- إعداد مشـروعـات الأدوات التشريعيـة الخـاصـة بالوزارة، وإبـداء الـرأي في مشـروعـات القـوانـين التي تحـال إليـها.
٣- إبـداء الـرأي الـقـانـوني في المـوضـوعـات التي تُحـال إليـها.
٤- إعـداد مشـروعـات العقـود والاتفاقيـات ومـذكـرات التفـاهـم المتعلقة باخـتـصـاصـات الوزارة، بالتنسيق مع الوحـدات الإداريـة المعنيـة.
٥- التحقيـق في الوقـائـع والمخالفـات المنسوبـة لموظفي الـوزارة، وإعـداد المـذكـرات اللازمة بنتائج التحقيـق مـشـفـوعـة بـالـرأي القـانـوني والتوصيـات، وعـرضـها على السلطة المختصة، ومتابعة تنفيـذ مـا يتخذ مـن قـرارات بشـأنـها.
٦- متـابعـة المنـازعـات والـقـضـايـا الـتـي تـكـون الـوزارة طـرفـاً فيهـا، بالتنسيـق مـع الجـهـات المختـصـة

 

المادة 6


تختـص إدارة التـعـاون الدولي بمـا يـلـي:
١- إعداد أوراق العمـل المتعلقة بالمؤتمرات والندوات والاجتمـاعـات المحـلـيـة والإقليميـة والدوليـة فـيـمـا يـخـص نشـاط الـوزارة، بالتنسيـق مـع الوحـدات الإداريـة المعنيـة والجـهـات المختصة.
٢- الإعـداد لحضور المؤتمرات والاجتمـاعـات الـتي تعقـدهـا المنظمـات والهيئـات الإقليميـة والدوليـة المختـصـة بقضـايـا الـوزارة، الـتي تشـارك الدولـة في عضـويـتـها.
٣- إعـداد التقاريـر والـردود التي تطلبهـا المنظمات والهيئات الإقليميـة والدوليـة فيمـا يـخـص نشـاط الـوزارة، بالتنسـيـق مـع الـوحـدات الإدارية المختصة.
٤- دراسـة ومتـابـعـة تنفيـذ تـوصـيـات المنظمـات والهيئـات الإقليميـة والدوليـة، فيمـا يـخـص نشـاط الـوزارة، بالتنسيق مـع الجهـات المخـتـصـة بـالـدولـة.
٥- إعداد التقارير اللازمـة بشـأن إسـهـامـات المنظمات والهيئات الإقليمية والدوليـة في تنميـة وتـدعـيـم الأنشـطـة المختلفـة في مجـالات عـمـل الوزارة، بالتنسيق مع الجهات المخـتـصـة.
٦- متـابـعـة سـداد المسـاهـمـات المـاليـة للدولـة في المنظمـات المعنيـة بمجـالات عـمـل الـوزارة الـتـي تـكـون الدولـة عـضـواً فيهـا، بالتنسيـق مـع الجـهـات المخـتـصـة.
۷- دراسة السبـل الـكـفـيـلـة بـتـفـعـيـل دور الدولة في مجـال نـشـاط الـوزارة مع المنظمات العـربـيـة والإقـلـيـمـيـة والـدوليـة.
۸- جمـع المعلـومـات والبحـوث الصـادرة عـن المنظمات والهيئـات ذات الصلة بقضـايـا الـوزارة مـحـلـيـاً وإقليمياً ودولياً، ونشـرهـا بـغـرض الاسـتـفـادة مـنها.

 

المادة 7


تختص إدارة العـلاقـات الـعـامـة والاتـصـال بمـا يـلـي:
١- إصـدار النشرات والبـرامـج الإعـلاميـة التي تهدف إلى التعـريـف بـدور الوزارة وأنشطتها واخـتـصـاصـاتـها، بالتنسيق مع الوحدات الإداريــة المعنيـة.
٢- رصـد ومـتـابـعـة مـا يُنشـر في الصـحـف ووسـائـل الإعـلام ومـواقـع التـواصـل الاجتماعي مـن مـوضـوعـات تتعلق بالوزارة واخـتـصـاصـاتـها، وعـرضـها على المسؤولين، وإعـداد الـردود المنـاسبـة، بالتنسيـق مـع الـوحـدات الإداريـة المعـنـيـة والجهات المخـتـصـة.
٣- التنسـيـق مـع الـوحـدات الإدارية المختصـة بشـأن إعـداد خطط وبـرامـج التـوعـيـة والتثقيـف وتطبيقهـا مـن خـلال الفعاليـات والأنشـطـة المختـلفـة بالتنسيق مع الجهات المعنيـة.
٤- تلقـي طـلـبـات وشكاوى المـراجـعـين وإحـالـتـها إلى الوحدات الإداريـة المعنيـة، والرد على اسـتـفـسـاراتـهـم.
٥- القيام بترتيبات السـفـر والضيـافـة والإقـامـة لضيـوف الوزارة وللـوفـود الـزائـرة والمغادرة، بالتنسيـق مـع الوحدات الإداريـة المعنيـة.
٦- تنظيـم الأنشطـة الثقافيـة والاجتماعيـة والرياضيـة لموظفي الوزارة، والإشراف على الاحتفالات التي تقيمها أو تشارك فيـها الوزارة.
۷- تنظيم المؤتمرات والندوات والمعارض التي تقيمهـا الـوزارة، وإعـداد الموازنة اللازمـة لذلك، بالتنسيق مع الوحدات الإداريـة المعنيـة.
۸- اسـتـقبـال طلبـات كبار السن والأشخاص ذوي الإعـاقـة، بالتنسيـق مـع الوحـــدات الإداريـة المختصة، لإنجـاز مـعـامـلاتـهـم وتـذليــل أي مشكلات أو صـعـوبـات قـد تـواجـهـهـم.

 

المادة 8


تختص إدارة الموارد البـشـريـة بما يلي:
١- تنفيـذ القـوانـين واللوائح والنـظـم المتـعـلـقـة بشـؤون الموظفين.
٢- إعـداد الدراسـات الخـاصـة بتخطيـط المـوارد البشـريـة بالـوزارة، وتحـديـد احتياجـات الـوزارة مـن الوظـائـف والموظـفـين، بالتنسيـق مع الجهـة المختـصـة والوحـدات الإداريـة المختـلفـة.
٣- إعداد مشروع وصـف وتصنيـف وترتيب الوظائف، ومتـابـعـة تنفيـذه، وتطـويـره.
٤- إعداد مشروع مـوازنـة البـاب الأول، بالتنسـيـق مـع الوحدات الإداريـة المعنية.
٥- تنفيذ إجـراءات تقييـم الأداء للمـوظـفـين.
٦- القيـام بإجـراءات التـعـيـين والنـقـل والنـدب والإعـارة.
۷- دراسـة استحـقـاق الموظفين للإجـازات، وفقاً للـقـانـون.
٨- إعـداد الدراسـات الخـاصـة بالهيـكل الوظيـفـي للـوزارة، وتقـديـم المقتـرحـات اللازمة لتطـويـر التنظيـم الإداري بـها.
٩- تحـديـد الاحتياجـات التـدريبيـة لمـوظـفـي الـوزارة، بالتنسيـق مـع الـوحـدات الإدارية المختلفة، ومتابعة تنفيذها، وتقييـم مـدى الاسـتـفـادة مـنـها.
۱۰- اتخاذ الإجراءات اللازمـة لتخصيص السـكـن الحـكـومي وصـرف بـدل الأثـاث لموظفي الوزارة، بالتنسيـق مـع الجـهـات المخـتـصـة.
۱۱- وضـع وتحـديـث قـواعـد البيـانـات الخـاصـة بشـؤون موظفي الوزارة.

 

المادة 9


تختـص إدارة الشؤون المـاليـة والإداريـة بمـا يـلـي:
١- تنفيـذ القـوانـيـن والـلـوائـح والنـظـم المـاليـة والإداريـة المتعلقـة بـعـمـل الوزارة.
٢- إعـداد مشـروع الموازنـة السـنـويـة للـوزارة، وحـسـابـها الخـتـامي، بالتنسيـق مـع الوحـدات الإداريـة المعـنـيـة.
۳- توفير احتيـاجـات الـوزارة ووحداتها الإدارية المختلفـة مـن المـسـتـلـزمـات والأجهـزة اللازمة لأداء مهـامـها، بالتنسيق مـع الوحـدات الإداريـة المعنية.
٤- القيام بأعمـال المشـتـريـات والمنـاقـصـات والمزايـدات، وفـق الأنظمـة والقـواعـد المعمـول بـها في الـدولـة.
٥- الإشـراف على تنفيـذ الاعتمـادات، وتـدقـيـق حـسـابـات الإيرادات والمصـروفـات.
٦- تسـلـم وتـسـلـيـم وتصنيف وقـيـد البـريـد الـوارد والصـادر.
۷- تنـظـيـم أرشـيـف الوزارة، وحفظ الوثـائـق، وفقاً لأحـدث الطـرق.
٨- إعـداد مـسـتـنـدات الصـرف، وسـائـر المعـامـلات الماليـة الأخـرى.
۹- تحـصـيـل الـرسـوم ومقـابـل الخـدمـات التي تؤديـهـا الـوزارة.
۱۰- القيـام بـجـمـيـع أعـمـال الخـدمـات الإداريـة.
١١- الإشراف على مخازن الوزارة.

 

المادة 10


تختص إدارة نـظـم المـعـلـومـات بما يلي:
١- إعـداد خطط وسيـاسـات اسـتـخـدام الحاسب الآلي في أنشـطـة الـوزارة، ومتـابـعـة تـنـفـيـذهـا.
۲- برمجـة وحـفـظ واستـرجـاع وتطـويـر نظم المعلـومـات والبيانات اللازمـة لأنشـطـة الـوزارة.
۳- تأمين الأنظمـة والبرمجيـات والشبكات والبنية التحتيـة لنـظم المعـلـومـات مـن مـخـاطـر الحـوادث الإلكتـرونيـة.
٤- إعـداد خطط التعـافـي مـن آثـار الحـوادث الإلـكـتـرونيـة حـال وقـوعـها أو تقليـل حـدوثها أو تصعيـدهـا، وفـق الـسيـاسـات والمبادئ التـوجـيـهيـة لأمـن المعـلـومـات، بالتنسـيـق مـع الجـهـة المخـتـصـة.
٥- تـوفـيـر وصيـانـة الأجهـزة والبـرامـج والشبكات الإلكتـرونيـة اللازمـة لنـظم العمـل الآليـة بالـوزارة، بالتنسيـق مـع الوحـدات الإداريـة المعـنيـة.
٦- تصميـم وتشـغـيـل وإدارة قـواعـد البيانات والمعلـومـات بأنـواعـها المختـلفـة.
۷- تقـديـم الدعم الفني اللازم لمسـتـخـدمي الشبكات الإلكتـرونيـة بالوزارة، وتدريبـهم على استخـدام الأنظمـة والبـرامـج وأجـهـزة الحـاسـب الآلي وملحـقـاتـها.
٨- إنشاء ومتـابعـة وتحديث موقع الوزارة على شبكة المعلـومـات الـدوليـة.

 

المادة 11


 
تختـص إدارة المـكـتـبـات بمـا يـلـي:
١- الإشراف على المـكـتـبـات العـامـة وتطـويـرهـا.
٢- إصـدار التراخيص للمكتبات غيـر التجـاريـة، والإشراف والـرقـابـة عليـهـا وفـق الخطط والبـرامـج والـسـيـاسـات المعتمـدة.
٣- إعـداد الخطط والسيـاسـات والبـرامـج بشأن دار الكتب القطـريـة.
٤- تنفيذ الإجـراءات اللازمة لإيداع المصنفـات بـدار الكتـب القـطـريـة.
٥- وضع الخطط والبـرامـج للتشـجـيـع عـلـى القـراءة.
٦- المشـاركـة في المعـارض والمؤتمرات والندوات المحليـة والـدوليـة في مجـال الـكـتـب والمـكـتـبـات.
٧- الإشـراف على فـعـالـيـات مـعـرض الـكـتـاب.
٨- تـوفـيـر مـخـتـلـف مـصـادر المعـلـومـات اللازمـة لـتـقـديـم الخـدمـات المعـلـومـاتيـة، والبـبـلـيـوغـرافيـة، للهيئـات ولجمهـور المثـقـفـين والدارسين والبـاحـثـين.
٩- وضـع شـروط ومـعـايـيـر إنـشـاء المـكـتـبـات.

 

المادة 12


تختص إدارة الإصـدارات والتـرجـمـة بمـا يـلـي:
١- إصـدار الـكـتـب للمؤلفين في مجـالات عـمـل الـوزارة.
۲- تـرجـمـة الأعمـال المتميـزة مـن اللغـات الأخـرى إلى اللغـة العـربـيـة، وتـرجمـة الأعمـال الإبـداعـيـة المنتـقـاة مـن اللغـة العـربـيـة إلى اللغـات الأخـرى.
٣- إصـدار الـكـتـب والنـشـرات التي تتضـمـن تعـريـفـاً بالدولـة في مختلـف النـواحـي، وتزويـد السـفـارات والبعثات الدبلــومـاسـيـة بـهـا، بالتنسـيـق مـع الجـهـات المعـنيـة بالدولـة.
٤- إصـدار المـجـلات المتـخـصـصـة في مجـالات عـمـل الـوزارة.
٥- تـرجـمـة الوثـائـق والـكـتـب والبـحـوث العلميـة والمكاتبـات الأجنبيـة المتعلقة باخـتـصـاصـات الـوزارة.

 

المادة 13


تختـص إدارة الثـقـافـة والـفـنـون بمـا يـلـي:
١- الإشـراف الـعـام عـلـى الجـهـات العـامـلـة في مجـال الثقـافـة، وإصـدار التراخيص اللازمـة بشـأنـها.
۲- دعـم ورعـايـة الحـركـة الثقافيـة والفنيـة، وتـوفيـر سـبـل تـطـويـرهـا.
٣- الإشراف على تنظيـم البيـئـة الثـقـافيـة في الدولـة.
٤- تنظيـم الفـعـالـيـات الثقافيـة والفنيـة، والاحتـفـاء بـهـا فـي المنـاسبـات الوطنيـة.
٥- وضـع شـروط ومعاييـر إنشـاء الأنديـة والمـراكـز واللجـان والجمعيـات ذات الطابع الثقافي، وإصـدار التراخيص اللازمة وفقاً للقـانـون.
٦- الترخيـص بإنشـاء الفـرق الفنيـة ومـزاولـة أعمـالـها، ومتـابعـة أعـمـال تـلك الفـرق.
٧- أرشـفـة الإنتـاج الفني والثـقـافـي.
۸- اقـتـراح حـل أو دمـج المـراكـز المعنيـة بالثـقـافـة والفنـون الـتـي تشـرف عليها الوزارة.
٩- الإشراف على مـسـرح قـطـر الـوطـني.
١٠- إعـداد السيـاسـات والدراسات والخطط التطـويـريـة المتعلقة بالثقـافـة والفـنـون، بالتنسـيـق مـع الإدارات المخـتـصـة.
۱۱- منح تراخيص إقـامـة الأعـمـال الـفـنـيـة، والـرقـابـة عـلـيـها.
۱۲- رصد وتحـلـيـل الأنشـطـة الثقـافيـة ذات العـلاقـة في الداخـل والخارج.
۱۳- تطـويـر عـلاقـات فـاعـلـة مع المثقفين والمؤسسات الثـقـافـيـة.
١٤- تنـظيـم وإدارة السنوات الثقافية في إطار التعاون الثـقـافـي الـدولي، بالتنسيق مع الجهات المعنية.

 

المادة 14


تختص إدارة التـراث والـهـويـة بمـا يـلـي:
١- الإشراف العـام على الجـهـات العـامـلـة في مجـال التـراث والهـويـة.
٢- وضـع شـروط ومعـايـيـر إنشـاء الفـرق الشعبيـة والمـراكـز التـراثـيـة، وإصـدار التراخيص اللازمـة لـهـا ومـتـابـعـة أعـمـالـها.
٣- الإشراف على المتـاحـف والآثـار، بالتنـسـيـق مـع الجـهـات المعـنيـة.
٤- إعـداد السيـاسـات والدراسات والخطط المتعلقة بالتراث، بالتنسـيـق مـع الـوحـدات الإداريـة المعـنيـة.
٥- دعـم ورعـايـة الأنشـطـة والحـرف التـراثـيـة.
٦- الحفاظ علـى التـراث الشعبي، والتـراث الثقـافـي الـوطني، وإبــراز خـصـائـصـه، وإعداد الدراسات والبحوث المتعلقة بـه.
٧- الإشراف على تنـظـيـم الفـعـاليـات التـراثـيـة في الدولـة.
۸- تنـظـيـم الفـعـالـيـات الـتـراثـيـة، والاحتـفـاء بـها في المناسبات الوطنيـة.
٩- إقـامـة الفـعـاليـات الـتـراثـيـة في الداخـل والخـارج، واسـتـقـبـال الفعـالـيـات التـراثـيـة مـن الدول الأخـرى، بالتنسيق مع الجهات المعنيـة.
۱۰- دعـم التـراث الثقافي الوطني والتـراث الشعبي.
١١- العـمـل علـى تـعـزيـز مـقـومـات الـهـويـة الـوطـنـيـة.
۱۲- أرشـفـة التـراث القـطـري والشـعـبي.
۱۳- منـح تـراخـيـص إقـامـة الأنشـطـة التـراثـيـة.

 

المادة 15


تختص إدارة الشؤون الإعـلاميـة بمـا يـلـي:
۱- اقتراح السيـاسـة الإعـلاميـة في الدولـة، والإشراف على تنفيذها، بالتنسيق مع الجهات المعنيـة.
٢- الإشـراف الـعـام عـلـى القطاع الإعلامي والصحـفي في ضـوء القـوانـين واللوائح والنـظـم المـعـمـول بـها بالتنسيق مع الجهات ذات العـلاقـة، ومنـح الـتـراخـيـص للإعـلامـيـين.
٣- تلقي طلبـات مـنـح تـراخـيـص فـتـح المكاتب الإعـلاميـة لوكالات الأنبـاء الأجنبيـة في الدولة، ودراستها، بالتنسيق مع الجهات ذات العلاقة.
٤- منـح الـتـرخـيـص لمـراسـلي وكالات الأنباء أو المطبـوعـات الصحـفـيـة أو الإذاعـات أو مـحـطـات التـلـفـزيــون الـعـربيـة أو الأجنبيـة.
٥- دراسـة الاتجاهـات والـتـغـطـيـات الإعـلاميـة والصحـفـيـة المحـليـة وتحليلها، واستـخـلاص اتجـاهـات الـرأي الـعـام الـداخـلـي وتقييمها، ورفـع الـتـوصـيـات بـشـأنـها.
٦- متـابـعـة وتحـلـيـل مـا يـصـدر عـن المـؤسـسـات الإعـلاميـة والإعـلام الإلـكـتـرونـي فـي الـداخـل والخـارج، وتقـديـم تـقـاريـر فـي هـذا الشـأن لرفعها للجـهـات المعنيـة.
۷- تحـلـيـل مـا يـهـم الـرأي العـام الـداخـلـي والخـارجـي، ورفـع الـتـوصـيـات اللازمة لذلك·
۸- وضـع الخـطـة الإعـلاميـة للاحـتـفـاء بالمنـاسـبـات الـوطنيـة.
٩- اقتراح الأدوات التشـريـعـيـة ذات الصلة بالمجـال الإعـلامي.
١٠- اقتراح أفـضـل الـوسـائـل للارتقاء بمهنة الإعلام والصحـافـة، وتطـويـر أداء العـمـل المؤسسي.
۱۱- تشجيع القطـريـين على العـمـل في المجـال الإعـلامي والصـحـفي.
١٢- إقـامـة عـلاقـات فـاعـلـة مـع المؤسسات الإعـلاميـة والصحفيـة المحليـة بهدف تشجيع القطريين على العمل في المجـال الإعلامي والصحفي.

 

المادة 16


تختـص إدارة التـطـويـر الإعـلامي بمـا يـلـي:
١- اقتـــراح أفضـل الوسـائـل للارتقاء بالمحتـوى الإعـلامي، وتطـويـر أداء الإعـلامـيـين والصحفيين.
٢-  تعـزيـز قـاعـدة المـواهـب الوطنيـة في القطاع الإعـلامي.
٣- التنسيق والتعـاون مـع الجـهـات ذات العـلاقـة فـي تـطـويـر المهـارات والقدرات المهـنـيـة للـطـاقـات الوطنيـة في قـطـاع الإعـلام.
٤- إعـداد الدراسات والبحـوث ذات الصلة بالمجـالات الإعـلاميـة.
٥- دعـم مـبـادرات تـطـويـر المحتـوى الإعـلامي.
٦- وضـع شـروط ومعاييـر إنشـاء الأنـديـة والمـراكـز واللجـان والجمعيات ذات الطابع الإعـلامي، وإصـدار الموافـقـات والتراخيص الخاصة بها.

 

المادة 17


تختص إدارة الـبـث الإذاعـي بمـا يـلـي:
۱- بحـث ودراسـة طلبــات التـرخـيـص بإنشـاء وتملـك مـحـطـات البـث الإذاعـي المسـمـوع والمـرئـي، وإصـدارهـا.
٢- الإشراف علـى المـواد المسمـوعـة والمـرئـيـة الـتـي تبثـهـا مـحـطـات البث الإذاعي المسموع والمـرئـي الخـاصـة.

 

المادة 18


تختص إدارة المطـبـوعـات والمصـنـفـات الفـنـيـة بمـا يـلـي:
١- تنفيذ القـوانـين واللوائح والنظم المتعلقة بالمطبـوعـات والنـشـر.
۲- بحـث ودراسـة طلبـات تـراخـيـص المطـبـوعـات الصحـفـيـة، واتخـاذ إجـراءات استصـدارهـا وفـقـاً للـقـانـون المنـظم.
٣- إصـدار تـراخـيـص المطـابـع، ودور النشـر، والمكتبات التجـاريـة، ومـؤسـسـات استيراد وتصـديـر وتـوزيـع المـطـبـوعـات.
٤- إصـدار تـراخـيـص مـزاولـة مـهـنـة الصـحـافـة.
٥- إصدار تراخيـص مـزاولـة أنشـطـة الـدعـايـة والإعـلان والـعـلاقـات العـامـة والإنتاج الفـنـي وبـيـع وتـوزيـع المصـنـفـات الفـنـيـة.
٦- الموافقـة علـى إصـدار وتوزيـع مـطـبـوعـات ونشـرات البعثات والهيئـات الدبلــومـاسـيـة والقنصـلـيـة المعتمـدة لـدى الدولـة، بالتنسيـق مـع الوحدات الإداريـة المعـنيـة والجـهـات المخـتـصـة.
٧- الـرقـابـة علـى المطبـوعـات والمصنـفـات الفنيـة المحـليـة والمستوردة، وفـقـاً للـقـانـون المنـظم.

 

المادة 19


يصدر بتحـديـد اخـتـصـاصـات كـل مـن مـكـتـب الوزيـر، ومكتـب وكيـل الوزارة، ومكتـب وكيـل الـوزارة المسـاعـد، قـرار مـن الـوزيـر.

 

المادة 20


يجـوز بقـرار مـن مـجـلـس الـوزراء، بناء علـى اقـتـراح الـوزيـر، تعـديـل تنظيم الوحدات الإداريـة الـتي تتـألـف منهـا الـوزارة بالإضـافـة أو الحـذف أو الدمج، وتعيين اخـتـصـاصـاتـهـا وتـعـديـلـها.

 

المادة 21


يجـوز بقـرار مـن الوزير إنشاء ما تقتضيه المصلحـة العـامـة مـن أقسـام في الوحدات الإدارية التي تتـألـف منهـا الـوزارة، أو إلغاؤهـا أو دمجها، وتعيين اخـتـصـاصـاتـهـا وتـعـديـلها، ولا يـكـون قـرار الوزيـر نـافـذاً إلا بعـد اعتماده مـن مـجـلـس الـوزراء.

 

المادة 22


يُلغى القرار الأمـيـري رقم (٧) لسنة ٢٠١٦ المشار إليـه.

 

المادة 23


على جميع الجهات المختصة، كـل فيما يخصه، تنفيذ هذا القـرار. ويـُعـمـل بـه مـن تـاريـخ صـدوره. ويُنشر في الجـريـدة الـرسـميـة.
