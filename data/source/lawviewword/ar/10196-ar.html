قــرار أميــري رقـم (48) لسنـة 2025 بـتـعـيـين سـفـيـر فـوق الـعـادة مـفـوض لـدى جـمـهـوريـة مـصـر الـعـربـيـة 48 / 2025
عدد المواد: 2
فهرس الموضوعات

المواد (1-2)



نـحـن تمـيــم بن حـمــد آل ثـاني                        أميــر دولــة قـطـــر ،
بـعـد الاطــلاع عـلى الـدستــور،
وعـلى الـقـانــون رقــم (17) لسـنـة 2021 بشــأن الـسـلـكـين الـدبـلــومــاسـي والـقـنــصـلي ،
وعـلى الـقـرار الأمـيـري رقــم (73) لسـنـة 2023 بـنـقـل سـفـراء فـوق الـعـادة مـفـوضـين إلـى ديـوان عـام وزارة الخـارجـيـة ،
وعــلى اقـتــراح رئيـس مجـلـس الـوزراء ووزيــر الخــارجــيــــة ،
قـررنـــا مــا يلــي :

المواد

المادة 1


يُعــين الـشـيـخ/ جــاســم بـن عـبـدالـرحـمـن بـن مــحــمــد الــعــبــدالــرحــمــن آل ثـانـي ، ســفـيــراً فـــــوق الـعـــادة مـفــوضـــاً مــن لــدنـــا لـــدى جـمـهـوريـة مـصـر الـعـربـيـة .

 

المادة 2


عـلى جــمـيــع الجـهــات المخـتـصــة ، كـل فيــمـا يخــصـه ، تنــفـيــذ هــذا الـقـــرار . ويُعـمـل بــه مـن تـاريــخ صــدوره . ويُنــشـــر في الجــريــدة الـرسـمـيـة
