مـــرســـوم رقـــم (100) لـسـنـــة 2025 بـالـتـصـديــق عـلـى اتـفـاقيـة المـسـاهـمـة الأسـاسـيـة بـين صـنـدوق قـطـر لـلـتـنـمـيـة فـي دولـة قـطـر والـصـنـدوق الـدولـي لـلـتـنـمـيـة الـزراعـيـة 2024 100 / 2025
عدد المواد: 2
فهرس الموضوعات

المواد (1-2)



نـحـن تمـيـم بـن حـمــد آل ثـانـي           أمـيـر دولـة قـطـــر،
بـعـد الاطــلاع عـلـى الـدسـتـور،
وعـلـى وثـيـقـة الـتـصـديـق الـصــادرة فـي الخـامس مـن شـهــر ذي الحجة عـام 1446 هــجـريـة ، المـوافــق للأول مـن شـهـر يـونـيـو عـام 2025 مـيـلاديـة ،
وعـلـى اقـتـراح رئـيـس مـجـلـس الـوزراء ووزيــــر الخـــارجــيـــة ،
وعـلـى مـشـروع المـرسـوم المـقـدم مـن مـجـلـس الـوزراء ،
رسمنـا بمـا هـو آت :
 

المواد

المادة 1


صُــودق عـلـى اتـفـاقـيـة المسـاهـمـة الأسـاسـيـة بـين صـنـدوق قـطـر للـتـنميـة فـي دولــة قـطـر والـصـنـدوق الـدولي للـتـنمـية الـزراعيـة 2024 ، المـوقـعـة بمـديـنـة الـدوحـة بــتـاريـخ 2025/04/21 ، المـرفـق نـصـهـا بـهـذا المـرسـوم ، وتـكـون لـهـا قـوة الـقـانـون ، وفـقـاً لـلـمـادة (68) مـن الـدسـتـور .

 

المادة 2


عـلـى جـمـيـع الجـهـات المـخـتـصـة ، كـل فـيـمـا يـخـصـه ، تـنـفـيـذ هـذا المـرســوم . ويُـعـمـل بـه مـن تـاريـخ صــدوره . ويُـنـشــر فـي الجـريــدة الـرسـمـيـة .
