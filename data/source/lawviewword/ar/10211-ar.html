مـــرســـوم رقـــم (103) لـسـنـــة 2025 بـالـتـصـديــق عـلـى اتـفـاقـيـة لـلـتـعـاون فـي مـجـال بـحـوث تـقـنـيـات مـصـايـد الأسـمـاك والاسـتـزراع المـائـي بـين دولـة قـطـر وجـمـهـوريـة الـصـين الـشـعـبـيـة 103 / 2025
عدد المواد: 2
فهرس الموضوعات

المواد (1-2)



نـحـن تمـيـم بـن حـمــد آل ثـانـي          أمـيـر دولـة قـطـــر ،
بـعـد الاطــلاع عـلـى الـدسـتـور،
وعـلـى وثـيـقـة الـتـصـديـق الـصــادرة فـي الـسـابـع عـشـر مـن شـهــر صـفـر عـام 1447 هــجـرية ، الموافــق للحـادي عشـر من شهـر أغسطـس عام 2025 ميلاديـة ،
وعـلـى اقـتـراح رئـيـس مـجـلـس الـوزراء ووزيـر الخـارجـيـة ،
وعـلـى مـشـروع المـرسـوم المـقـدم مـن مـجـلـس الـوزراء ،
رسمنـا بمـا هـو آت :
 

المواد

المادة 1


صُـودق عـلـى اتـفـاقـيـة للـتـعـاون فـي مـجـال بـحـوث تـقـنـيـات مـصـايـد الأسـمـاك والاسـتـزراع المـائـي بـين إدارة الـثـروة الـسـمـكيـة بـوزارة الـبـلـديـة فـي دولـة قـطـر ومـعـهـد بـحـوث مـصـايـد الأسـمـاك للـبـحـر الأصـفـر بـالأكـاديـمـيـة الـصـيـنـيـة لـعـلوم مـصايـد الأسـمـاك فـي جـمـهـوريـة الصـيـن الـشـعـبـيـة ، المـوقـعـة بمـديـنـة فوزهو- فوجيان بــتـاريـخ 17 /07/ 2025 ، المـرفـق نـصـهـا بـهـذا المـرسـوم ، وتـكـون لـهـا قـوة الـقـانـون ، وفـقـاً لـلـمـادة (68) مـن الـدسـتـور .

 

المادة 2


عـلـى جـمـيـع الجـهـات المـخـتـصـة ، كـل فـيـمـا يـخـصـه ، تـنـفـيـذ هـذا المـرســوم . ويُـعـمـل بـه مـن تـاريـخ صــدوره . ويُـنـشــر فـي الجـريــدة الـرسـمـيـة .
