مرسوم رقم (42) لسنة 2025 بالتصديق على اتفاقية بشأن الإعفاء المتبادل من متطلبات تأشيرة الدخول لحاملي الجوازات الدبلوماسية والخاصة والخدمة بين حكومة دولة قطر وحكومة جمهورية كوستاريكا 42 / 2025
عدد المواد: 2
فهرس الموضوعات

المواد (1-2)



نـحـن تمـيـم بـن حـمــد آل ثـانـي              أمـيـر دولـة قـطــر ،
بـعـد الاطـلاع عـلـى الـدستـور،
وعـلـى وثـيـقـة الـتـصـديـق الـصـادرة فـي الـثـاني عـشـر من شـهـر رجـب عام 1446 هـجـريـة ، الـمـوافـق لـلـثـانـي عـشـر مـن شـهـر يـنـايـر عـام 2025 مـيـلاديـة ،
وعـلى اقـتــراح رئيـس مجـلــس الــوزراء ووزيـــر الخـــارجــية ،
وعـلى مـشـــروع المــرســـوم المـقـــدم مـن مجـلـس الــوزراء ،
رسـمنــا بمـا هــو آت :
 

المواد

المادة 1


صُــودق عـلـى اتـفـاقـيـة بـشـأن الإعـفـاء الـمـتـبـادل مـن مـتـطـلـبـات تـأشـيـرة الـدخـول لـحـامـلـي الـجـوازات الـدبـلـومـاسـيـة والـخـاصـة والـخـدمـة بـيـن حـكـومـة دولـة قـطـر وحـكـومـة جـمـهـوريـة كـوسـتـاريـكـا ، المـوقـعـة بمــدينــة سـان خـوسـيـه بتــاريـخ 20/11/2024 ، المـرفـق نـصــهـا بـهـذا المـرسـوم ، وتــكــون لــهـا قــوة الـقـانـون ، وفـقـاً لـلـمـادة (68) مـن الدسـتـور .

 

المادة 2


عـلـى جـمـيـع الجـهـات المـخـتـصـة ، كـل فـيـمـا يـخـصـه ، تـنـفـيـذ هـذا المـرســوم . ويُـعـمـل بـه مـن تـاريـخ صــدوره . ويُـنـشــر فـي  الجـريــدة الـرسـمـيـة.
