الـفـصـل الأول - تـعـاريـف

المادة 1


فـي تـطـبـيـق أحـكـام هـذا الـقـانـون ، يُقـصـد بالـوزيـر ، كـل وزيـر أو وزيـر للـدولـة يـكـون عـضـواً فـي مـجـلـس الـوزراء ، ويُقـصـد بـشـاغـل درجـة وزيـر ، كـل مـن يـشـغـل درجـة وزيـر بالـتـعـيين أو المـنـح مـن رؤسـاء الأجـهـزة الـحـكـومـيـة أو شـاغـلـي الـوظـائـف الـعـامـة ، وذلـك مـع مـراعـاة حـكـم المـادة (30/فـقـرة ثـانـيـة) مــن هــذا الـقـانـون .

 

الـفـصـل الـثـانـي - تـعـــيين الـوزراء

المادة 2


يـكـون تـعـيين الـوزراء بأمـر أمـيـري ، بـنـاءً عـلـى اقـتـراح رئـيـس مـجـلـس الـوزراء ، ويـحـدد الأمـر الأمـيـري الـصـادر بـتـشـكـيـل مـجـلـس الـوزراء ، وزراء الـدولـة الأعـضـاء فـي المـجـلـس .
وللأمـيـر أن يـعـهـد إلـى رئـيـس مـجـلـس الـوزراء أو أحـد نـوابـه أو أحـد الـوزراء ، بمـهـام وزارة أو أكـثـر ، أو رئـاسـة أحـد الأجـهـزة الـحـكـومـيـة أو الـهـيـئـات أو المـؤسـسـات الـعـامـة بالإضـافـة إلـى مـهـام منصـبـه .
 

 

المادة 3


بالإضـافـة إلـى الـشـروط المـنـصـوص عـلـيـهـا فـي الـدسـتـور ، يُشـتـرط فـيـمـن يُعـــين وزيـــراً مـا يـلـي :
1- أن يـكـون مـن ذوي الـكـفـاءة والـخـبـرة .
2- أن يـكـون مـحـمـود الـسـيـرة حـسـن الـسـمـعـة ، ولم يـسـبـق إدانـتـه بـحـكـم نـهـائـي فـي جـنـايـة أو بـعـقـوبـة مـقـيـدة للـحـريـة فـي جـريمـة مـخـلـة بالـشـرف أو الأمـانـة ، ولـو كـان قـد رُد إلـيـه اعـتـباره .
 

 

المادة 4


يـؤدي الـوزراء قـبـل تـولـيـهـم مـنـاصـبـهـم ، الـيـمـين المـنـصـوص عـلـيـهـا فـي الـدسـتـور .
ويـكـون أداء الـيـمـين أمـام الأمـيـر ورئـيـس مـجـلـس الـوزراء ، ويـعـيـد الـوزراء أداء الـيـمـين كـلمـا شـمـلـهـم الـتـعـديـل الـوزاري .
 

 

المادة 5


للأمـيـر أن يـعـهـد إلـى أحـد الـوزراء بالـقـيـام بـعـمـل مـن يـخـلـو مـنـصـبـه مـن الـوزراء .
ولـرئـيـس مـجـلـس الـوزراء أن يـعـهـد إلـى أحـد الـوزراء بالـقـيـام بـعـمـل مـن يـتـغـيـب مـن الـوزراء فـي إجـازة أو مـهـمـة رسـمـيـة بالـخـارج .
 

 

الـفـصـل الـثـالـث - اخـتـصـاصـات الـوزراء

المادة 6


بالإضـافة إلـى الاخـتصـاصـات المقـررة بمـقتـضـى أحـكـام الـدسـتـور والـقـانـون وغـيـرهـمـا مـن الأدوات الـتـشـريـعـيـة الأخــرى ، يـتـولـى الـوزيـر الاخـتـصـاصـات الـتـالـيـة : 
1- الإشـراف عـلـى إعـداد الـخـطـة الاسـتـراتـيـجـيـة والـخـطـة الـتـنـفـيـذيـة للـوزارة ، وتـنـفـيـذ الـسـيـاسـات والمـشـاريـع والـبـرامـج المـتـعـلـقـة باخـتـصـاصـات الـوزارة بـعـد اعـتـمـادهـا .
2- عـرض المـوضـوعـات والـبـيـانـات والـتـقـاريـر المـتـعـلـقـة بـنـشـاط الـوزارة أو مـا يـتـبـع الـوزيـر مـن الأجـهـزة الـحـكـومـيـة أو الـهـيـئـات والمـؤسـسـات الـعـامـة ، عـلـى مـجـلـس الـوزراء .
3- تـنـفـيـذ الـتـشـريـعـات المـتـعـلـقـة بـنـشـاط الـوزارة وإصـدار الـقـرارات الـلازمـة لـتـنـفـيـذ تـلـك الـتـشـريـعـات بمـا يـطـابـق أحـكـامـهـا فـي جـمـيـع الـنـواحـي الـفـنـيـة والإداريـة والمـالـيـة .
4- الإشـراف عـلـى الأجـهـزة الـحـكـومـيـة أو الـهـيـئـات أو المـؤسـسـات الـعـامـة الـتـابـعـة لـه ، وفـقـاً لمـا تـقـرره الأدوات الـتـشـريـعـيـة المـنـظـمـة لـتـلـك الـجـهـات .
5- إصـدار قـرارات تـحـديـد رسـوم الـخـدمـات الـتـي تـؤديـهـا الـوزارة أو مـا يـتـبـع الـوزيـر مـن الأجـهـزة الـحـكـومـيـة أو الـهـيـئـات والمـؤسـسـات الـعـامـة ، أو تـعـديـل الـرسـوم أو إلـغـائـهـا أو الإعـفـاء مـنـهـا ، بـعـد اعـتـمـادهـا مـن مـجـلـس الـوزراء .
6- إصـدار الـقـرارات المـتـعـلـقـة بـشـؤون مـوظـفـي الـوزارة وفـقـاً للـقـانـون المـنـظـم لـذلـك . 
7- تـصـريـف شـؤون الـوزارة فـي إطـار الـسـيـاسـة الـعـامـة للـدولـة ، والإشـراف عـلـى الأداء الـعـام بـهـا .
8- تمـثـيـل الـوزارة فـي المـؤتمـرات والـهـيـئـات والمـنـظـمـات الـعـربـيـة والإقـلـيـمـيـة والـدولـيـة فـيـمـا يـتـصـل باخـتـصـاصـات الـوزارة .
9- الإشـراف علـى إعـداد مـشـروع المـوازنـة الـسـنـويـة للـوزارة وحـسـابـهـا الـخـتـامـي .
10- اقـتـراح الـتـشـريـعـات المـتـعـلـقـة بـنـشـاط الـوزارة.
 

 

الـفـصـل الـرابـع - حـقـوق الـوزراء وامـتـيـازاتـهـم

المادة 7


يـسـتـحـق الـوزيـر المـخـصـصـات الـشـهـريـة والمـزايـا والمـكـافـآت الـتـي يـصـدر بـتـحـديـدهـا قـرار أمـيـري .
وإذا كـانـت المـخـصـصـات الـشـهـريـة والمـزايـا والمـكـافآت الـتـي يـتـقـاضـاهـا الـوزيـر قـبـل تـعـيينـه ، تـزيـد عـلـى مـا هـو مـقـرر لـه ، تُطـبـق بـشـأنـه الـقـواعـد الـتـي يـُحــددهـا الـقــرار الأمــيـري المـنـصـوص عـلـيـه فـي الـفـقـرة الـسـابـقـة .
وفـي جـمـيـع الأحـوال ، لا يـجـوز الـجـمـع بـين المـخـصـصـات أو المـزايـا أو المـكـافـآت الـتـي تتـفـق فـي مـنـاط اسـتـحـقـاقـهـا .
وإذا كـان للـوزيـر فـي جـهـة عـمـلـه الـسـابـقـة مـدة تـخـولـه اسـتـحـقـاق المـعـاش ، أو كـان يـتـقـاضـى مـعـاشـاً ، اسـتـحـق الـجـمـع بـين المـعـاش
المـشـار إلـيـه وراتـبـه فـي مـنـصـبـه الـوزاري ، وذلـك وفـقـاً للـقـواعـد الـتـي يُـحـددهـا الـقـرار الأمـيـري المـنـصـوص عـليـه فـي الـفـقـرة الأولـى مـن هـذه المـادة .
 

 

المادة 8


تـكـون إجـازات الـوزراء ومـددهـا وشـروطـهـا ، وفـقـاً لمـا يـحـدده مـجـلـس الـوزراء . 

 

المادة 9


يـكـون سـفـر الـوزراء فـي المـهـام الـرسـمـيـة خـارج الـدولـة بـقـرار مـن رئـيـس مـجـلـس الـوزراء ، وفـقـاً للـنـظـام الـذي يـعـتـمـده مـجـلـس الـوزراء .

 

المادة 10


يـسـتـحـق الـوزيـر المـوفـد فـي مـهـمـة رسـمـيـة خـارج الـدولـة المـخـصـصـات المـالـيـة وبـدل الـتـمـثـيـل ، عـلـى الـنـحـو الـذي يُـحـدده الـقـرار الأمـيـري المـنـصـوص عـلـيـه فـي المـادة (7) مـن هـذا الـقـانـون .

 

الـفـصـل الـخـامـس - واجـبات الـوزراء والأعـمـال المـحـظـورة عـلـيـهـم

المادة 11


بالإضـافـة إلـى الـواجـبـات المـنـصـوص عـلـيـهـا فـي الـدسـتـور والـقـانـون وغـيـرهـمـا مـن الأدوات الـتـشـريـعـيـة الأخــرى ، يـجـب عـلـى الـوزيـر مـا يـلـي :
1- أداء مـهـام مـنـصـبـه بالأمـانـة والـشـرف .
2- تـنـفـيـذ الـتـعـلـيـمـات والـتـوجـيـهـات الـتـي تـصـدر إلـيـه مـن الأمـيـر أو رئـيـس مـجـلـس الـوزراء فـي حـدود الـقـانـون .
3- اسـتـهـداف الـصـالـح الـعـام وإنـكـار المـصـالـح الـذاتـيـة .
 

 

المادة 12


بالإضـافـة إلـى المـحـظـورات المـنـصـوص عـلـيـهـا فـي الـقـانـون، يـحـظـر عـلـى الـوزيـر مـا يـلـي :
1- مـخـالـفـة أحـكـام الـدسـتـور .
2- اسـتـغـلال الـنـفـوذ للـحـصـول عـلـى فـائـدة أو مـيـزة ذاتـيـة لـنـفـسـه أو لـغـيـره .
3- إفـشـاء الأسـرار الـتـي يـطـلـع عـلـيـهـا بـحـكـم مـنـصـبـه ولـو بـعـد انـتـهـاء خـدمـتـه .
4- الـتـدخـل فـي شـؤون الـعـدالـة بـطـريـق مـباشـر أو غـيـر مـباشـر .
5- الـتـدخـل فـي الـعـمـلـيـات الانـتـخـابـيـة أو إجـراءاتـهـا بأي صـورة مـن الـصـور بـقـصـد الـتـأثـيـر فـي نـتـيـجـتـهـا .
6- ممـارسـة الـنـشـاط الـتـجـاري إذا كـان مـتـعـارضـاً مـع طـبـيـعـة الـعـمـل الـذي يـقـوم بـه .
 

 

الـفـصـل الـسـادس - مُسـاءلـة الـوزراء

المادة 13


تتـم مـسـاءلـة الـوزراء ، وفـقـاً للإجـراءات المـنـصـوص عـلـيـهـا فـي هـذا الـقـانـون ، إذا ارتـكـبـوا أثـنـاء شـغـلـهـم المـنـصـب إحـدى الـجـرائـم الآتـيـة :
1- الـجـرائـم المـتـعـلـقـة بأمـن الـدولـة الـداخـلـي أو الـخـارجـي .
2- الـجـرائـم المـتـعـلـقـة بالـوظـيـفـة الـعـامـة .
3- الـجـرائـم المـتـعـلـقـة بالانـتـخـاب .
4- الـجـرائـم المـتـعـلـقـة بـسـيـر الـعـدالـة .
5- جـرائـم غـسـل الأمـوال .
وتُطـبـق فـي شـأنـهـم الـعـقـوبات المـقـررة قـانـونـاً .
 

 

المادة 14


تـكـون إحـالـة الـوزيـر ، أو شـاغـل درجـة وزيـر ، إلـى الـتـحـقـيـق أو المـحـاكـمـة بـقـرار مـن الأمـيـر ، بـنـاءً عـلـى طـلـب مـن الـنـائـب الـعـام .
 

 

المادة 15


يُعـتـبـر الـوزيـر فـي إجـازة بـراتـب إجـمـالـي مـن تـاريـخ إبـلاغـه بـقـرار الإحـالـة للـتحـقـيـق أو المـحـاكـمـة ، ولا تـنـتـهـي الإجـازة إلا بحـفـظ الـتـحـقـيـق ، أو صـدور أمـر بألا وجـه لإقـامـة الـدعـوى ، أو بانـقـضـاء الـدعـوى لأي سـبـب قـبـل صـدور حـكـم فـيـهـا ، أو الـحـكـم فـي الـدعـوى بـبـراءتـه .
فـإذا صـدر ضـده حـكـم بات بالإدانـة انـتـهـت خـدمـتـه مـن تـاريـخ صـدور الـحـكـم .
 

 

المادة 16


يـتـولـى الـنـائـب الـعـام أو مـن يـفـوضـه الـتـحـقـيـق مـع الـوزيـر بمـعـاونـة مـن يـرى الاسـتـعـانـة بـهـم مـن أعـضـاء الـنـيـابـة الـعـامـة .

 

المادة 17


إذا تـبـين بـعـد انـتـهـاء الـتـحـقـيـق أن هـنـاك جـريـمـة مـن الـجـرائـم المـبـينـة فـي المـادة (13) مـن هـذا الـقـانـون ، وأن الأدلـة كـافـيـة ، فـعـلـى الـنـائـب الـعـام أن يـعـد قـرار الاتـهـام وقـائـمـة بأدلـة الـثـبـوت بالـنـسـبـة للـوزيـر وأي فـاعـلـين آخــرين أو شـركـاء ، وأن يـأمـر بإحـالـة الـدعـوى إلـى المـحـكـمـة المـنـصـوص عـلـيـهـا فـي المـادة (19) مـن هـذا الـقـانـون .
فـإذا أسـفـر الـتـحـقـيـق عـن عـدم وجـود جـريمـة ، أو أن الـوقـائـع المـنـسـوبـة للـوزيـر لا صـحـة لـهـا ، أو أن الأدلـة عـلـيـهـا غـيـر كـافـيـة ، أمـر بألا وجـه لإقـامـة الـدعـوى ، أو بـحـفـظ الـتـحـقـيـق بـقـرار مـسـبـب ، بـحـسـب الأحـوال .
 

 

المادة 18


عـلـى الـنـائـب الـعـام ، إعـلان الـوزيـر بـصـورة مـن أمـر الإحـالـة وقـائـمـة أدلـة الـثـبـوت قـبـل بـدء المـحـاكـمـة بـعـشـرة أيـام عـلـى الأقـل ، وإعـلان شـهـود الإثـبات بالـجـلـسـة المـحـددة ، أو إعـلانـه بالأمـر بألا وجـه لإقـامـة الـدعـوى أو بأمـر الـحـفـظ ، بـحـسـب الأحـوال .
كـمـا يـجـب إخـطـار رئـيـس مـجـلـس الـوزراء بـصـورة مـن قـرار الاتـهـام وقـائـمـة أدلـة الـثـبـوت أو بالأمـر بألا وجـه لإقـامـة الـدعـوى أو الأمـر بالـحـفـظ ، وذلـك خـلال ثـمـان وأربـعـين سـاعـة مـن يـوم صـدوره .
 

 

المادة 19


تتـولـى مـحـاكـمـة الـوزيـر مـحـكـمـة خـاصـة ، تُشـكـل مـن خـمـسـة مـن الـقـضـاة يـخـتـارهـم المـجـلـس الأعـلـى للـقـضـاء ، كـمـا يـخـتـار اثـنـين آخــرين كـعـضـوين احـتـيـاطـيين لإكـمـال تـشـكـيـل المـحـكـمـة فـي حـالـة غـيـاب أحـد الأعـضـاء الأصـلـيين أو قـيـام مـانـع لـديـه ، وتـخـتـص هـذه المـحـكـمـة بمـحـاكـمـة الـوزراء وأي فـاعـلـين آخــرين أو شـركـاء مـن غـيـر الـوزراء ، كـمـا تـخـتـص بـنـظـر الـجـرائـم المـرتـبـطـة ارتـباطـاً لا يـقـبـل الـتـجـزئـة بالـجـرائـم المـعـروضـة عـلـيـهـا.
ويـباشـر الـنـائـب الـعـام أو مـن يـفـوضـه مـن أعـضـاء الـنـيـابـة الـعـامـة مـهـمـة الادعـاء أمـام المـحـكـمـة ، ويـتـولـى قـلم كـتـاب مـحـكـمـة الاسـتـئـنـاف أعـمـال قـلم كـتـاب هـذه المـحـكـمـة .
 

 

المادة 20


تـكـون الأحـكـام الـصـادرة مـن المـحـكـمـة المـشـكـلـة طـبـقـاً لأحـكـام المـادة الـسـابـقـة غـيـر قـابـلـة للـطعـن فـيـهـا إلا بـطـريـق الـتـمـييـز ، ومـع ذلـك إذا صـدر الـحـكـم غـيـابـيـاً جـازت المـعـارضـة فـيـه أمـام ذات المـحـكـمـة بالإجـراءات المـعـتـادة المـنـصـوص عـلـيـهـا فـي قـانـون الإجـراءات الـجـنـائـيـة .
وفـي جـمـيـع الأحـوال ، تـقـضـي المـحـكـمـة بإلـزام المـحـكـوم عـلـيـه بـرد مـا أفـاده مـن الـجـريمـة .
 

 

المادة 21


يـكـون حـسـاب مـدة انـقـضـاء الـدعـوى الـجـنـائـيـة عـن الـجـرائـم الـتـي يـرتـكـبـهـا الـوزيـر مـن تـاريـخ تـرك المـنـصـب ، وفـي حـالـة تـعـدد المـتـهـمـين لا يـبـدأ حـسـاب المـدة بالـنـسـبـة لأحـدهـم إلا مـن تـاريـخ تـرك الـباقـين وظـائـفـهـم .

 

المادة 22


تـسـري أحـكـام قـانـون الإجـراءات الـجـنـائـيـة المـشـار إلـيـه ، فـيـمـا لم يـرد بـشـأنـه نـص خـاص فـي هـذا الـفـصـل .

 

الـفـصـل الـسـابـع - انـتـهـاء خـدمـة الـوزراء

المادة 23


تـنـتـهـي خـدمـة الـوزيـر لأحـد الأسـباب الآتـيـة :
1- إعـادة تـشـكـيـل مـجـلـس الـوزراء كـلـيـاً أو جـزئـيـاً .
2- قـبـول الاسـتـقـالـة .
3- الإعـفـاء مـن المـنـصـب .
4- الـوفـاة .
 

 

المادة 24


يـسـتـحـق الـوزيـر مـكـافـأة نـهـايـة الـخـدمـة عـنـد انـتـهـاء خـدمـتـه، ولا يـحـول ذلـك دون اسـتـحـقـاقـه مـكـافـأة نـهـايـة الـخـدمـة المـسـتـحـقـة لـه مـن جـهـة عـمـلـه قـبـل تـعـيينـه ، مـتـى كـان مـسـتـحـقـاً لـهـا وفـقـاً للـقـانـون .
ويُـحـدد الـقـرار الأمـيـري المـنـصـوص عـلـيـه فـي المـادة (7) مـن هـذا الـقـانـون مـقـدار هـذه المـكـافـأة وشـروط اسـتـحـقـاقـهـا وضـوابـط مـنـحـهـا وحـالات الـحـرمـان مـنـهـا .
 

 

المادة 25


يـسـتـحـق الـوزيـر بـعـد انـتـهـاء خـدمـتـه ، مـعـاشـاً شـهـريـاً يـعـادل راتـبـه الأسـاسـي الـذي كـان يـتـقـاضـاه ، بالإضـافـة إلـى الـعـلاوة الاجـتـمـاعـيـة وبـدل الـسـكـن ، وفـقـاً لمـا يُـحـدده الـقـرار الأمـيـري المـنـصـوص عـلـيـه فـي المـادة (7) مـن هـذا الـقـانـون ، مـع مـراعـاة مـا يـلـي :
1- فـي حـالـة الـوفـاة ، يُصـرف المـعـاش كـامـلاً للمـسـتـحـقـين عـنـه ، وفـقـاً للـضـوابـط المـقـررة بمـوجـب قـانـون الـتـأمـينـات الاجـتـمـاعـيـة المـشــار إليــه .
2- فـي حـالـة الاسـتـقـالـة ، يـشـتـرط لاسـتـحـقـاق المـعـاش ، أن يـكـون قـد قـضـى فـي المـنـصـب سـنـتـين كـامـلـتـين عـلـى الأقـل .
3- فـي حـالـة الإعـفـاء مـن المـنـصـب ، يـجـوز الـحـرمـان مـن المـعـاش ، كـلـه أو بـعـضـه ، بـقـرار مـن الأمـيـر . 
وفـي جـمـيـع الأحـوال يـجـوز للـوزيـر الـجـمـع بـين أكـثـر مـن مـعـاش مـتـى تـوفـر بـشـأنـه مـنـاط الاسـتـحـقـاق .
 

 

المادة 26


إذا انـتـهـت خـدمـة الـوزيـر ، ولم تتـوفـر بـشـأنـه شـروط اسـتـحـقـاق المـعـاش وفـقـاً لأحـكـام هـذا الـقـانـون ، وكـانـت لـه مـدة خـدمـة سـابـقـة فـي إحـدى الـجـهـات الـخـاضـعـة لأحـكـام قـانـون الـتـأمـينـات الاجـتـمـاعـيـة أو الـقـانـون رقــم (2) لسـنــة 2022 ، المـشـار إلـيـهـمـا ، اسـتـحـق مـعـاشـاً عـن تـلـك المـدة ، عـلـى أن يـلـتـزم بـسـداد جـمـيـع الاشـتـراكـات المـسـتـحـقـة عـلـيـه وفـقـاً لأحـكـام الـقـانـون المـنـظـم لـذلـك ، بـحـسـب الأحـوال .

 

الـفـصـل الـثـامـن - وزراء الـدولـة مـن غـيـر أعـضـاء مـجـلـس الـوزراء وشـاغـلـو درجـة وزيـر

المادة 27


يـكـون تـعـيين وزراء الـدولـة مـن غـيـر أعـضـاء مـجـلـس الـوزراء، بأمـر أمـيـري ، وتـسـري عـلـيـهـم أحـكـام الـفـصـلـين الـخـامـس والـسـادس والمـادتـين (9) ، (10) مـن هـذا الـقـانـون .
ويـحـدد الـقـرار الأمـيـري المـنـصـوص عـلـيـه فـي المـادة (7) مـن هـذا القـانـون ، حـقـوق وامـتيـازات وزراء الـدولـة مـن غـيـر أعـضـاء مـجـلـس الـوزراء .
وتـسـري عـلـى وزراء الـدولـة المـكـلـفـين باخـتـصـاصـات أو مـهـام مُـحـددة بمـوجـب الأمـر الأمـيـري الـصـادر بـتـعـيينـهـم أحـكـام الـفـصـول مـن الـرابـع إلـى الـسـابـع مـن هـذا الـقـانـون .
 

 

المادة 28


للأمـيـر تـعـيين رؤسـاء الـجـهـات أو الأجـهـزة الـحـكـومـيـة بـدرجـة وزيـر ، أو مـنـح الـدرجــة لـهـم أو لـغـيـرهـم مـن شـاغـلـي الـوظـائـف الـعـامـة .
وللأمـيـر تـكـلـيف أي مـنـهـم بالقـيـام بـرئـاسـة أكـثـر مـن جـهـاز حـكـومـي ، أو بالـقـيـام بـعـمـل مـن يـخـلـو مـنـصـبـه .
ولـرئـيـس مـجـلـس الـوزراء ، فـي حـدود الأداة الـتـشـريـعـيـة المـنـظـمـة للـجـهـة أو الـجـهـاز الـحـكـومـي ، أن يـعـهـد إلـى أي مـنـهـم بالـقـيـام بـعـمـل مـن يـتـغـيـب أو يـقـوم بإجـازة أو مـهـمـة رسـمـيـة بالـخـارج .
 

 

المادة 29


تـسـري الأحـكـام المـنـصـوص عـلـيـهـا فـي الـفـصـول مـن الـرابـع إلـى الـسـابـع مـن هـذا الـقـانـون ، عـلـى رؤسـاء الأجـهـزة الـحـكـومـيـة بـدرجـة وزيـر ، فـيـمـا عـدا حـكـم المــادة (9) مـن هــذا الـقــانــون .
وتـسـري الأحـكـام المـنـصـوص عـلـيـهـا فـي الـفـصـلـين الـرابـع والـسـابـع مـن هـذا الـقـانـون ، عـلـى شـاغـل درجـة وزيـر بالـوظـائـف الـعـامـة الأخــرى ، فـيـمـا عـدا حـكـم المــادة (9) مـن هــذا الـقــانــون .
 

 

المادة 30


للأمـيـر مـنـح درجـة رئـيـس مـجـلـس الـوزراء أو نـائـب رئـيـس مـجـلـس الـوزراء لمـن يُقـدر مـنـحـهـا لـه ، مـن رؤسـاء الأجـهـزة الـحـكـومـيـة وشـاغـلـي الـوظـائـف الـعـامـة .
وللأمـيـر مـنـح درجـة رئـيـس مـجـلـس الـوزراء أو نـائـب رئـيـس مـجـلـس الـوزراء أو درجـة وزيـر لمـن يُقـدر مـنـحـهـا لـه ، مـن غـيـر شـاغـلـي الـوظـائـف الـعـامـة ، ولا تـسـري عـلـيـه أحـكـام هـذا الـقـانـون .
ويُحـدد الـقـرار الأمـيـري المـنـصـوص عـلـيـه فـي المـادة (7) مـن هـذا الـقـانـون ، حـقـوق وامـتـيـازات مـن يُمـنـح تـلـك الـدرجـة ، وآلـيـة تـحـديـد مـسـتـحـقـي المـعـاش المـقـرر لـه عـنـد وفـاتـه .
 

 

المادة 31


مـع مـراعـاة حـكـم المـادة (24) مـن هـذا الـقـانـون ، تـنـتـهـي خـدمـة شـاغـل درجـة وزيـر ، إذا حـل مـحـلـه غـيـره فـي مـنـصـبـه ولم تُـسـنـد إلـيـه مـهـام وظـيـفـيـة أخــرى ، وتـسـري بـشـأنـه الأحـكـام الـخـاصـة المـتـرتـبـة عـلـى انـتـهـاء الـخـدمـة وفـقـاً لأحـكـام هــذا الـقــانـون .

 

الـفـصـل الـتـاسـع - أحـكـام عـامـة

المادة 32


تـسـري عـلـى رئـيـس مـجـلـس الـوزراء ونـائـب رئـيـس مـجـلـس الـوزراء ومـن فـي درجـتـهـمـا مـن شـاغـلـي الـوظـائـف الـعـامـة ، ذات الأحـكـام الـواردة فـي هــذا الـقــانـون ، مـع مـراعــاة طـبـيـعـة مـنـصـبـيـهـمـا .

 

المادة 33


يـكـون إفـصـاح المـخـاطـبـين بأحـكـام هـذا الـقـانـون عـن الـحـالات الـتـي قـد تـفـضـي إلـى تـضـارب المـصـالـح ، عـنـد الـتـعـيين أو المـنـح ، بـحـسـب الأحـوال ، وطـوال مـدة الـعـمـل ، وفـقـاً للـقـواعـد الـتـي يـصــدر بــهـا قــرار أمــيـري .

 

المادة 34


يُحـدد الـقـرار الأمـيـري المـنـصـوص عـلــه فـي المـادة (7) من هـذا الـقـانـون ، الـضـوابـط الأخــرى للمـخـصـصـات المـالـيـة الـتـي يـتـقـاضـاهـا مـن يـشــغـل درجــة وزيـر ، ممـن يـعـمـلـون بأي مـن الـجـهـات الـتـي تُنـظـم شـؤونـهـا قــوانــين أو أنـظـمـة وظـيـفـيـة خـاصـة .

 

المادة 35


يُصـدر مـجـلـس الـوزراء القـرارات الـلازمـة لـتنـفـيـذ أحـكـام هـذا الـقـانـون .

 

المادة 36


يُـلـغـى الـقـانـون رقـم (21) لـسـنـة 2004 المـشـار إلـيـه ، كـمـا يُـلـغـى كـل حـكـم يـخـالـف أحـكـام هـذا الـقـانـون .

 

المادة 37


عـلـى جـمـيـع الـجـهـات المـخـتـصـة ، كـل فـيـمـا يـخـصـه ، تـنـفـيـذ هـذا الـقـانـون . ويُعـمــل بـه من تـاريـخ صــــدوره . ويُـنـشـر فـي الـجـريـدة الـرسـمـيـة .
