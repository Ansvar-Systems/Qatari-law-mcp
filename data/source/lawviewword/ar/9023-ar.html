قرار أميري رقم (43) لسنة 2022 بالهيكل التنظيمي لوزارة التنمية الاجـتـمـاعـيـة والأسـرة 43 / 2022
عدد المواد: 22
فهرس الموضوعات

المواد (1-22)



نحن تميم بن حمد آل ثاني أمير دولة قطر،
بعد الاطلاع على الدستور،
وعلى قانون الموارد البشرية المدنية الصادر بالقانون رقم (15) لسنة 2016، المعدل بالقانون رقم (23) لسنة 2020،
وعلـى القـرار الأميري رقـم (57) لسنـة 2021 بتعييـن اخـتـصـاصـات الوزارات،
وعلى اقتراح مجلس الوزراء،

المواد

المادة 1


يتـكـون الهيكل التنظيمـي لـوزارة التنمية الاجـتـمـاعـيـة والأسـرة مـن الوحدات الإداريـة المبينـة بالخريطة التنظيميـة المـرفـقـة بهذا القـرار، وهي:
أولاً: الوحدات الإداريـة الـتـابـعـة للـوزيـر:
۱- مـكـتـب الـوزيـر.
۲- المـكـتـب الفـنـي.
٣- إدارة التدقيـق الـداخـلي.
٤– إدارة التـعـاون الدولي.
٥– إدارة التـخـطـيـط والجـودة والابتكار.
٦- إدارة الشـؤون الـقـانـونيـة.
 
ثانياً: الوحدات الإداريـة التـابـعـة لوكيل الوزارة:
۱- مكتـب وكيـل الـوزارة.
٢- إدارة العـلاقـات الـعـامـة والاتـصـال.
٣- إدارة الموارد البـشـريـة.
٤- إدارة الشؤون الماليـة والإداريـة.
٥- إدارة نـظـم المـعـلـومـات.
ثالثـاً: الوحـدات الإداريـة التـابـعـة لوكيـل الـوزارة المسـاعـد لـشـؤون التنـمـيـة الاجـتـمـاعـيـة:
۱- مكتـب وكيـل الـوزارة المسـاعـد.
٢- إدارة الـضـمـان الاجـتـمـاعـي.
٣- إدارة تـنـمـيـة المجـتـمـع.
٤- إدارة إسكان المـواطـنـين.
٥- إدارة الجـمـعـيـات والمـؤسـسـات الخـاصـة.
رابعاً: الوحدات الإداريـة التـابـعـة لوكيـل الـوزارة المسـاعـد لـشـؤون الأسـرة:
۱- مكتب وكيـل الـوزارة المسـاعـد.
٢- إدارة الحـمـايـة الاجـتـمـاعـيـة.
٣- إدارة الـتـنـميـة الأسـريـة.
٤- إدارة الـرعـايـة المجـتـمـعـيـة.
٥- إدارة التمكين الأسـري.

 

المادة 2


يختـص المكتـب الفـنـي بمـا يـلـي:
١- إبـداء الـرأي الفـني في المـوضـوعـات التي تُحال إليـه مـن الـوزيـر.
۲- دراسـة وتقييم التقاريـر والمشـروعـات والبـرامـج المتعلقة بأنشـطـة الوزارة، التي تُحال إليـه مـن الـوزيـر.
۳- بحـث ودراسـة ومـتـابـعـة المـوضـوعـات الـتـي تُحال إليـه مـن الوزيـر، بالتنسيق مع الوحـدات الإداريـة المعنيـة.
٤- إعـداد ملخص للمـواضـيـع والتقـاريـر والـدراسـات الـتـي يـلـزم عـرضـها على الـوزيـر، وإبـداء الـرأي بشـأنـها.

 

المادة 3


تختص إدارة التـدقـيـق الـداخـلـي بمـا يـلـي:
١- إعداد مشروع خـطـة التـدقيـق السنـويـة المبنيـة عـلـى المخـاطـر للوزارة، ورفعـهـا للـوزيـر لاعتمـادهـا، وإعـداد تقـاريـر بـنـتـائـج التـدقيـق.
۲- مـراقبـة مـدى الالتزام بتنفيـذ القـوانـين واللوائح والقرارات المنظمـة للعـمـل.
۳- مـراجـعـة التعليمات الماليـة، واقـتـراح مـا يـلـزم لـهـا مـن تعـديـلات، ومـراقبـة تنفيـذهـا.
٤- مـراقبـة الأعمـال الإداريـة وإجـراءات تعيين الموظفين، وإجازاتـهم، وتـرقـيـاتـهم، وغير ذلك ممـا يـتـعـلـق بـشـؤون الموظفين، والتحقـق مـن الالتزام بالقوانين واللوائح والقرارات المنظمـة لذلك.
٥- مـراقبـة الأعمـال المـاليـة والمحاسبيـة، وسندات الصرف، وإجـراءات القيـد في السجـلات المحاسبيـة، والتـأكـد مـن تطبيـق القـواعـد المعتمدة في هـذا الشـأن.
٦- تقييم المخاطـر المتعلقة بالتدقيـق المالي والإداري، واقتراح إجـراءات الحـد منـها.
۷- متـابعـة الإجـراءات التـصـحـيـحـيـة المطـلـوبـة مـن الـوحـدات الإداريـة المعـنيـة، وفـق نتائج التدقيـق، وإعـداد التقارير الدوريـة بشـأنـها.
٨- التـأكـد مـن سـلامـة الأمـوال والعـهـد والأثاث والأدوات والمبـاني والمـوجـودات الأخرى التي تملكها الوزارة أو تخضع لإشـرافـها.
٩- الاشتـراك في أعمـال الجـرد الـدوري والمفاجئ على المخازن والعهـد والتـأكـد مـن دقـة التسجيـل بـالـدفـاتـر وسـلامـة الإجـراءات المطبقـة.
۱۰- متـابعـة نتائج تقاريـر الجهـات الـرقـابيـة، وإبـداء التوصيـات المنـاسبـة بشأنها.

 

المادة 4


تختص إدارة التـعـاون الدولي بمـا يـلـي:
١- إعداد أوراق العمـل المتعلقة بالمؤتمرات والندوات والاجتمـاعـات المحـليـة والإقليميـة والدوليـة فـيـمـا يـخـص نشـاط الـوزارة، بالتنسيـق مـع الوحـدات الإداريـة المعنيـة والجـهـات المختـصـة.
٢- الإعـداد لحضـور المؤتمرات والاجتمـاعـات الـتي تعقـدهـا المنظمـات والهيئـات الإقليميـة والدوليــة المختصة بقضـايـا الـوزارة، الـتي تشـارك الدولة في عضـويـتـها.
٣- إعـداد التقاريـر والـردود التي تطلبهـا المنظمات والهيئات الإقليميـة والدوليـة فـيـمـا يـخـص نشـاط الـوزارة، بالتنسيـق مـع الوحـدات الإداريـة المخـتـصـة.
٤- دراسـة ومـتـابـعـة تنفيـذ تـوصـيـات المنظمات والهيئـات الإقليميـة والدوليـة، فيمـا يـخـص نشـاط الـوزارة، بالتنسـيـق مـع الجهـات المخـتـصـة بـالـدولـة.
٥- إعداد التقارير اللازمـة بشـأن إسـهـامـات المنظمات والهيئات الإقليميـة والدوليـة في تنميـة وتـدعـيـم الأنشـطـة المختلفـة في مجـالات عمـل الوزارة، بالتنسيق مع الجهات المخـتـصـة.
٦- متـابعـة سـداد المساهمـات المـاليـة للدولـة في المنظمـات المعنيـة بمجـالات عمـل الـوزارة الـتـي تكـون الدولـة عـضـواً فيهـا، بالتنسيـق مـع الجـهـات المخـتـصـة.
۷- دراسـة السـبـل الـكـفـيـلـة بـتـفـعـيـل دور الدولـة فـي مـجـال نـشـاط الـوزارة مع المنـظـمـات العـربـيـة والإقـلـيـمـيـة والـدوليـة.
۸- جمـع المـعـلـومـات والبحـوث الصـادرة عـن المنظمات والهيئـات ذات الصـلة بقـضـايـا الـوزارة مـحـلـيـاً وإقليمياً ودولياً، ونشـرهـا بـغـرض الاسـتـفـادة مـنها.
۹- تـرجـمـة الـوثـائق والكتب والبـحـوث العلميـة والمكاتبـات الأجنبيـة المتعلقة باخـتـصـاصـات الـوزارة.

 

المادة 5


تختص إدارة التخطيط والجـودة والابتكار بمـا يـلـي:
١- إعـداد الخطـة الاستراتيجيـة للـوزارة، بالتنسيق مع الجهات المختصة والوحـدات الإداريـة المعنيـة، واتخاذ إجـراءات اعتمـادهـا.
۲- حـصـر التـحـديـات القـائمـة، وتحـديـد وتقييم المخاطر المؤسسيـة المتـوقـع حدوثـها، والـتـي قـد تـؤثـر عـلـى تحقيـق أهـداف الـوزارة، وتقـديـم المقترحات بشأنها، بالتنسيق مع الوحدات الإدارية المعنية.
٣- إدارة وتجـهـيـز خـطط بديلة تضمن استمراريـة تشغيـل الأعمـال بالـوزارة في حـالات الطوارئ والأزمـات، بالتنسيـق مـع الـوحـدات الإداريـة المختـلفـة كـل فـيـما يخـصـها.
٤- رفـع مـشـروعـات الخطط التنفيـذيـة والمشـاريـع والبـرامـج الخـاصـة بالـوحـدات الإدارية المختـلفـة، للاعتمـاد مـن قبـل الـوزيـر.
٥ - متـابـعـة تنفيذ الخطط المتعلقة بأنشـطـة الـوزارة، وتقييـم نـتـائـجـها، ورفـع تـقـاريـر دوريـة بشـأنـها.
٦- دراسة المشـاكـل والمعـوقـات التـي تـواجـه تنفيـذ الخطـة الاستراتيجيـة والخطط التنفيـذيـة، واقتراح الحلول المناسبة لها.
۷- تقييـم جـودة الخـدمـات المقـدمـة مـن الـوحـدات الإداريـة، بالتنسيـق مع الجهـة المختـصـة.
۸ - مراجـعـة وتقييـم جـودة الأداء علـى مـسـتـوى الوحـدات الإداريـة، ووضـع مـقـتـرحـات لرفع الكفاءة الـعـامـة بـها.
٩- التحـقـق مـن انسجام نـظـم وأسـالـيـب العـمـل مـع اخـتـصـاصـات وأهـداف الوزارة، واقـتـراح خطـط تـطـويـر تـلـك النظـم والأساليـب، وتبسيط الإجراءات، بالتنسيق مع الوحدات الإدارية المعنية والجهة المختصة.
۱۰- دراسـة مـشـاكل ومعـوقـات العمـل بالـوزارة، وبحـث أسبـابـها، واقـتـراح الحلول المنـاسبـة لـها.
۱۱- تلقي المبادرات والمقترحات والأفكار الإبـداعـيـة، ودراستها بالتنسيـق مع الوحدات الإداريـة المعنيـة والجـهـة المختـصـة.
۱۲- رصد وتجميـع وتحـلـيـل البيانات الإحصائيـة الخـاصـة بأنشـطـة الـوزارة، والتنسيق مع الوحدات الإدارية المختصـة بشـأن التقاريـر الإحصائيـة المطـلـوبـة مـن الـوزارة.
۱۳- التنسيـق مـع الوحدات الإداريـة المعنيـة والجهـة المختصة فيما يتعلـق بالمؤشرات الـدوليـة المتعلقة بأنشـطـة واخـتـصـاصـات الوزارة.

 

المادة 6


تختص إدارة الشـؤون الـقـانـونيـة بمـا يـلـي:
١- بحـث ودراسـة ومـتـابـعـة المسـائـل القـانـونيـة الخـاصـة بنشـاط الـوزارة، بالتنسيق مع الوحدات الإداريـة المعنيـة.
٢- إعـداد مشـروعـات الأدوات التشريعيـة الخـاصـة بالـوزارة، وإبـداء الرأي في مشـروعـات القـوانـين التي تُحال إليـها.
٣- إبـداء الـرأي القـانـوني في المـوضـوعـات التي تُحال إليـها.
٤- إعـداد مشـروعـات العقـود والاتفاقيـات ومـذكـرات التفـاهـم المتعلقة باخـتـصـاصـات الوزارة، بالتنسيق مع الوحـدات الإدارية المعنيـة.
٥- التحقيـق في الوقـائـع والمخالفــات المنسوبـة لموظفي الـوزارة، وإعـداد المذكرات اللازمة بنتائج التحقيـق مـشـفـوعـة بـالـرأي القـانـوني والتـوصـيـات، وعـرضـها على السلطة المختصة، ومتابعـة تنفيـذ ما يُتخذ مـن قـرارات بشـأنـها.
٦- مـتـابـعـة المنـازعـات والقـضـايـا الـتـي تكـون الـوزارة طرفاً فيهـا، بالتنسيـق مـع الجـهـات المخـتـصـة.

 

المادة 7


تختص إدارة العـلاقـات الـعـامـة والاتـصـال بمـا يـلـي:
١- إصـدار النشرات والبـرامـج الإعـلاميـة التي تهدف إلى التـعـريـف بـدور الوزارة وأنشطتها واخـتـصـاصـاتـها، بالتنسيق مع الوحـدات الإداريـة المعـنـيـة.
۲- رصـد ومـتـابـعـة مـا يُنشر في الصـحـف ووسـائـل الإعـلام ومـواقـع التـواصـل الاجـتـمـاعـي مـن مـوضـوعـات تتعلق بالوزارة واخـتـصـاصـاتـها، وعـرضـها على المسؤولين، وإعـداد الـردود المنـاسبـة، بالتنسيـق مـع الـوحـدات الإداريـة المعـنـيـة والجهات المخـتـصـة.
٣- التنسيق مع الوحدات الإداريـة المختصة بشأن إعـداد خطط وبـرامـج التـوعـيـة والتثقيـف، وتطبيقهـا مـن خـلال الفعاليـات والأنشـطـة المختـلفـة بالتنسـيـق مـع الجـهـات المعنيـة.
٤- تلقـي طـلـبـات وشكاوى المـراجـعـين وإحـالـتـها إلى الوحدات الإداريـة المعنيـة، والرد على اسـتـفـسـاراتـهم.
٥- القيام بترتيبات السفر والضيـافـة والإقـامـة لضيوف الوزارة وللـوفـود الـزائـرة والمغادرة، بالتنسيـق مـع الوحدات الإدارية المعنية.
٦- تنظيم الأنشـطـة الثقافيـة والاجتماعيـة والرياضيـة لموظفي الوزارة، والإشراف على الاحتفالات التي تقيمها أو تشارك فيها الوزارة.
۷- تنظيم المؤتمرات والندوات والمعـارض التي تقيمهـا الـوزارة، وإعـداد الموازنة اللازمة لذلك، بالتنسيق مع الوحدات الإدارية المعنيـة.
۸- استقبال طلبـات كبار السن والأشخاص ذوي الإعـاقـة، بالتنسيـق مـع الوحـــدات الإداريـة المختصـة، لإنجـاز مـعـامـلاتـهـم وتـذليـل أي مشكلات أو صـعـوبـات قـد تـواجـهـهـم.

 

المادة 8


تختص إدارة الموارد البـشـريـة بمـا يـلـي:
١- تنفيذ القوانين واللوائح والنـظـم المتعلقة بشـؤون الموظفين.
٢- إعداد الدراسـات الخـاصـة بتخطيط الموارد البشـريـة بالـوزارة، وتحـديـد احتياجـات الـوزارة مـن الوظـائـف والموظـفـين، بالتنسيـق مع الجهـة المختـصـة والوحـدات الإداريـة المختـلفـة.
٣- إعـداد مشروع وصـف وتصنيـف وترتيـب الوظــائف، ومتـابعـة تنفيـذه، وتطـويـره.
٤- إعـداد مشـروع مـوازنـة البـاب الأول، بالتنسيـق مـع الـوحـدات الإداريـة المعنية.
٥- تنفيذ إجـراءات تقييـم الأداء للمـوظـفـين.
٦- القيام بإجـراءات التـعـيـين والنـقـل والنـدب والإعـارة.
۷- دراسة استحـقـاق الموظفين للإجـازات، وفقاً للـقـانـون.
٨- إعداد الدراسات الخاصة بالهيكل الوظيفي للوزارة، وتقـديـم المقـتـرحـات اللازمة لتطـويـر التنظيـم الإداري بـها.
۹- تحـديـد الاحتياجات التـدريبيـة لمـوظـفـي الـوزارة، بالتنسيـق مـع الوحـدات الإدارية المختلفة، ومتابعة تنفيذها، وتقييـم مـدى الاسـتـفـادة مـنـها.
۱۰- اتخاذ الإجـراءات اللازمة لتخصيص السكن الحكومي وصـرف بدل الأثـاث لموظفي الوزارة، بالتنسيـق مـع الجـهـات المـخـتـصـة.
۱۱- وضـع وتحـديـث قـواعـد البـيـانـات الخـاصـة بشـؤون موظفي الوزارة.

 

المادة 9


تختص إدارة الشؤون المـالـيـة والإداريـة بمـا يـلـي:
۱- تنفيـذ القـوانـين واللـوائـح والنظم المالية والإداريـة المتعلقة بعمـل الوزارة.
٢- إعـداد مشـروع الموازنـة السنـويـة للـوزارة، وحـسـابـها الخـتـامي، بالتنسـيـق مـع الوحدات الإداريـة المعنية.
۳- تـوفـيـر احـتـيـاجـات الوزارة ووحداتها الإدارية المختلفـة مـن المـسـتـلـزمـات والأجـهـزة اللازمة لأداء مهـامـها، بالتنسيق مع الوحدات الإداريــة المعنية.
٤- القيام بأعمـال المشـتـريـات والمنـاقـصـات والمـزايـدات، وفـق الأنظمـة والقـواعـد المعمـول بها في الدولة.
٥- الإشـراف على تنفيـذ الاعتمادات، وتـدقـيـق حـسـابـات الإيرادات والمصـروفـات.
٦- تسـلم وتسـلـيـم وتصنيف وقيـد البـريـد الـوارد والصـادر.
٧- تنـظـيـم أرشـيـف الوزارة، وحـفـظ الـوثـائـق، وفقاً لأحدث الطرق.
٨- إعـداد مستندات الصرف، وسـائـر المعـامـلات الماليـة الأخـرى.
۹- تحـصيـل الـرسـوم ومـقـابـل الخـدمـات التي تؤديـها الوزارة.
۱۰- القيام بجـمـيـع أعـمـال الخـدمـات الإداريـة.
١١- الإشراف على مـخـازن الوزارة.

 

المادة 10


تختص إدارة نـظـم المـعـلـومـات بمـا يـلـي:
١- إعـداد خطط وسيـاسـات اسـتـخـدام الحاسب الآلي في أنشـطـة الـوزارة، ومـتـابـعـة تـنـفـيـذهـا.
۲- برمجـة وحـفـظ واستـرجـاع وتـطـويـر نـظم المعلومات والبيانات اللازمــة لأنشـطـة الـوزارة.
٣- تأمين الأنظمـة والبرمجيـات والشبكات والبنية التحتيـة لنـظم المعـلـومـات مـن مـخـاطـر الحـوادث الإلكتـرونيـة.
٤- إعـداد خطط التعـافـي مـن آثـار الحـوادث الإلكتـرونيـة حـال وقـوعـها أو تقليـل حـدوثها أو تصعيـدهـا، وفـق الـسيـاسـات والمبادئ التـوجـيـهيـة لأمـن المعـلـومـات، بالتنسيق مع الجهـة المـخـتـصـة.
٥- توفير وصيـانـة الأجهـزة والبـرامـج والشبكات الإلكتـرونيـة اللازمـة لنـظم العمـل الآليـة بالـوزارة، بالتنسـيـق مـع الـوحـدات الإداريـة المعـنيـة.
٦- تصـمـيـم وتشـغيـل وإدارة قـواعـد البـيـانـات والمـعـلـومـات بأنـواعـها المختلفة.
۷- تقـديـم الدعم الفني اللازم لمسـتـخـدمي الشبكات الإلكتـرونيـة بالوزارة، وتدريبهم على استخـدام الأنظمـة والبـرامـج وأجـهـزة الحـاسـب الآلي وملحـقـاتـها.
٨- إنشاء ومتـابـعـة وتحديث موقع الوزارة على شبكة المعلـومـات الـدوليـة.

 

المادة 11


تختص إدارة الـضـمـان الاجـتـمـاعـي بمـا يـلـي:
۱- تـلـقـي ودراسـة طـلـبـات الحـصـول علـى خـدمـات الضـمـان الاجـتـمـاعـي والمـسـاعـدات الاجـتـمـاعـيـة الأخـرى.
۲- تقـريـر حـالات المسـتـفـيـدين بمعـاش الضمان الاجتماعي، وتحـديـد مـقـداره، وفقاً لشروط وضوابط اسـتـحـقـاقـه.
۳- متـابـعـة حـالات المستـفـيـديـن بمعـاش الضمان الاجـتـمـاعي، للتـأكـد مـن اسـتـمـرار تـوافـر شـروط اسـتـحـقـاقـهـم.
٤- اتخاذ التدابير اللازمـة لتنفيـذ البـرامـج لتـأهـيـل المنتـفـعـيـن بنـظـام الضمان الاجتماعي، وفقاً لأحـكـام الـقـانـون، بالتنسيـق مـع الجـهـات المختصة.
٥- إعداد الدراسات والبحوث في مجـال الضمان والتأمين الاجتماعي، وإعـداد التـقـاريـر اللازمـة بـشـأنـها.
٦- تـقـديـم خـدمـات الإغاثـة الـعـاجـلـة للمنكوبين، وفقاً للقوانين واللوائح المعـمـول بـها، بالتنسيق مع الجـهـات المختصـة بـالـدولـة.

 

المادة 12


تختص إدارة تـنـمـيـة المجـتـمـع بما يلي:
۱- اقـتـراح السيـاسـات والاستراتيجيـات والخطط الوطنيـة الخـاصـة بالتنميـة الاجتماعيـة، بالتنسيـق مـع الجـهـات المعـنـيـة والوحدات الإدارية المخـتـصـة بالوزارة.
٢- إصـدار التعليمـات والضـوابـط الـرقـابيـة المنـظـمـة لمقـدمي الخدمات الاجتماعيـة في الدولـة، والتـعـاون معـها في تطبيـقـها، وتـقـديـم الـدعـم اللازم لـرفـع قـدراتـها.
۳- تنـظـيـم وتـطـويـر الـعـمـل الاجتماعي وفـق المعـايـيـر الـوطنيـة والدوليـة، بالتنسيق مع الجـهـات المعـنيـة.
٤- وضـع شـروط ومعـايـيـر إنـشـاء الـكـيـانـات الاجـتـمـاعـيـة، بالتنسيق مع الجهات المعنيـة.
٥- رصـد ودراسـة القـضـايـا المتعلقة بمجال التنمية الاجتماعية، والظـواهـر الاجتماعيـة النـاجـمـة عـن تـطـور المجتمع، واقـتـراح الحلول المنـاسبـة لعلاجـها.
٦- إعـداد البحوث والدراسات الخاصة بمجـال العمـل الاجـتـمـاعـي وتنميـة المجتمع، وإعـداد التـقـاريـر اللازمـة بـشـأنـها.
۷- دعـم البحوث والدراسات الاجتماعية، بالتنسيق مع الجهات المعنيـة.
 

 

المادة 13


تختص إدارة إسكان المـواطـنـين بما يلي:
١- تنفيذ أحكام الـقـوانـين واللوائح والقرارات المتعلقة بالإسكان.
۲- اقتراح وتنفيذ السياسات المتعلقة بنظام الإسكان والقرارات المنفذة لها.
٣- إجـراء الدراسات والبـحـوث اللازمـة لـرسـم وتـطـويـر سيـاسـات الإسـكان في الـدولـة، وإعـداد التـقـاريـر اللازمـة بشـأنـها.
٤- اقتراح النماذج الهنـدسيـة للمساكن، بالتنسيق مع الجهات المختصة.
٥- التنسيق والتـعـاون مـع الجـهـات الحكومية والمؤسسات الماليـة الـتي يتـصـل نـشـاطـهـا بـعـمـل الإدارة.
٦- تـلـقـي ودراسـة طـلـبـات الانتفاع بنظام الإسكان، واتخاذ الإجـراءات اللازمـة بشـأنها، وفقاً للقـوانـين والقـواعـد والإجـراءات المقررة.
۷- تـلـقـي الشكاوى المتعلقة بالانتفاع بنظام الإسكان، والعـمـل علـى تسـويتـها طبقاً لأحـكـام الـقـانـون.
٨- إعـداد قـاعـدة بـيـانـات حـول إسكان المـواطـنين، وتحديثهـا، وإعـداد الـتـقـاريـر اللازمـة بشـأنـها.
٩- الإشراف على بناء المساكن والوحدات السكنية اللازمة لذوي الحـاجـة، واتخاذ اللازم لبنـاء الإضـافـات، وصيـانـة المسـاكـن التـابـعـة لـلإدارة أو هـدمـها وإعـادة بنـائـها، وتسليمها للمنتـفـعـيـن بـها.

 

المادة 14


تختص إدارة الجـمـعـيـات والمـؤسـسـات الخـاصـة بمـا يـلـي:
۱- تنفيذ القـوانـين واللوائح والقرارات المتعلقة بالجمـعـيـات والمـؤسـسـات الخاصة.
۲- تسـجيـل وشـهـر الجمعيات والمؤسسات الخـاصـة، والإشراف عليها.
٣- الاحـتـفـاظ بمـلـف لـكـل جـمـعـيـة أو مـؤسسـة خاصـة يـضـم نـسـخـة مـن عـقـد أو وثيقة تأسيسها بحـسـب الأحـوال، ونـظـامـها الأساسي، وشـهادة تسجيلها، والقرارات الصادرة بشـأنـها.
٤- تلقي ودراسـة طـلـبـات دعـم الجـمـعـيـات والمـؤسـسـات الخـاصـة بما يـسـاعـدهـا على أداء أعـمـالـهـا وتحـقـيـق أهـدافـها.
٥- إعداد الدراسات والبحوث المتعلقة بالجمعيات والمؤسسات الخـاصـة، وإعـداد التـقـاريـر اللازمـة بـشـأنـها.
٦- تنـظـيـم وتـطـويـر الـعـمـل الاجـتـمـاعـي التـطـوعي، والإشراف والـرقـابـة عليـه.

 

المادة 15


تختص إدارة الحـمـايـة الاجـتـمـاعـيـة بما يلي:
۱- تـوعـيـة المجـتـمـع بمفهـوم الحـمـايـة الاجـتـمـاعيـة.
٢- الإشراف على دور الـرعـايـة الاجتماعيـة المخصصة لإيـواء ورعـايـة وتأهيـل الفـئـات المسـتـهـدفـة.
٣- إعـداد وتنفيـذ بـرامـج الحمـايـة الاجـتـمـاعـيـة للـوقـايـة مـن انـحـراف الأطـفـال، وتأهيـل المنحرفين، ومـواجـهـة المشكلات الاجتماعيـة.
٤- اتخاذ الإجـراءات اللازمـة لـرعـايـة وحـمـايـة الأطـفـال مـجـهـولي الأب والأبـويـن.
٥- اتخاذ الإجـراءات اللازمـة للانـدمــــاج الاجتمـاعـي مـع الجهـات الحكومية وغير الحكومية المخـتـصـة.
٦- مسـاعـدة المسجونين المفرج عنهم والمدمنين المتـعـافـين وأسـرهـم عـلـى التكيف والانـدمـاج الاجـتـمـاعي، بالتنـسـيـق مـع الجـهـات الحكوميـة وغيـر الحكومية المخـتـصـة.
٧- تـوفـيـر بـيـئـة صـالحـة ومـنـاخ اجـتـمـاعـي مـنـاسـب وفرص عـمـل للمفـرج عنـهم للحيـلـولـة دون عـودتـهـم إلى الانحراف، وذلك بالتنسيـق مـع الجـهـات الحكومية وغيـر الحكومية المخـتـصـة.
٨- إعداد الدراسات والبحـوث المتعلقة بالحـمـايـة الاجـتـمـاعيـة، وإعـداد التـقـاريـر اللازمـة بشـأنـها.

 

المادة 16


تختص إدارة الـتـنـمـيـة الأسـريـة بما يلي:
١- إعداد وتنفيذ الاستـراتـيـجـيـات والخطط والسياسات الوطـنـيـة المتعلقة بالأسرة والمرأة والطـفـل.
۲- تـوعـيـة المجتمع بالتحـديـات والقضـايـا الأسـريـة والاجتماعيـة المعـاصـرة وآثـارهـا وطـرق الـوقـايـة مـنـها.
٣- إعـداد وتنفيـذ مـشـاريـع وبـرامج تنميـة وتـطـويـر إبـداعـات الأطفـال والنـاشـئـة ثـقـافـيـاً وتـعـلـيـميـاً وتـرفـيـهـياً، بالتنسيق مع الجهات الحكومية وغيـر الحكومية المخـتـصـة.
٤- تنفيـذ بـرامـج تنـميـة وتـطـويـر قـدرات المـرأة، بالتـعـاون مـع الجـهـات الحكومية وغيـر الحـكـوميـة المخـتـصـة.
٥- تـوعـيـة وتثقيـف المجـتـمـع بـحـقـوق المـرأة والـطـفـل، وتنفيـذ بـرامـج التـدريـب للـعـامـلين معـهم، بالتعاون مع الجهات الحكومية وغيـر الحكوميـة المخـتـصـة.
٦- تنـظـيـم المـؤتـمـرات والندوات وورش العـمـل لمنـاقـشـة قـضـايـا الأسـرة والمرأة والـطـفـل، بالتنسيـق مـع الـوحـدة الإدارية المختصة.
٧- إعـداد الدراسات والبحـوث المتعلقـة بالتنميـة الأسـريـة، وإعـداد التقارير اللازمـة بشـأنـها.
۸- جمع وتحـلـيـل البيانات الإحصائيـة المتعلقة بالأسـرة، بالتنسيـق مـع الـوحـدة الإداريـة المعـنيـة.

 

المادة 17


تختص إدارة الـرعـايـة المجـتـمـعـيـة بمـا يـلـي:
١- إعداد وتنفيذ الاستـراتـيـجـيـات والخطط والسياسات الوطنية المتعلقة بكـبـار الـسـن والأشخاص ذوي الإعـاقـة.
۲- تـطـويـر وتـنـفـيـذ البـرامـج والخـدمـات اللازمـة لـرعـايـة وتـأهيـل كـبـار الـسـن والأشخاص ذوي الإعـاقـة، بالتـعـاون مـع الجـهـات الحكوميـة وغيـر الحـكـوميـة ذات الـعـلاقـة.
۳- تـوعـيـة وتثقيـف المجتمـع بـحـقـوق كبـار السـن والأشـخـاص ذوي الإعـاقـة، وتنفيـذ بـرامـج التـدريـب المـوجـهـة للعـامـلـين مـعـهم، بالتـعـاون مع الجهات الحكومية وغيـر الحـكـوميـة المخـتـصـة.
٤- اقـتـراح تـعـديـل التـشـريـعـات المتعلقـة بـكـبـار الـسـن والأشـخـاص ذوي الإعـاقـة، بالتـعـاون مع الإدارة المخـتـصـة بالوزارة.
٥- تأهيـل الأشخـاص ذوي الإعـاقـة بالمـهـارات الاجتماعيـة اللازمـة للتكيـف والانـدمـاج في المجتمع.
٦- تنـظـيـم الـدورات والمؤتمرات وورش العـمـل لمنـاقـشـة القـضـايـا المتعلقة بـكـبـار الـسـن وبالأشـخـاص ذوي الإعـاقـة، بالتنسيـق مـع الجهات المعنيـة.
٧- إعـداد قـاعـدة مـتـكـامـلـة مـن البيـانـات والمـعـلـومـات الخـاصـة بـكـبـار الـسـن والأشخاص ذوي الإعـاقـة.
٨- إعـداد الـدراسـات والبـحـوث المتعلقـة بـكـبـار الـسـن والأشـخـاص ذوي الإعـاقـة، وإعداد التقارير اللازمـة بشـأنـها.

 

المادة 18


تختص إدارة التمكين الأســري بمـا يـلـي:
۱- دعم وتشجيع المشـروعـات التـنـمـويـة المـوجـهـة لتمكين الأفراد والأسـر اقـتـصـاديـاً، بالتنسيق مع الجهات ذات العلاقة.
۲- تـوجـيـه وتـشـجيـع جـهـود الـقـطاع الخاص ومنظمات المجتمـع المـدني بهدف إنشاء أو تمـويـل مـشـروعـات تـنـمـويـة لتمكين الأفراد والأسـر اقتصـاديـاً.
۳- تـطـويـر وتنفيذ مـشـروعـات دعـم وتـنـميـة الأسـر المنتـجـة مـن خـلال التـدريـب وتـوفـيـر الخـدمـات المـسـاعـدة.
٤- تـوفـيـر المجـالات والفـرص المنـاسبــة لـتـسـويـق مـنـتـجـات الأفـراد والأسر المنتـجـة.
٥- تشجيع الادخـار الـعـائـلـي وتـرشـيـد الاسـتـهـلاك.
٦- تـقـديـم الإرشاد والتـوجـيـه والاستشارات الفنيـة والـدعـم اللازم لتـحـقـيـق اسـتـقـرار الأسر المنتـجـة.
٧- إعـداد الإحصاءات والدراسات والبحوث المتعلقة بالفئات المستـهـدف تمكينها من الأفراد والأسـر.

 

المادة 19


يصـدر بتحـديـد اخـتـصـاصـات كـل مـن مـكـتـب الوزيـر، ومكتـب وكيـل الوزارة، ومكتـب وكيـل الـوزارة المسـاعـد، قـرار مـن الـوزيـر.

 

المادة 20


يجـوز بقـرار مـن مـجـلـس الـوزراء، بناءً علـى اقـتـراح الـوزيـر، تعـديـل تنظيم الوحدات الإداريـة الـتـي تتـألـف منهـا الـوزارة بالإضـافـة أو الحـذف أو الدمـج، وتعيين اخـتـصـاصـاتـهـا وتـعـديـلـها.

 

المادة 21


يجـوز بقـرار مـن الوزير إنشاء ما تقتضيه المصلحـة العـامـة مـن أقسـام في الوحدات الإداريـة التي تتـألـف منهـا الـوزارة، أو إلغاؤهـا أو دمجها، وتعيين اخـتـصـاصـاتـهـا وتـعـديـلها، ولا يـكـون قـرار الوزيـر نـافـذاً إلا بعـد اعتماده مـن مـجلـس الـوزراء.

 

المادة 22


على جميع الجهات المختصة، كـل فيما يخصه، تنفيذ هـذا القـرار. ويُعمل بـه مـن تـاريـخ صـدوره. ويُنشر فـي الجـريـدة الـرسـميـة.
