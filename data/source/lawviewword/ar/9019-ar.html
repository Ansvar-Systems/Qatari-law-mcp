وعلـى القـرار الأميـري رقـم (12) لسنـة 2019 بالهيكل التنظيمي لوزارة الـتـجـارة والصنـاعـة،

المواد

المادة 1


يتكون الهيكل التنظيمـي لـوزارة التـجـارة والصنـاعـة مـن الـوحـدات الإداريـة المبينة بالخريـطـة التنظيـميـة المرفقة بهـذا الـقـرار، وهي:
أولاً: الوحدات الإداريـة التـابـعـة للـوزيـر:
۱- مـكـتـب الـوزيـر.
۲- المـكـتـب الفـنـي.
٣- إدارة التـدقـيـق الـداخـلي.
 
٤- إدارة التـعـاون الدولي والاتـفـاقـيـات الـتـجـاريـة.
٥- إدارة الشـؤون القـانـونيـة.
ثانياً: الوحدات الإداريـة التـابـعـة لوكيـل الـوزارة:
۱- مكتـب وكيـل الـوزارة.
٢- إدارة التـخـطـيـط والجـودة والابتكار.
ثالثاً: الوحدات الإداريـة التـابـعـة لوكيل الوزارة المسـاعـد لشؤون التـجـارة:
۱- مكتـب وكيـل الـوزارة المسـاعـد.
٢- إدارة التـسـجـيـل والتراخيص التـجـاريـة.
٣- إدارة شـؤون الشـركـات.
٤- إدارة حـمـايـة حـقـوق الملكيـة الفـكـريـة.
٥- إدارة النـافـذة الـواحـدة.
رابعـاً: الوحدات الإداريـة التـابـعـة لوكيـل الـوزارة المسـاعـد لشـؤون الـصـنـاعـة وتنميـة الأعـمـال:
۱- مـكـتـب وكيـل الـوزارة المسـاعـد.
٢- إدارة الـتـنـمـيـة الصـنـاعـيـة.
٣- إدارة دعـم تنـافـسـيـة المنتـج الـوطـني.
٤- إدارة تنميـة الأعـمـال.
٥- إدارة تنميـة التبادل التجـاري وترويج الاستثمـار.
خامساً: الوحدات الإداريـة التابعـة لوكيـل الـوزارة المسـاعـد لشـؤون المستهلك:
۱- مكتـب وكيـل الـوزارة المسـاعـد.
٢- إدارة حـمـايـة المستهلك ومـكـافـحـة الـغـش الـتـجـاري.
 
٣- إدارة التـمـويـن والمـخـزون الاستراتيجي.
٤- إدارة التـراخـيـص الـنـوعـيـة ومـراقـبـة الأسـواق.
٥- إدارة حـمـايـة المنـافـسـة.
سادساً: الوحدات الإداريـة التـابـعـة لوكيـل الـوزارة المسـاعـد لـشـؤون الخـدمـات المشتـركـة:
۱- مكتـب وكيـل الـوزارة المسـاعـد.
٢- إدارة العـلاقـات الـعـامـة والاتصـال.
٣- إدارة الموارد البشـريـة.
٤- إدارة الشـؤون المـاليـة والإداريـة.
٥– إدارة نـظـم المـعـلـومـات
 

 

المادة 2


يختص المكتـب الـفـنـي بما يلي:
١- إبداء الرأي الفـني في المـوضـوعـات التي تُحـال إليـه مـن الوزيـر.
۲- دراسـة وتقييم التقـاريـر والمشـروعـات والبـرامـج المتعلقة بأنشـطـة الوزارة، التي تُحال إليـه مـن الوزيـر.
۳- بحث ودراسـة ومـتـابـعـة المـوضـوعـات الـتـي تُحـال إليـه مـن الوزيـر، بالتنسيق مع الوحـدات الإداريـة المعـنيـة.
٤- إعـداد ملخص للمـواضـيـع والتقاريـر والـدراسـات الـتـي يـلـزم عـرضـها على الـوزيـر، وإبـداء الـرأي بشـأنـها.
 

 

المادة 3


تختص إدارة التـدقـيـق الـداخـلـي بمـا يـلـي:
١- إعداد مشروع خـطـة التـدقيـق السنـويـة المبنيـة علـى المخـاطـر للوزارة، ورفعـهـا للـوزيـر لاعتمـادهـا، وإعـداد تقاريـر بـنـتـائـج التـدقـيـق.
۲- مـراقبـة مـدى الالتزام بتنفيذ القـوانـين واللوائح والقرارات المنظمة للعمـل.
۳- مـراجعـة التعليمات الماليـة، واقـتـراح مـا يـلـزم لـهـا مـن تعـديـلات، ومـراقبـة تنفيـذهـا.
٤- مـراقبـة الأعمـال الإداريـة وإجـراءات تعيين الموظفين، وإجازاتـهم، وتـرقـيـاتـهم، وغير ذلـك ممـا يتعلـق بـشـؤون الموظفين، والتحقـق مـن الالتزام بالقـوانـين واللوائح والقرارات المنـظـمـة لذلك.
٥- مـراقبـة الأعـمـال المـاليـة والمحاسبيـة، وسنـدات الصرف، وإجـراءات القيد في السجلات المحاسبية، والتـأكـد مـن تطبيـق الـقـواعـد المعتمدة في هـذا الشـأن.
٦- تقييـم المخاطـر المتعلقة بالتدقيـق المالي والإداري، واقتـراح إجـراءات الحـد منـها.
۷- متـابـعـة الإجـراءات التصحيحيــة المطـلـوبـة مـن الـوحـدات الإداريـة المعـنـيـة، وفـق نتائج التـدقيـق، وإعـداد التقارير الدوريـة بشـأنـها.
۸- التـأكـد مـن سـلامـة الأمـوال والعهـد والأثـاث والأدوات والمبـاني والمـوجـودات الأخرى التي تملكها الوزارة أو تخضع لإشـرافـها.
۹- الاشتـراك في أعمـال الجـرد الـدوري والمفاجئ على المخـازن والعهـد والتـأكـد مـن دقـة التسجيـل بـالـدفـاتـر وسـلامـة الإجـراءات المطبقـة.
۱۰- متـابعـة نتائج تقاريـر الجـهـات الـرقـابيـة، وإبـداء التـوصـيـات المنـاسبـة بشـأنـها
 

 

المادة 4


تختص إدارة التعاون الدولي والاتـفـاقـيـات الـتـجـاريـة بما يلي:
١- إعـداد أوراق العمـل المتعلقـة بالمؤتمرات والندوات والاجتمـاعـات المحلية والإقليميـة والدوليـة فـيـمـا يـخـص نشـاط الـوزارة، بالتنسيـق مع الوحـدات الإداريـة المعنيـة والجـهـات المختـصـة.
٢- الإعـداد لحضـور المؤتمرات والاجتمـاعـات ولجان العمـل المشتـركة واللجـان الـفـنـيـة، ومـا فـي حـكـمها، الـتـي تـعـقـدهـا المنظمـات والهيئات الإقليميـة والـدوليـة المخـتـصـة بقضايا التجـارة والصنـاعـة، والتي تشـارك الدولة في عضـويتـها.
٣- إعـداد التقاريـر والـردود التي تطلبهـا المنظمات والهيئات الإقليميـة والدوليـة فيمـا يـخـص نشـاط الـوزارة، بالتنسـيـق مـع الـوحـدات الإدارية المخـتـصـة.
٤- دراسـة ومـتـابـعـة تنفيـذ تـوصـيـات المنظمـات والهيئـات الإقليميـة والدوليـة فـيـمـا يـخـص نشـاط الـوزارة، بالتنسـيـق مـع الجـهـات المخـتـصـة بـالـدولـة.
٥- دراسـة ومـتـابـعـة تـنـفـيـذ التـزامـات وقرارات وتوصيات منظمـة التجـارة العـالميـة، وإعـداد التقارير اللازمـة عـن أنشطتها، بالتنسيـق مـع الجهات ذات العلاقة بالدولة، ومتـابـعـة تـنـفـيـذ تلك الجـهـات للالتزامـات والقرارات والتـوصـيـات الصـادرة عـن المنظمـة وفقاً للالتـزامـات الـواردة بقـوائـم عـروض الدولـة.
٦- إعـداد التقاريـر اللازمـة بشـأن إسـهـامـات المنظمـات والهيئـات الإقليميـة والدوليــة في تنميـة وتـدعـيــم الأنشـطـة المختلفـة في مجـالات عمـل الوزارة، بالتنسيق مع الجهات المخـتـصـة.
۷- متـابـعـة سـداد المسـاهـمـات المـاليـة للدولـة في المنظمات المعنيـة بمجـالات عـمـل الـوزارة والتي تكـون الدولـة عـضـواً فيهـا، بالتنسيـق مـع الجـهـات المختصة.
۸- دراسـة السـبل الكفيلة بتفعيل دور الدولـة في المجالات التـجـاريـة والصناعيـة مع المنظمات والهيئات الإقـلـيـمـيـة والـدوليـة.
۹- جمـع المعلـومـات والبحـوث الصـادرة عـن المنظمات والهيئـات ذات الصلة بالتجارة والصناعـة، محلياً وإقليمياً ودوليـاً، ونشـرهـا بغـرض الاسـتـفـادة مـنها.
۱۰- المشاركة في إعـداد مشـروعـات الاتفاقيـات التجـاريـة والصناعيـة المتعلقـة باختصـاصـات الـوزارة، بالتنسيـق مـع الـوحـدات الإداريـة المعنيـة والجـهـات المختصة.
١١- التنسيـق مـع الوحدات الإداريـة المـعـنيـة بشـأن إبـرام الاتفاقيـات الاقـتـصـاديـة والتجـاريـة والصناعيـة ذات الصلة بعمـل الـوزارة، وحـضـور الاجتـمـاعـات واللجـان المنبثقة عنـها.
١٢- التنسيـق مـع الوحدات الإداريـة المعـنـيـة فيما يتعـلـق بإعـداد أوراق العـمـل اللازمـة للـفـريـق الـتـفـاوضـي للـدولـة.
۱۳- تـرجمـة الوثـائـق والـكتـب والبحوث العلميـة والمكاتبـات الأجنبيـة المتعلقة باخـتـصـاصـات الـوزارة.
 

 

المادة 5


تختص إدارة الشؤون القـانـونيـة بمـا يـلـي:
۱- بحـث ودراسـة ومـتـابـعـة المسـائل القـانـونيـة الخـاصـة بنشـاط الـوزارة، بالتنسيق مع الوحدات الإداريـة المعنيـة.
٢- إعداد مشـروعـات الأدوات التشريعيـة الخـاصـة بالوزارة، وإبـداء الـرأي في مشـروعـات القـوانـين التي تُحـال إليـها.
٣- إبـداء الـرأي الـقـانـوني في المـوضـوعـات التي تُحـال إليـها.
٤- إعـداد مشـروعـات العقـود والاتفـاقـيـات ومـذكـرات التفـاهـم المتعلقـة باخـتـصـاصـات الـوزارة، بالتنسيق مع الوحدات الإدارية المعنيـة.
٥- التحقيـق في الـوقـائـع والمخالفـات المنسوبـة لموظفي الـوزارة، وإعـداد المذكــرات اللازمـة بنـتـائج التحقيـق مـشـفـوعـة بالـرأي الـقـانـوني والتـوصـيـات، وعـرضـها علـى السـلطـة المختصة، ومتابعـة تنفيذ ما يُتخذ مـن قـرارات بشـأنـها.
٦- متـابـعـة المنـازعـات والـقـضـايـا الـتـي تـكـون الـوزارة طـرفـاً فيهـا، بالتنسيـق مـع الجـهـات المختـصـة.
 

 

المادة 6


تختص إدارة التـخـطـيـط والجـودة والابتكار بمـا يـلـي:
١- إعـداد الخطة الاستراتيجيـة للـوزارة، بالتنسيق مع الجهات المختصة والوحـدات الإدارية المعنيـة، واتخاذ إجـراءات اعتمـادهـا.
۲- حـصـر التـحـديـات القـائمـة، وتحـديـد وتقييـم المخـاطـر المؤسسيـة المتـوقـع حـدوثـها، والتي قد تؤثـر علـى تـحـقـيـق أهـداف الوزارة، وتقـديـم المقترحات بشأنها، بالتنـسـيـق مـع الوحدات الإداريـة المعـنيـة.
 
٣- إدارة وتجـهـيـز خـطط بديلـة تضمن استمراريـة تشغيـل الأعمـال بالوزارة في حـالات الطوارئ والأزمـات، بالتنسيق مـع الوحـدات الإدارية المختـلفـة كـل فيما يخصها.
٤- رفع مشـروعـات الخطط التنفيـذيـة والمشـاريـع والبـرامـج الخـاصـة بالـوحـدات الإداريـة المختـلفـة، للاعتمـاد مـن قبـل الـوزيـر.
٥- متـابعـة تنفيذ الخطط المتعلقة بأنشـطـة الـوزارة، وتقييـم نـتـائـجـها، ورفع تقارير دوريـة بشـأنـها.
٦- دراسة المشـاكـل والمعـوقـات الـتـي تـواجـه تنفيذ الخطة الاستراتيجيـة والخطط التنفيـذيـة، واقتراح الحلول المنـاسبـة لها.
٧- تقييـم جـودة الخـدمـات المقـدمـة مـن الـوحـدات الإداريـة، بالتنسيـق مع الجهـة المختـصـة.
۸- مـراجـعـة وتقيـــم جـودة الأداء علـى مـسـتـوى الوحـدات الإداريـة، ووضع مقترحات لرفع الكفاءة الـعـامـة بـها.
٩- التحـقـق مـن انسجـام نـظـم وأسـالـيـب العـمـل مـع اختصـاصـات وأهـداف الوزارة، واقتراح خطـط تـطـويـر تـلـك النـظـم والأساليـب، وتبسيط الإجراءات، بالتنسيق مع الوحدات الإدارية المعنيـة والجهة المختصة.
۱۰- دراسـة مـشـاكل ومعـوقـات العمـل بالـوزارة، وبحـث أسبـابـها، واقـتـراح الحلول المنـاسبـة لـها.
۱۱- تلقـي المبــادرات والمقترحـات والأفكار الإبـداعـيـة، ودراستها بالتنسيق مع الوحدات الإدارية المعنيـة والجـهـة المختـصـة.
12- رصـد وتجـمـيـع وتحـلـيـل البيانات الإحصائيـة الخـاصـة بأنشـطـة الـوزارة، والتنسيق مع الوحدات الإدارية المختـصـة بـشـأن التقارير الإحصائيـة المطـلـوبـة مـن الـوزارة.
١٣- التنسيـق مـع الوحدات الإداريـة المعنيـة والجهـة المختصة فيما يتعلـق بالمؤشرات الـدوليـة المتعلقة بأنشـطـة واخـتـصـاصـات الوزارة.
 

 

المادة 7


تختص إدارة التسجيـل والـتـراخـيـص الـتـجـاريـة بمـا يـلـي:
١- تنفيذ أحكام القـوانـين واللوائح المنظمـة للشـؤون التجـاريـة، وعلى الأخص السـجـل التجـاري، وسـجـل الـوكـلاء التـجـاريـين، ومزاولـة الأعمال التجـاريـة، والشـركـات، والسجـل الاقتصادي المـوحـد.
٢- الإشراف على تنـظيـم ومـراقـبـة الأسـواق، في مجـال اخـتـصـاص الوزارة، بالتنسيـق مـع الجـهـات المختصة.
٣- القيـد في السجل التجـاري وسـجـل الـوكـلاء التجـاريـين، والـرقـابـة على أنشطتهم.
٤- دراسـة طلبات التراخيص للأنشطة التجـاريـة، وإبـداء الـرأي بشأنها.
٥- قيد الأنشـطـة الاقـتـصـاديـة والمهنيـة والحـرفـيـة.
٦- إصـدار التراخيص التـجـاريـة المتـعـلـقـة بمزاولـة الأعـمـال.
٧- المصادقة على شهادات المنشأ القـطـريـة والخليجيـة والعـربـيـة والـدوليـة.
۸- مـراجـعـة عـقـود الوكالات التجـاريـة، والتـأكـد مـن اسـتـيـفـائـها للشروط الـواجـب تـوافـرهـا.
۹- مـتـابـعـة وتـنـفـيـذ الضـوابـط والشـروط المتعلقة بمبـدأ المعاملـة بالمثـل بالنسبـة للـدول المصدرة للسلع والبضائـع الـتـي يـكـون لـهـا وكـلاء محليين في الـدولـة
 

 

المادة 8


تختص إدارة شـؤون الـشـركـات بمـا يـلـي:
١- تنفيذ أحـكـام القـوانـين واللـوائـح المنـظـمـة لتـأسـيـس ومـراقـبـة الشـركـات التـجـاريـة.
۲- تنفيـذ التشريعـات المنـظـمـة لخـدمـات الأعـمـال الاستشـاريـة في نطـاق اختصاصات الوزارة.
٣- الإشراف على تـأسـيـس الشـركـات التـجـاريـة.
٤- التفتيــش عـلـى الـشـركـات الـتـجـاريـة، ومـراقـبـة تـنـفـيـذهـا أحـكام التـشـريـعـات التـجـاريـة.
٥- مـتـابـعـة تـنـفـيـذ أحـكام التشـريـعـات الخـاصـة بمـكـافـحـة غـسـل الأمـوال وتمـويـل الإرهـاب، وذلـك في نطـاق اخـتـصـاصـها، وبـعـد التنسيـق مع الجـهـات المختصـة بالـدولـة.
٦- تنظيـم مـهـنـة تـدقيـق ومـراجـعـة الحـسـابـات، وفقاً للقـوانـين واللـوائـح المنظمة.
٧- إعـداد سـجـلات مـدقـقـي الحـسـابـات، وقـيـدهـم فـي القـسـم المخـتـص، والإشراف على أعـمـالـهـم
 

 

المادة 9


تختص إدارة حـمـايـة حـقـوق الملكيـة الفـكـريـة بما يلي:
١- تـوعـيـة المـؤلـفـين وفـنــانـي الأداء بالطــرق المثلى لممارسـة حـقـوقـهم المالية والأدبـيـة ومسـاعـدتـهم.
۲- تـسـويـة المنازعات التي تنشأ بين المؤلفين أو فناني الأداء ومنتـجـي التسجيلات السمعيـة وأصـحـاب الحـقـوق وتـصـامـيـم الـدوائـر المـتـكـامـلـة وحـقـوق بـراءات الاخـتـراع والـغيـر، وفقاً لأحكام القـانـون، متـى اتـفـق الأطراف على ذلك.
۳- دراسـة ومـتـابـعـة مـا يُـثـار مـن قـضـايـا تـتـعـلـق بـحـق المـؤلـف والحـقـوق المجاورة على المستـوى المحـلـي والـدولي، وتقـديـم الاقـتـراحـات أو التـوصـيـات فـي هـذا الـشـأن.
٤- دراسـة طلبـات إيــــداع المصـنـفـات والتـسـجـيـلات السمعيـة وأداءات فـنـاني الأداء والبـرامـج الإذاعـيـة، والبـت فيـها.
٥- مـكـافـحـة كـل أنـواع الـتـعـدي عـلـى حـق المـؤلـف والحـقـوق المجاورة والأسـرار الـتـجـاريـة.
٦- استقبال طـلـبـات إيداع وتسـجـيـل بـراءات الاختراع وحـمـايـة تـصـامـيـم الدوائر المتكاملـة وفـحـصـها، والبـت فيـها، والإعلان عنها، وإعـداد السجلات الخاصـة بذلك.
۷- اسـتـقـبـال طـلبـات التـأشيـر بالتـغيـيـرات والتنـازل والتـرخـيـص الخاصـة ببـراءات الاخـتـراع وحـمـايـة تـصـامـيـم الـدوائـر المتكاملـة، وقـيـدهـا في السـجـلات في حـال قـبـولـها والإعـلان عنـها، وإصـدار الشـهـادات الخاصـة بـذلك.
۸- قيـد طـلـبـات تـسـجـيـل العـلامـات والبيـانـات والأسمـاء التجـاريـة، والـرسـوم والنماذج الصـنـاعـيـة، والمؤشرات الجـغـرافـيـة.
٩- تنفيذ أحكام التـشـريـعـات المتعلقة بـحـمـايـة حـقـوق الملكيـة الفـكـريـة وحـمـايـة الأسـرار الـتـجـاريـة، وأحكام الاتـفـاقـيـات الـدوليـة ذات الصلة التي انضمت إليـهـا الـدولـة.
١٠- إبـداء الـرأي في المسـائـل المتعلقـة بمـشـروعـات القـوانـين والاتفاقيـات الـدوليـة الخـاصـة بحـقـوق الملكيـة الفـكـريـة وحـمـايـة الأسـرار الـتـجـاريـة.
 

 

المادة 10


تختص إدارة الـنـافـذة الـواحـدة بمـا يـلـي:
۱- تنـظـيـم سـيـر العـمـل الخـدمـات النـافـذة الـواحـدة، وتأديـة كـافـة الخدمات اللازمة لتبسيط وتسهيل الإجـراءات الخاصـة بـبيـئـة الأعـمـال والاستثمـار، بالتنسيـق مـع الـوحـدة الإداريـة المعـنـيـة والجـهـات المختصة.
۲- تـوفـيـر كـافـة الخـدمـات الإداريـة والفـنـيـة اللازمة لتسهيـل العـمـل لخـدمـات النـافـذة الواحدة، بالتنسيق مع الجهات المعنيـة.
٣- التنسيـق مـع الوحـدات الإداريـة المعـنـيـة والجـهـات المختصـة بشـأن تقـديـم خـدمـات النـافـذة الـواحـدة، مـن أجـل إنجاز إجـراءات تأسيس الشركات التجـاريـة، وإجراءات إنشاء المصانع، وإجـراءات تأسيس أيـة كـيـانـات اقـتـصـاديـة أخـرى، وإنجاز إصـدار التراخيص اللازمة لممارسة أي من الأنشطة التجـاريـة أو الاقتصـاديـة المعتمدة في الدولة.
٤- تلقي ودراسـة شكاوى ومقـتـرحـات المـراجـعـين، بالتنسيـق مـع الـوحـدات الإداريـة والجـهـات المختصة، واتخاذ مـا يـلـزم بشـأنـها.
٥- التنسيق مع الجهات المختصة وذات العـلاقـة لتبسيط وتـطـويـر إجـراءات كـافـة الخـدمـات المقـدمـة بالنـافـذة الـواحـدة، وتطـويـر مـؤشـرات واضحة لقيـاس مـسـتـوى جـودة الخـدمـات، ووضع معيار زمني لـكـل منـها.
٦- التقييم الدوري للخـدمـات المقـدمـة مـن النـافـذة الـواحـدة، وتطبيـق مـفـاهـيـم الجـودة الشاملة لتحسين الأداء والارتقاء بمستـوى الخـدمـات، بالتنسيق مع الوحدات الإداريـة والجـهـات المختصة.
۷- اقـتـراح تـطـويـر آليـات النـظـم المـطـبقـة في مختـلف الجهات المعنيـة بالدولـة فـيـما يتعـلـق بـبيـئـة الأعـمـال والاستثمـار، بالتنسيـق مـع الـوحـدة الإدارية المختصة والجهات ذات العـلاقـة.
 

 

المادة 11


تختص إدارة التنميـة الـصـنـاعـيـة بمـا يـلـي:
١- إعـداد استراتيجية التنميـة الصـنـاعـيـة والخطط الصناعـيـة، ومـتـابـعـة تنفيـذهـا.
٢- إعـداد الـدراسـات الخـاصـة بمـواقـع المـوارد التـعـديـنـيـة وحـجـمـها ونـوعـها ومـواصـفـاتـها، وتحـديـد فـرص استثمارهـا.
۳- دراسـة فــرص الاسـتـثـمـار الـصـنـاعيـة الجـديـدة، وإعـداد دراسـات الجـدوى الاستثماريـة المبـدئـيـة للفـرص الـواعـدة مـنـها، والعـمـل عـلـى ترويجها في الداخل والخارج، بالتنسيق مع الجهات المختصة بالدولـة.
٤- إعـداد الـدراسـات والـتـقـاريــر عـن واقـع الصـنـاعـة والتنميـة الصناعيـة بالـدولـة، وسبـل تنميتها، وتقـديـم المقترحات والحلول لمعـوقـات الصـنـاعـة في القطاعات المختلفة، بالتعاون مع الجهات المخـتـصـة.
٥- إعـداد دراسات وخطط وبـرامـج شبكة النظـام المتكامـل للبيـانـات والمعـلـومـات المتعلقة بالأنشـطـة الصناعيـة بـالـدولـة، بالتنسيـق مـع الـوحـدة الإداريـة المعـنيـة.
٦- التـعـاون مـع الجـهـات والهيئات والمنظمات المتخصصة الخليجيـة والعـربـيـة والدولية والهيئات والمؤسسات والشركات الاستشـاريـة في مجـال إعداد الدراسات والبحوث والتسويق والتخطيط الصناعي، في الـداخـل والخـارج، وتـوطـيـد العـلاقـة مـعـها.
۷- وضـع الاشتـراطـات الـعـامـة الـتـي تـلتـزم بـهـا المشـروعـات الصناعيـة، وتـقـديـم الاستشارات الـفـنـيـة الممكنة للمستثمرين في الصناعـة وأصحـاب المشـروعـات الصنـاعـيـة.
۸- تـلـقـي طـلـبـات الترخيص للمشـروعـات الصناعيـة، وتسجيلها وبحثها، وإبداء الرأي بشـأنـها، وإصـدار المـوافـقـات المبدئيـة.
۹- التـوصـيـة بإصدار أو وقف أو إلغاء تـراخـيـص المشروعات الصناعيـة.
١٠- إجراء القـيـد في السـجـل الصناعي، وتجـديـده، وتعـديـلـه، وإصـدار شـهـادات التسجيـل الصناعي، ومتـابـعـة البيانات للمنشآت الصناعيـة المسجلة.
۱۱- التفتيش والـرقـابـة على المشـروعـات والمنشآت الصناعيـة، للتـأكـد مـن التـزامـها بأحكام التـشـريـعـات والأنـظـمـة الصناعيـة.
١٢- التـوصـيـة بمنـح أو إلغـاء أو استرداد المـزايـا والإعـفـاءات المقـررة للمشروعات الصناعيـة المرخصة، وفقاً للتشـريـعـات الـسـاريـة.
۱۳- اسـتـخـدام نظام التصنيف الدولي للصنـاعـة والتـفـريـعـات اللازمة لها، وتحـديـد رقـم التـصـنـيـف لـكـل منـشـأة صنـاعـيـة.
١٤- التنسيـق مـع مـراكـز المعـلـومـات الخـليـجـيـة فـيـمـا يـتـعـلـق بقـواعـد المعـلـومـات الصناعية الخليجية.
25- تنفيذ التـشـريـعـات الخـاصـة بالمنـاطـق الصنـاعـيـة بـالـدولـة.
١٦- إعـداد السيـاسـات الخـاصـة بالمنـاطـق الصناعيـة، ومتـابعـة تـنـفـيـذهـا بعد اعتـمـادهـا.
١٧- إعـداد استـراتـيـجـيـة اخـتـيـار وتـطـويـر المـنـاطـق الصناعيـة، بالتنسيـق مع الجـهـات المخـتـصـة بـالـدولـة.
۱۸- الترويج للمنـاطـق الصناعيـة التـابـعـة للوزارة، وجذب الاستثمـارات الصناعيـة إليـها، بالتنسيق مع الجهات ذات الصلة.
۱۹- تخطيط وتطـويـر واستـغـلال المنـاطـق الصناعيـة، وإعـادة تنـظيـم وتـأهـيـل المنـاطـق الصناعيـة القـائـمـة، والإشراف عليـها.
۲۰- التعاون والتنسيق مع الجـهـات الحـكـومـيـة لـتـوفـيـر الخـدمـات اللازمـة للمشـروعـات الصناعيـة في المنـاطـق الصنـاعيـة.
 

 

المادة 12


تختص إدارة دعـم تـنـافـسـيـة المـنـتـج الـوطـنـي بـمـا يـلـي:
۱- تنفيـذ التـشـريـعـات الخـاصـة بـدعـم تنـافـسيـة المنتـجـات الـوطـنيـة ومـكـافـحـة الممـارسـات الضارة بها في التجـارة الـدوليـة، بالتنسيـق مع الجهات المختصـة بـالـدولـة.
۲- تـوفـيـر الـبـيـانـات والمعـلـومـات في مجـال دعـم تنافسيـة المنتـجـات الوطنيـة أثـنـاء مـراحـل الإنتـاج، ومكافـحـة الممارسات الضـارة بـهـا في التجارة الدوليـة، بالتنسيق مع الجهات المختصة بالـدولـة.
٣- إعداد الدراسات اللازمـة لـرفـع كفـاءة المنتـجـات القـطـريـة وتـعـزيـز قـدرتـها التنافسيـة في الأسـواق الـداخـلـيـة والخـارجـيـة، ورفـع الاقـتـراحـات والـتـوصـيـات فـي هـذا الـشـأن.
٤- التنسيق مع المنظمات الـدوليـة والجـهـات المعـنـيـة في الدول الأخـرى، فيما يتعلـق بمـكـافـحـة الممارسات الضـارة بالمنتـجـات الـوطنيـة في التـجـارة الـدوليـة.
٥- إجـراء الأبحاث والدراسات عن الممارسات الضارة بالمنتجات الوطنيـة، وتـقـديـم المقـتـرحـات بـشـأنـها.
٦- التنسيق مع المنـظـمـات العـربـيـة والإقـلـيـمـيـة والدوليـة، لاتخاذ التدابيـر اللازمـة ضـد الممـارسـات الضـارة بالمنتـجـات الـوطنيـة، بالتنسيـق مع الجهات المختصـة بالـدولـة.
۷- تـلـقـي ودراسـة مـعـوقـات المشـروعـات الصناعيـة المتعلقة بالمنتـجـات القـطـريـة وقـدرتـها التنـافـسـيـة، ورفـع الاقـتـراحـات والـتـوصـيـات في هـذا الـشـأن.
 

 

المادة 13


تختص إدارة تـنـمـيـة الأعـمـال بمـا يـلـي:
۱- تنفيـذ أحـكام التشـريـعـات المنظمـة لـريـادة الأعـمـال والمشـروعـات الصغيرة والمتـوسـطـة.
٢- تنفيذ أحكام التشـريـعـات المنـظـمـة للشراكة بين القطاعين الحكومي والخاص.
۳- دعـم وتـنـمـيـة الشـراكة بين القـطـاعـين الحكومي والخـاص.
٤- اقتراح السياسات والبـرامـج اللازمة لتطـويـر وتنميـة قطـاع الأعمـال، وتشجيع الاستثمار المحلي، وجذب الاستثمار الأجنبي، بالتنسيـق مع الوزارات والإدارات المعنيـة.
٥- إعـداد الدراسات المتعلقة بتطـويـر وتنميـة بيئـة قطـاع الأعمـال والاستثـمـار، والتجـارة الداخـلـيـة، والعـلاقـات التجـاريـة الخارجيـة، بالتنسيق مع الإدارات المعنيـة.
٦- اقتراح آليـات حـديـثـة لـتـطـويـر الإجراءات المؤسسية لقطاع الأعمـال والاستثمار، بالتنسيـق مـع الـوحـدة الإداريـة المعـنيـة.
٧- إعـداد الأدلـة الإرشـاديـة المتعلقة بمجـالات الأعمـال والاستثمـار بالدولـة، والمشاركة في تحـديـث قـاعـدة بـيـانـات الخـدمـات الـعـامـة المـقـدمـة لـقـطـاع الأعـمـال والاسـتـثـمـار.
۸- تعـزيـز ريـادة الأعمـال لبنـاء قـطـاع المنشآت الصغيـرة والمتـوسـطـة، وتقـديـم الـدعـم الفـنـي والإداري اللازم لتشجيع المشروعات الصغيـرة والمتـوسـطـة.
٩- التنسيـق بـيـن الجـهـات الحـكـومـيـة وغيـر الحكوميـة لتنفيـذ آليـة لتحفيز الشباب على الاستثمار في المشـروعـات الصغيرة والمتوسطـة.
 

 

المادة 14


تختص إدارة تـنـمـيـة التبادل التـجـاري وترويج الاستثمار بمـا يـلـي:
١- تنفيذ أحكام التشريعات المنظمة لاستثمار رأس المال غيـر القـطـري في النشاط الاقـتـصـادي.
٢- تنفيذ أحكام التشريعات المنظمـة لـدعـم التجـارة الخارجيـة وتنميـة الصادرات.
۳- اقـتـراح سـيـاسـات تـشـجـيـع ودعـم الأعـمـال المشتركة بين المستثمرين القطريين والمستثمرين غيـر القـطـريـين.
٤- اقـتـراح السيـاسـات والبرامج اللازمة لتطـويـر نـظـم وأسـالـيـب الـتـرويـج والإعـلان عـن فـرص الاستثمـار المتـاحـة بالدولـة، وإعـداد أدلـة إرشـاديـة لـتـعـريـف المستـثـمـريـن بـها.
٥- إعـداد الدراسـات لـدعـم الأنشطة الاستثـمـاريـة بالدولـة، ودراسـات الجـدوى المبدئيـة المتعلقـة بـتـحـديـد الفـرص الاستثماريـة، وتنـويـع مصـادر الـدخـل، والتنميـة المستـدامـة في الدولـة، بالتـنـسـيـق مـع الـوحـدات الإداريـة المعنيـة.
٦- الترويج للمنـاطـق الاقـتـصـاديـة والمنـاطـق الحـرة في الدولة، بما يحقق استراتيجيـة هـذه المنـاطـق وأهـداف استقطاب الاستثمار الخارجي، ويـدعـم الخـطـة الاقـتـصـاديـة الـوطـنيـة.
٧- التـرويـج للمشـروعـات والأعمـال اللازمـة للاستثمار في مـجـالات استغلال وتصنيـع ونـقـل وتـسـويـق المنتـجـات.
۸- اتخاذ الإجـراءات اللازمة لتشجيع الاستثمـار وتيسيـر الإجراءات للمستثمرين، بالتنسيق مع الجـهـات المختصة.
٩- المشاركة في المنتـديـات والمعـارض والندوات والمؤتمرات التي تهـدف إلى التعـريـف بالسيـاسـة الاستثماريـة للدولـة، وإعـداد الدراسـات اللازمة لذلك.
۱۰- دراسـة وتحـلـيـل مـشـروعـات الاتفـاقيـات التجـاريـة والصناعيـة، بالتنسيق مع الوحدات الإداريـة المعـنيـة.
١١- إعـداد قـاعـدة البيـانـات والمعلـومـات وأدلـة التعـريـف المتعلقة بـفـرص الاسـتـثـمـار المتـاحـة بـالـدولـة، والأعـمـال المـشـتـركة.
12- اقـتـراح سيـاسـات واستراتيجيات وخطط وبـرامـج تـنـميـة الصـادرات والـتـبـادل الـتـجـاري.
۱۳- اقـتـراح تعيين الملحقين التجاريين، والإشراف الفني على أعـمالـهـم.
١٤- مـتـابـعـة أعـمـال المـلـحـقـين الـتـجـاريـين ودورهـم فـي تـعـزيـز فـرص التعـاون وتنميـة التبادل التجـاري بين الدولـة والـدول الأخـرى، وإمـدادهـم بالمعـلـومـات اللازمـة في هذا الشـأن.
15- مـتـابـعـة تـنـفـيـذ بـرامـج تـنـميـة الصـادرات والتبادل التجـاري، وتعـزيـز القدرة التنـافـسـيـة للمـنـتـجـات القـطـريـة في الأسواق الخارجيـة.
١٦- دراسـة الأسـواق المحليـة والخارجيـة، واقـتـراح مـا يـلـزم لتنشيط الصادرات للأسـواق الخـارجـيـة.
۱۷- التنسيق والتعاون مع المؤسسات الدوليـة المعـنيـة بتنميـة الصـادرات.
۱۸- تـوفـيــر قـاعـدة البيانات والمعلـومـات والخـدمـات المسـانـدة لعمليـات التـصـديـر، وإجـراء عمليات المسح الخارجي للأسـواق بغـرض الـنـفـاذ إليها، والترويج للسلع والخـدمـات القـطـريـة فـيـها.
١٩- إعـداد الدراسات اللازمـة لأعمـال الفـريـق التفـاوضي للدولـة، ودراسـات حـيـال المفاوضـات التجـاريـة والصناعيـة، بالتنسيق مـع الوحدات الإداريـة المعنيـة.
۲۰- اقـتـراح بـرامـج التـدريـب والـتـأهـيـل للـكـوادر القطـريـة العـامـلـة في مجـال التـصـديـر والتسـويـق الدولي، ومتـابعـة تـنـفـيـذهـا بالتنسيـق مع الجهات المعنيـة.
 

 

المادة 15


تختص إدارة الـتـمـويـن والمـخـزون الاستـراتـيـجـي بمـا يـلـي:
١- وضع الخطـة اللازمة لتـوفـيـر المـواد التموينيـة والأعـلاف، والمحـافـظـة على مخزون استراتيجي منها.
٢- إعـداد ومتـابـعـة طـرح المنـاقـصـات والممارسات الخاصـة بـتـوفـيـر المـواد التـمـويـنـيـة والأعـلاف.
٣- إعـداد ومـسـك سـجـلات تـوريـد المـواد الـتـمـويـنـيـة المتـعـاقـد عليـها.
٤- مـتـابـعـة تـوريـد المـواد التمـويـنـيـة والأعـلاف المطـلـوبـة.
٥- تسـلم العينـات المقـدمـة مـن المتناقصين في المنـاقـصـات، وإرسالهـا للتحليل المخبري، والتـأكـد مـن مـطـابقتها للشروط والمـواصـفـات.
٦- فتـح الاعتمـادات المستنـديـة للمـوردين، بالتنسيـق مـع الـوحـدة الإدارية المعـنيـة بالوزارة والبنوك المخـتـصـة.
۷- تـوفـيـر وشـراء مـواد المسـاعـدات والإغاثـة الـتـي تقـدمـها الدولـة للدول الشقيـقـة والصديقـة، وفقاً للتعـليمـات الصادرة في هـذا الشـأن، ومتـابـعـة تـسـلم وتخـزيـن هـذه المـواد وإرسالها للجـهـات المختصة، وذلك بالتنسيق مع الجـهـات المختصة في الـدولـة.
۸- تسـلـم الشـحـنـات الـواردة مـن المـواد التمـويـنـيـة والأعـلاف، وأخـذ عينـات عشـوائـيـة منهـا، وإرسالها إلى الـوحـدة الإداريـة المعـنيـة للتـأكـد مـن مـطـابـقـتـهـا للشروط والمواصـفـات المـطـلـوبـة.
۹- تـخـزيـن المـواد التمـويـنـيـة والأعـلاف ومـواد المسـاعـدات والإغاثـة بطرق التخزين المتـعـارف عليها.
۱۰- مـسـك دفـتـر لـكـل صـنـف، وتسجيـل وارد والمنصـرف مـنـه والـرصـيـد المتبقي.
١١- إصـدار التراخيص الخـاصـة بتشغيـل وإدارة المخـابـز الـتـي تحـصـل على دعـم مـن الدولة، ومـراقبـة آليـة صـرف أذونات الطـحـين.
12- صـرف الكميـات المـطـلـوبـة مـن المـواد التموينيـة والأعـلاف لمـراكـز التـوزيـع والجـهـات الحكوميـة والمـواطـنـين بمـوجـب أذونـات تـسـلـيـم صـادرة عـن الوحـدة الإداريـة المعـنيـة.
١٣- إصدار أذونات التسليم الـتـي يـتـم بمـوجـبـهـا صـرف المـواد التموينيـة والأعـلاف مـن المخـزن، وذلك في ضـوء الأرصـدة المتـاحـة بالمخـزن مـن كل صنف.
١٤- صـرف أذونـات الـبـيـع للمـواطـنـين.
15- مـراقـبـة مـدى تـقـيـد المتعاملين في السلع التمـويـنـيـة المـدعـومـة بالتـزامـاتـهم، واتخاذ الإجراءات والتدابير اللازمـة وفـق الـقـانـون.
١٦- اقـتـراح تـطـويـر الأدوات التـشـريـعـيـة بمـا يـدعـم تـحـقـيـق الأمـن الغـذائي، بالتنسيق مع الجهات المخـتـصـة بالـدولـة.
۱۷- تنفيـذ الـتـشـريـعـات الخـاصـة بتنظيم وإدارة المخزون الاستراتيجي للسلع الغذائيـة والاستهلاكيـة.
۱۸- تقـديـم التـوصـيـات الخـاصـة لمـواجـهـة التـقـلـبـات غـيـر المتـوقـعـة في السـوق، مثـل الهبوط المفاجئ في الإنتاج المحلي والعالمي والكوارث الطبيعيـة والمتغيرات الجـيـوسـيـاسـيـة.
١٩- التنسيق مع الجهات المخـتـصـة بالدولـة في مجـال تـأمين الاحتياجات الضروريـة مـن السلع الغذائيـة والاستهلاكيـة.
۲۰- وضع نظام إلكتروني لإدارة المخزون الاستراتيجي وضمـان الاحتـفـاظ الدائـم بمـخـزون الأمـان.
۲۱- تـأمــــين واستعـمـــال المخـــزون الاستراتيجي للسلع الغذائيـة والاستهلاكيـة عنـد الحـاجـة بـكفـاءة عـاليـة.
 

 

المادة 16


تختص إدارة التـراخـيـص الـنـوعـيـة ومـراقـبـة الأسـواق بما يلي:
١- تنفيذ أحـكـام التـشـريـعـات المنـظـمـة لمـراقـبـة الأسـواق.
۲- القيام بالتفتيش على المصـانـع والمحـال والمخازن وغيـرهـا، وفـحـص الدفـاتـر، للتـأكـد مـن التـزامـهـا بتنفيذ أحكام الـقـانـون.
٣- إصـدار تـراخـيـص التنزيلات، ومـراقبـة المحـال التي تجري تخـفـيـضـات عـامـة للأسعار، وفـحـص مستنـداتـهـا وفـواتيـرهـا للتـأكـد مـن جـديـتـها في ذلك.
٤- إصـدار التراخيص اللازمة للتـرويـج عـن بـيـع السلع والخـدمـات.
٥- الإشراف على سـحـب الجـوائـز وتـوزيـع الـهـدايـا.
٦- إصـدار الـتـراخـيـص اللازمة لأعمال الدلالة، ومـراقبـة أعـمـال القائمين عليـها.
٧- الإشراف على أعـمـال المـزادات.
۸- تلقي الشكاوى مـن المستهلكين ضمن اخـتـصـاص الإدارة، والتحـقـق منـها، ومتـابعـة اتخاذ الإجراءات المنـاسبـة بشـأنـها.
۹- دراسة وتقييم أسعار السلع والخـدمـات المتداولة بالأسـواق.
۱۰- دراسـة طـلـبـات زيادة أسعار السلع والخـدمـات المقـدمـة مـن المـزوديـن.
١١- إعـداد الدراسـات الخـاصـة بأسعار السلع والمـواد والخـدمـات، بهدف تحـلـيـل أسباب ارتفـاعـها، واقتراح وسـائـل تصحيحـها، وآليـات مـنـع الرفع غير المبرر للأسعار، وتقـديـم التقارير اللازمة في هذا الصـدد.
 

 

المادة 17

۹- دراسة وتقييم أسعار السلع والخـدمـات المتداولة بالأسـواق.
۱۰- دراسـة طـلـبـات زيادة أسعار السلع والخـدمـات المقـدمـة مـن المـزوديـن.
١١- إعـداد الدراسـات الخـاصـة بأسعار السلع والمـواد والخـدمـات، بهدف تحـلـيـل أسباب ارتفـاعـها، واقتراح وسـائـل تصحيحـها، وآليـات مـنـع الرفع غير المبرر للأسعار، وتقـديـم التقارير اللازمة في هذا الصـدد.
 

 

 

المادة 18


تختص إدارة حـمـايـة المنـافـســة بمـا يـلـي:
١- تنفيذ القوانين واللوائح الخـاصـة بـحـمـايـة المنافسـة ومنع الممارسات الاحـتـكـاريـة.
۲- تـوفـيـر البيانات والمعـلـومـات عـن النشاط الاقـتـصـادي والتـجـاري.
۳- نـشـر ثـقـافـة المنافسـة، والتنـسـيـق مـع الجهات المعنيـة فـي الـدول الأخرى فيما يتعلق بحـمـايـة المنافسة ومنع الممارسات الاحـتـكـاريـة.
٤- الإشراف على فحص الإخـطـارات المتعلقة بنقـل حـقـوق الملكيـة وتمـلك الأصـول وإقـامـة الاتحـادات والانـدمـاجـات، وتقييـم آثـارهـا علـى المنـافـسـة، واقـتـراح الإجراءات الخاصـة بـها.
٥- إجـراء الأبحـاث والتحـريـات عـن الممـارسـات المخـلـة بالمنـافـسـة، والتـأكـد مـن وجـودهـا، وتقـديـم المقـتـرحـات بـشـأنـهـا.
٦- إعـداد الـدراسـات عـن سـيـر الأسـواق والقـطـاعـات الاقتصـاديـة بالدولة، وتقـديـم المقترحات لتعـزيـز المنافسـة والحد من الممارسات الاحتـكـاريـة، بالتنسيق مع الجهات المعنيـة
 

 

المادة 19


تختص إدارة العـلاقـات الـعـامـة والاتصـال بـمـا يـلـي:
١- إصـدار النشرات والبـرامـج الإعـلاميـة الـتـي تهـدف إلى التعـريـف بدور الوزارة وأنشـطـتـها واختصاصاتـها، بالتنسيـق مـع الـوحـدات الإداريـة المعنيـة.
۲- رصـد ومـتـابـعـة مـا يُنشـر في الصـحـف ووسـائـل الإعـلام ومـواقـع التـواصـل الاجتماعي مـن مـوضـوعـات تتـعـلـق بـالـوزارة واخـتـصـاصـاتـها، وعـرضـها على المسؤولين، وإعـداد الـردود المنـاسبـة، بالتنسيـق مـع الـوحـدات الإداريـة المعـنـيـة والجـهـات المخـتـصـة.
٣- التنسيق مع الوحدات الإدارية المختصة بشأن إعـداد خطط وبـرامـج التوعيـة والتثقيـف المتعلقة بأنشـطة واخـتـصـاصـات الـوزارة، وتطبيقها مـن خـلال الفعاليـات والأنشـطة المخـتـلـفـة، بالتنسيـق مـع الجهات ذات العـلاقـة.
٤- تـلـقـي طـلـبـات وشكاوى المـراجـعـين، وإحالتها إلى الوحدات الإداريـة المعنية، والرد على استفساراتهم.
٥- القيام بترتيبات السـفـر والضيـافـة والإقـامـة لضيـوف الـوزارة وللـوفـود الـزائـرة والمغادرة، بالتنسيـق مـع الوحدات الإدارية المعنيـة.
٦- تنظيـم الأنشطـة الثقـافيـة والاجتماعيـة والرياضيـة لموظفي الوزارة، والإشراف على الاحتفالات التي تقيمها أو تشارك فيـهـا الـوزارة.
۷- تنظيـم المؤتمرات والندوات والمعـارض التي تقيمهـا الـوزارة، وإعـداد الموازنة اللازمـة لذلك، بالتنسيق مع الوحدات الإداريـة المعنيـة.
۸- اسـتـقـبـال طلبـات كبار السن والأشخاص ذوي الإعـاقـة، بالتنسيـق مـع الوحدات الإدارية المختصة، لإنجاز معاملاتهم وتذليـل أي مشكلات أو صـعـوبـات قـد تـواجـهـهـم

 

المادة 20


تختص إدارة الموارد البـشـريـة بما يلي:
١- تنفيذ القـوانين واللوائح والنـظـم المتعلقة بـشـؤون الموظفين.
٢- إعداد الدراسات الخاصة بتخطيط الموارد البشريـة بالـوزارة، وتحـديـد احتياجـات الوزارة مـن الـوظـائـف والموظفين، بالتنسيـق مـع الجـهـة المختصة والوحـدات الإداريـة المختـلفـة.
٣- إعداد مشروع وصـف وتصنيـف وترتيب الوظائف، ومتـابعـة تنفيذه، وتطـويـره.
٤- إعداد مشروع موازنـة الباب الأول، بالتنسـيـق مـع الوحـدات الإداريـة المعنية.
٥- تنفيذ إجـراءات تقييـم الأداء للمـوظـفـين.
٦- القيـام بإجـراءات التـعـيـين والنـقـل والنـدب والإعـارة.
۷- دراسـة اسـتـحـقـاق الموظفين للإجـازات، وفقاً للـقـانـون.
٨- إعـداد الدراسـات الخـاصـة بالهيكل الوظيفي للـوزارة، وتقـديـم المقتـرحـات اللازمة لتطـويـر التنظيـم الإداري بـها.
۹- تحـديـد الاحتياجـات التـدريبيـة لمـوظـفـي الـوزارة، بالتنسـيـق مـع الوحـدات الإداريـة المختلفة، ومتـابعـة تـنـفـيـذهـا، وتقييـم مـدى الاستـفـادة مـنـها.
۱۰- اتخاذ الإجراءات اللازمة لتخصيص السـكـن الحـكـومي وصــرف بدل الأثـاث لموظفي الوزارة، بالتنسيـق مـع الجـهـات المخـتـصـة.
۱۱- وضـع وتحـديـث قـواعـد البـيـانـات الخـاصـة بشـؤون موظفي الوزارة
 

 

المادة 21


تختص إدارة الـشـؤون المـالـيـة والإداريـة بمـا يـلـي:
۱- تنفيـذ القـوانـين واللـوائـح والنـظـم المـاليـة والإداريـة المتعلقة بعمـل الوزارة.
٢- إعـداد مشـروع الموازنـة السـنـويـة للـوزارة، وحسـابـها الخـتـامي، بالتنسـيـق مـع الوحدات الإداريـة المعنيـة.
۳- توفير احتياجات الوزارة ووحـداتـها الإدارية المختـلفـة من المستلزمات والأجـهـزة اللازمة لأداء مهـامـها، بالتنسيق مع الوحدات الإداريـة المعنية.
٤- القيام بأعمـال المشتـريـات والمنـاقـصـات والمزايدات، وفـق الأنظمـة والقـواعـد المعمـول بـها في الـدولـة.
٥- الإشراف على تنفيـذ الاعتمـادات، وتـدقـيـق حـسـابـات الإيرادات والمصـروفـات.
٦- تـسـلـم وتســليـم وتصنيف وقـيـد البـريـد الـوارد والصادر.
۷- تنـظيـم أرشـيـف الوزارة، وحـفـظ الـوثـائـق، وفقاً لأحـدث الـطـرق.
٨- إعـداد مسـتـنـدات الصرف، وسـائـر المـعـامـلات الماليـة الأخـرى.
٩- تحـصيـل الـرسـوم ومـقـابـل الخـدمـات التي تؤديـها الوزارة.
۱۰- القيام بجـمـيـع أعـمـال الخـدمـات الإداريـة.
١١- الإشـراف على مخازن الوزارة
 

 

المادة 22


تختص إدارة نظم المعـلـومـات بمـا يـلـي:
١- إعـداد خطط وسياسات اسـتـخـدام الحاسب الآلي في أنشـطـة الـوزارة، ومتـابـعـة تنفيـذهـا.
۲- برمجـة وحـفـظ واستـرجـاع وتـطـويـر نـظم المعلـومـات والبيـانـات اللازمـة لأنشـطـة الـوزارة.
٣- تـأمـين الأنظمـة والبرمجيـات والشبكات والبنية التحتيـة لنـظم المعـلـومـات مـن مـخـاطـر الحـوادث الإلكتـرونيـة.
٤- إعـداد خطط التعـافـي مـن آثـار الحـوادث الإلكتـرونيـة حـال وقـوعـها أو تقليـل حـدوثها أو تصعيـدهـا، وفـق الـسيـاسـات والمبادئ التـوجــيـهـيـة لأمـن المعـلـومـات، بالتنسيق مع الجهـة المختصة.
٥- تـوفـيـر وصـيـانـة الأجـهـــزة والبـرامـج والشبكات الإلـكـتـرونيـة اللازمة لنـظـم العـمـل الآليـة بالـوزارة، بالتنسـيـق مـع الوحـدات الإدارية المعنيـة.
٦- تصـمـيـم وتشـغـيـل وإدارة قـواعـد البـيـانـات والمعلـومـات بـأنـواعـها المختلفة.
۷- تقـديـم الدعم الفني اللازم لمسـتـخـدمي الشبكات الإلكتـرونيـة بالـوزارة، وتـدريـبـهـم علـى استخـدام الأنظمـة والبـرامـج وأجـهـــزة الحـاســب الآلي وملحـقـاتـها.
٨- إنشاء ومتـابعـة وتحديث موقع الوزارة على شبكة المعلـومـات الـدوليـة
 

 

المادة 23


يصدر بتحـديـد اختصـاصـات كـل مـن مـكـتـب الوزير، ومكتـب وكيـل الوزارة، ومكتـب وكيـل الـوزارة المسـاعـد، قـرار مـن الـوزيـر

 

المادة 24


يجـوز بقـرار مـن مـجـلـس الوزراء، بناءً على اقتراح الـوزيـر، تعـديـل تنظيـم الوحدات الإداريـة التي تتـألـف منهـا الـوزارة بالإضـافـة أو الحـذف أو الدمـج، وتعيين اختصـاصـاتـها وتعـديـلـها

 

المادة 25


يجـوز بقـرار مـن الوزير إنشاء ما تقتضيه المصلحـة العـامـة مـن أقـسـام في الوحدات الإداريـة الـتـي تتـألـف منهـا الـوزارة، أو إلغاؤهـا أو دمجها، وتعيين اختصـاصـاتـهـا وتـعـديـلها، ولا يـكـون قـرار الوزيـر نـافـذاً إلا بعـد اعتماده مـن مـجلـس الـوزراء

 

المادة 26


يُلـغـى القـرار الأمـيـري رقـم (12) لسنـة ٢٠١٩، وقـرار رئيـس مجلس الوزراء رقم (23) لسنة 2016، المشار إليـهـما.
