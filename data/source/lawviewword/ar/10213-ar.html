قـرار مجلـس الـوزراء رقـم (36) لسنـة 2025 بنـزع ملكيـة بعـض العقـارات للمنفعـة العامـة 36 / 2025
عدد المواد: 3
فهرس الموضوعات

المواد (1-3)



مجلس الوزراء ،
بـعـد الاطـلاع عـلـى الـدسـتـور،
وعـلى الـقـانـون رقـم (8) لـسنـة 2022 بـشـأن نـزع مـلـكـيـة الـعقـارات والاستـيلاء عـلـيـهـا مـؤقـتـاً للـمـنـفـعـة الـعـامـة ،
وعـلـى الـقـرار الأمـيـري رقـم (29) لـسنـة 1996 بـشـأن قـرارات مـجـلـس الوزراء الـتـي تـرفـع لـلأمـيـر لـلـتـصـديـق عـلـيـهـا وإصـدارهـا ، 
وعـلـى قـرار وزيـر الـبـلـديـة رقم (110) لـسنـة 2025 بـاعـتـبـار اسـتـمـلاك عـدد (11) عـقـاراً كـائـنـاً بمـنـطـقـة (أم عـبـيـريـة ) مـن أعـمـال المـنـفـعـة الـعـامـة ،
وعـلـى اقـتـراح وزيـر الـبـلـديـة ،
قـرر مـا يـلـي :
 

المواد

المادة 1


تُـنـزع لـلـمـنـفـعة الـعـامـة مـلـكـيـة الـعـقـارات الـكـائـنـة بمـنـطـقـة (أم عبـيـريـة) والمـبـينـة فـي قـرار وزيـر الـبـلـديـة رقـم (110) لـسنـة 2025 المـشـار إلـيـه.

 

المادة 2


تُـتـخـذ الإجـراءات المـنـصـوص عـلـيهـا فـي الـقـانـون رقم (8) لـسنـة 2022 المـشـار إلـيـه ، لـتـعـويـض مـلاك الـعـقـارات المـنـزوعـة مـلـكيـتـهـا بمـوجـب هـذا الـقـرار.

 

المادة 3


عـلـى جـمـيـع الجـهـات المـخـتـصـة ، كـل فـيـمـا يـخـصـه ، تـنـفـيـذ هـذا الـقرار. ويُـعمـل بـه مـن تـاريـخ نـشـره فـي الجـريـدة الـرسـمـيـة.
