قرار أميري رقم (36) لسنة 2022 بالهيكل التنظيمي لوزارة الصحة العامة 36 / 2022
عدد المواد: 32
فهرس الموضوعات

المواد (1-32)



نحن تميم بن حمد آل ثاني أمير دولة قطر،
بعد الاطلاع على الدستور،
وعلى قانون الموارد البشرية المدنية الصادر بالقانون رقم (15) لسنة 2016، المعدل بالقانون رقم (23) لسنة 2022،
وعلـى القـرار الأميـــــري رقـم (14) لسنـة 2019 بالهيكـل التنظيمـي لوزارة الصحـة العـامـة،
وعلـى القـرار الأميري رقـم (57) لسنـة 2021 بتعييـن اخـتـصـاصـات الوزارات،
وعلى اقتراح مجلس الوزراء،
قررنا ما يلي:

المواد

المادة 1


يتكون الهيكل التنظيمي لوزارة الصـحـة الـعـامـة مـن الـوحـدات الإداريـة المبينـة بالخريطة التنظيميـة المـرفـقـة بهذا القـرار، وهي:
أولاً: الوحدات الإداريـة التـابـعـة للـوزيـر:
۱- مـكـتـب الـوزيـر.
۲- المـكـتـب الفـنــي.
٣- إدارة التـدقـيـق الـداخـلـي.
٤- إدارة التخطيط الاستراتيجي والأداء والابتكار.
٥- إدارة الشـؤون الـقـانـونيـة.
 
٦- إدارة الـعـلاقـات الـعـامـة والاتصـال.
٧- إدارة التـمـويـل والـتـأمين الصـحـي.
ثانياً: الوحدات الإداريـة التـابـعـة لـوكـيـل الوزارة:
۱- مكتب وكيـل الـوزارة.
٢- إدارة العـلاقـات الصـحـيـة الـدوليـة.
٣- إدارة العـلاقـات الطـبيـة والعـلاج بالخـارج.
٤- إدارة الصـحـة الإلكتـرونيـة.
٥- مـركـز المعـلـومـات الصـحـيـة.
ثالثـاً: الوحـدات الإداريـة التـابـعـة لوكيـل الـوزارة المسـاعـد للشؤون الصـحـيـة:
۱- مكتـب وكيـل الـوزارة المسـاعـد.
٢- إدارة حـمـايـة الصـحـة ومـكـافـحـة الأمـراض الانـتـقـالـيـة.
٣- إدارة تـعـزيـز الصـحـة.
٤- إدارة بـرامـج الوقـايـة مـن الأمـراض غيـر الانـتـقـاليـة.
٥- إدارة القـومــسيـون الطبي.
٦- إدارة سـلامـة الـغـذاء.
٧- إدارة الـطـوارئ الصحيـة.
رابعاً: الوحـدات الإداريـة التـابعـة لوكيـل الـوزارة المسـاعـد لشؤون تنـظـيـم الـرعـايـة الصـحـيـة:
۱- مكتب وكيـل الـوزارة المسـاعـد.
٢- إدارة التـخـصـصـات الصـحـيـة.
٣- إدارة المـنـشـآت الصحيـة.
٤- إدارة جـودة الـرعـايـة الصـحـيـة.
٥- إدارة الصـيـدلـة والـرقـابـة الـدوائـيـة.
٦- إدارة تـنـظـيـم البـحـوث الصـحـيـة.
خامساً: الوحدات الإداريـة التابعة لوكيل الوزارة المسـاعـد لشؤون الخدمات المشـتـركـة:
۱- مكتـب وكيـل الـوزارة المسـاعـد.
٢- إدارة الموارد البـشـريـة.
٣- إدارة الشـؤون المـاليـة.
٤- إدارة المـشـتـريـات والمـنـاقـصـات.
٥- إدارة المـرافـق والخـدمـات الـعـامـة.
٦- إدارة نـظـم المـعـلـومـات.
 

 

المادة 2


يختـص المكتـب الفـنـي بمـا يـلـي:
١- إبداء الرأي الفـني في المـوضـوعـات التي تُحـال إليـه مـن الـوزيـر.
۲- دراسـة وتقييم التقاريـر والمشـروعـات والبـرامـج المتعلقة بأنشـطـة الوزارة، التي تُحال إليـه مـن الـوزيـر.
۳- بحث ودراسة ومتـابـعـة المـوضـوعـات التـي تحـال إليـه مـن الـوزيـر، بالتنسيق مع الوحدات الإداريـة المعنيـة.
٤- إعـداد ملخص للمـواضـيـع والـتـقـاريـر والـدراسـات الـتـي يـلـزم عـرضـها على الـوزيـر، وإبـداء الـرأي بشـأنـها.
٥- إبـداء الـرأي في التظلمات والشكاوى التي تُحـال إليـه مـن الـوزيـر.
 

 

المادة 3


تختص إدارة التـدقيـق الـداخـلـي بمـا يـلـي:
١- إعداد مشروع خـطـة التـدقـيـق السنـويـة المبنيـة علـى المخـاطـر للوزارة، ورفعـها للوزيـر لاعتمـادهـا، وإعـداد تقاريـر بـنـتـائـج التـدقـيـق.
۲- مـراقبـة مـدى الالتزام بتنفيـذ القـوانـين واللوائح والقرارات المنظمـة للعمـل.
۳- مـراجعـة التعليمات الماليـة، واقـتـراح مـا يـلـزم لـهـا مـن تـعـديـلات، ومـراقبـة تنفيـذهـا.
٤- مـراقبـة الأعمال الإداريـة وإجـراءات تعيين الموظفين، وإجازاتـهم، وتـرقـيـاتـهم، وغير ذلـك     ممـا يـتـعـلـق بـشـؤون الموظفين، والتحقـق مـن الالتزام بالقوانين واللوائح والقرارات المنظمة لذلك.
٥- مـراقبـة الأعمـال المـاليـة والمحاسبيـة، وسنـدات الصرف، وإجـراءات القيـد في السجـلات المحاسبيـة، والتـأكـد مـن تطبيـق القـواعـد المعتمدة في هـذا الشـأن.
٦- تقييـم المخـاطـر المتعلقة بالتدقيـق المالي والإداري، واقتراح إجـراءات الحـد منـها.
۷- متـابعـة الإجـراءات التصحيحيـة المطـلـوبـة مـن الوحـدات الإداريـة المعـنيـة، وفـق نتائج التدقيـق، وإعداد التقارير الدوريـة بشـأنـها.
۸- التـأكـد مـن سـلامـة الأمـوال والعهـد والأثاث والأدوات والمبـاني والمـوجـودات الأخرى التي تملكها الوزارة أو تخضع لإشـرافـها.
۹- الاشتـراك في أعمـال الجـرد الـدوري والمفاجئ علـى المخـازن والعهـد والتـأكـد مـن دقـة التسجيـل بـالـدفـاتـر وسـلامـة الإجـراءات المطبقـة.
۱۰- متـابعـة نتائج تقارير الجهـات الـرقـابيـة، وإبـداء الـتـوصـيـات المنـاسبــة بشأنها.
 

 

المادة 4


تختص إدارة التخطيط الاستراتيجي والأداء والابتكار بمـا يـلـي:
١- إعـداد الخطة الاستراتيجيـة للـوزارة، بالتنسيق مع الجهات المختصة والوحـدات الإدارية المعنيـة، واتخاذ إجـراءات اعتمـادهـا.
٢- إعداد الخطـة الاستراتيجيـة لـقـطـاع الـرعـايـة الصـحيـة، بالتنسـيـق مع الجهات المختصة والوحدات الإداريـة المعـنيـة، واتخاذ إجـراءات اعتمـادهـا.
۳- حـصـر الـتـحـديـات القـائمـة، وتحـديـد وتقييم المخاطـر المؤسسيـة المتـوقـع حـدوثـها، والتي قد تؤثر على تحقيق أهـداف الـوزارة، وتقديم المقترحات بشأنها، بالتنسيق مع الوحدات الإدارية المعنيـة.
٤- إدارة وتجهيـز خـطـط بـديـلـة تضمن استمراريـة تشغيـل الأعمـال بالــــوزارة في حـالات الطوارئ والأزمـات، بالتنسيق مـع الـوحـدات الإدارية المختـلفـة كـل فـيـمـا يـخـصـها.
٥- رفـع مـشـروعـات الخطط التنفيـذيـة والمشـاريـع والبـرامـج الخـاصـة بالـوحـدات الإدارية المختـلفـة، للاعتمـاد مـن قبـل الـوزيـر
٦- متـابعـة تنفيذ الخطط المتعلقة بأنشـطـة الـوزارة، وتقييـم نـتـائـجـها، ورفـع تـقـاريـر دوريـة بشـأنـها.
۷- مـتـابـعـة وتـقـديـم الـدعـم لـتـنـفـيـذ البـرامـج والمشـاريـع الاستراتيـجـيـة، بالتنسيق مع الجهات ذات الصلة والوحدات الإدارية المعنيـة، لضمـان تحـقـيـق الأهـداف الاستراتيـجـيـة التنمـويـة.
٨- دراسة المشـاكـل والمعـوقـات الـتـي تـواجـه تنفيـذ الخطـة الاستراتيجيـة والخطط التنفيـذيـة، واقتراح الحلول المناسبة لها.
۹- تقييـم جـودة الخـدمـات المقـدمـة مـن الـوحـدات الإداريـة، بالتنسيـق مع الجهـة المختـصـة.
۱۰- مـراجـعـة وتقييـم جـودة الأداء علـى مـسـتـوى الوحدات الإداريـة، ووضـع مـقـتـرحـات لـرفـع الـكفـاءة العـامـة بـها.
١١- التحـقـق مـن انسجام نـظـم وأساليـب العـمـل مـع اختصـاصـات وأهـداف الوزارة، واقتـراح خطـط تـطـويـر تـلـك النـظـم والأساليـب، وتبسيط الإجراءات، بالتنسيق مع الوحدات الإداريـة المعنية والجهة المختصة.
۱۲- دراسـة مشـاكل ومعـوقـات العـمـل بـالـوزارة، وبحـث أسبـابـها، واقـتـراح الحلول المنـاسبـة لـها.
۱۳- تلقـي المبـــــادرات والمقتـرحـات والأفكار الإبـداعـيـة، ودراستها بالتنسيق مع الوحدات الإداريـة المعنيـة والجـهـة المختصة.
١٤- التنسيـق مـع الوحدات الإداريـة المعنية والجهـة المختصة فيما يتعلـق بالمؤشرات الـدوليـة المتعلقة بأنشـطـة واخـتـصـاصـات الـوزارة.
١٥- إعـداد خطط لتنمية الاستثمار الصحي وتشجيع الشراكة بين مقدمي خـدمـات الـرعـايـة الصـحـيـة في القطاعين العـام والخـاص مـن خـلال الخـدمـات الصـحيـة، بالتنسيـق مـع الجـهـات المختصة والوحدات الإداريـة المعـنيـة.
١٦- اقـتـراح سيـاسـات وطـنـيـة لـتـطـويـر الخـدمـات الصـحيـة والطبيـة، مع التـركـيـز عـلـى الأمـراض ذات الأولـويـة، بالتنسيـق مـع الـوحـدات الإداريـة المعـنـيـة والجهات المختصة.
۱۷- تـطـويـر نمـاذج مبتكرة للـرعـايـة الصـحيـة مـبـنـيـة علـى مـخـطـطـات خـدمـات الـرعـايـة الصـحيـة ونسبـة الحـاجـة إليـهـا وقـدرة القـطـاع الصحي علـى تـوفـيـرهـا.
۱۸- العـمـل على تـأهـيـل الكوادر الوطنيـة لإدارة المشـروعـات الصـحـيـة.
١٩- إعـداد التـقـاريـر الشـامـلـة لـدعـم عمـليـات وضع السيـاسـات الصـحيـة، بالتنسيق مع الوحـدات الإداريـة المعـنيـة.
۲۰- مـتـابـعـة الـتـنـفـيـذ بـتـفـعـيـل آليـة المـراقبـة والتقييـم والمتـابعـة المسـتـمـرة لأداء الوحدات الإداريـة بالوزارة ومـقـدمـي خـدمـات الرعاية الصحيـة.
۲۱- تحـلـيـل الـتـقـاريـر والشكاوى المتعلقة بأنشـطـة مـقـدمـي خـدمـات الرعـايـة الصـحـيـة للتـأكـد من جودة الأداء، واستـخـلاص النتائج والمؤشرات اللازمـة لـدعـم وتـوجـيـه عـمـلـيـة التخطيط وإعـادة التـخـطـيـط لتلك الأنشـطـة، وتـقـديـم الـتـوصـيـات بـشـأنـها.
 

 

المادة 5


تختص إدارة الشؤون القـانـونيـة بمـا يـلـي:
۱- بحـث ودراسـة ومـتـابـعـة المسـائل الـقـانـونيـة الخاصـة بنشـاط الـوزارة، بالتنسيق مع الوحـدات الإداريـة المعنيـة.
٢- إعـداد مشـروعـات الأدوات التشريعيـة الخـاصـة بالـوزارة، وإبـداء الرأي في مشـروعـات القـوانـين التي تُحـال إليـها.
٣- إبـداء الـرأي الـقـانـوني في المـوضـوعـات التي تُحـال إليـها.
٤- إعـداد مشـروعـات العقـود والاتفاقيـات ومـذكـرات التفـاهـم المتعلقة باخـتـصـاصـات الوزارة، بالتنسيق مع الوحـدات الإداريـة المعنيـة.
٥- التحقيـق في الوقـائـع والمخالفــات المنسوبـة لموظفي الـوزارة، وإعــداد المذكــرات اللازمـة بنـتـائج التحقيـق مـشـفـوعـة بالـرأي الـقـانـوني والتـوصـيـات، وعرضها على السلطة المخـتـصـة، ومتـابعـة تنفيذ ما يتخذ مـن قـرارات بشـأنـها.
٦- متـابـعـة المنـازعـات والقـضـايـا الـتـي تـكـون الوزارة طـرفـاً فيـهـا، بالتنسيـق مـع الجـهـات المختـصـة.
 

 

المادة 6


تختص إدارة العـلاقـات الـعـامـة والاتـصـال بمـا يـلـي:
١- إصدار النشرات والبـرامـج الإعـلاميـة التي تهدف إلى التعـريـف بـدور الوزارة وأنشطتها واخـتـصـاصـاتـها، بالتنسيـق مـع الوحدات الإداريـة المعنيـة.
۲- رصـد ومـتـابـعـة مـا ينشـر في الصـحـف ووسـائـل الإعـلام ومـواقـع التـواصـل الاجـتـمـاعـي مـن مـوضـوعـات تتعلق بالوزارة واخـتـصـاصـاتـها، وعـرضـها على المسؤولين، وإعـداد الـردود المنـاسبـة، بالتنسيـق مـع الـوحـدات الإداريـة المعـنـيـة والجهات المختصة.
٣- التنسـيـق مـع الوحدات الإداريـة المختصـة بشـأن إعـداد خطط وبـرامـج التـوعـيـة والتثقيـف، وتطبيقهـا مـن خـلال الفعاليـات والأنشـطـة المختـلفـة بالتنسـيـق مـع الجـهـات المعنيـة.
٤- تلقـي طـلـبـات وشكاوى المـراجـعـين، وإحـالـتـها إلى الوحدات الإداريـة المعنيـة، والرد على اسـتـفـسـاراتـهم.
٥- استقبال طلبـات كبار السن والأشخـاص ذوي الإعـاقـة، بالتنسيق مع الوحـــدات الإداريـة المخـتـصـة، لإنجـاز مـعـامـلاتـهـم وتـذليـل أي مشكلات أو صـعـوبـات قـد تـواجـهـهـم.
٦- القيام بترتيبـات السـفـر والضيـافـة والإقـامـة لضيـوف الـوزارة وللـوفـود الـزائـرة والمغادرة، بالتنسيـق مـع الوحدات الإداريـة المعـنيـة.
۷- تنظيـم الأنشطـة الثقـافـيـة والاجتماعيـة والرياضيـة لموظفي الوزارة، والإشراف على الاحتفالات التي تقيمها أو تشارك فيـها الوزارة.
۸- تنظيـم المـؤتمرات والندوات والمعـارض التي تقيمهـا الـوزارة، وإعـداد الموازنة اللازمة لذلك، بالتنسيق مع الوحدات الإداريـة المعنـيـة.
۹ - تـرجـمـة الـوثـائـق والـكـتـب والبحـوث العلميـة والمكاتبـات الأجنبيـة المتـعـلـقـة باخـتـصـاصـات الـوزارة.
 

 

المادة 7


تختص إدارة التـمـويـل والـتـأمين الصحي بما يلي:
۱- تـوفـيـر نـظـام تـأمـيـن صـحـي مـتـكـامـل بـجـودة عاليـة يـتـســم بالـكفـاءة والاسـتـدامـة.
۲- وضع السيـاسـات والخطط والمعاييـر والنـظم والإجـراءات اللازمـة لتطبيـق نـظام التأمين الصحي وفـق أحـكام القـانـون المنظـم لـتـقـديـم خـدمـات الـرعـايـة الصـحـيـة داخـل الـدولـة، والإشراف على تنفيذه.
۳- تحـديـد حـزم خـدمـات الـرعـايـة الصـحـيـة الأسـاسيـة حـسـب فـئـات المستفيدين، ومـسـتـوى خـدمـات الـرعـايـة الصـحـيـة الـواجـب تـقـديمـها لكل فئة، وتحـديـد شـروط وضوابط شبكة مـقـدمـي خـدمـات الـرعـايـة الصحيـة التي يمكن لشـركـات التـأمين الصحي التـعـاقـد مـعـها.
٤- تـسـعـيـر وتحـديـد آلـيـات تمـويـل خـدمـات الـرعـايـة الصـحيـة ضـمـن نظـام التـأمـين الصـحي، وحـسـاب وتطبيـق ومـراجـعـة أقـسـاط التأمين الصحي، وتحـديـد آليـات تحـصيـلها وآليـات الـدفـع لمقـدمي الخـدمـات الصـحـيـة.
٥- تسـجيـل شـبـكـة مـقـدمـي خـدمـات الـرعـايـة الصـحيـة وشركات التأمين، وفقاً للمعاييـر والشـروط المـوضـوعـة والمعـتـمـدة في القـانـون والقرارات المنظمة.
٦- إعـداد وتطبيـق نـظـم تـرميـز الخـدمـات الصـحـيـة المختـلفـة، ونـظم تصنيف المرضى، وتعيين الحـد الأدنى للبيـانـات اللازمـة، وتحـديـد آليـات رفـع المـطـالـبـات المـاليـة وآليـات التـحـكـيـم الخـاصـة بـها ودفـعـها.
٧- إعـداد وتطبيـق آليـات لـتـوعـيـة أطـراف العـلاقـة التـأمينيـة بنـظام التأمين الصحي، وآليـات تـسـويـة المنازعات المتعلقـة بنـظـام التـأمـين الصـحـي.
۸ -مـراقـبـة أداء أطـراف العـلاقـة التـأمـيـنـيـة لضمـان التـزامـهـم بالقـوانـين والقرارات المنظمـة .
٩- وضـع القـواعـد الخـاصـة بـالـدعـم التـأمـيـنـي، واقـتـراح الفئات المشمـولـة بـه.
١٠ -إصـدار التعليمـات والنشـرات المـوضـحـة للسيـاسـات والإجـراءات المتعلقة بـتـنـفـيـذ التشـريـعـات ذات الصـلـة بالتـأمين الصحي.
 

 

المادة 8


تختـص إدارة الـعـلاقـات الصـحـيـة الـدوليـة بمـا يـلـي :
١- إعداد أوراق العـمـل المتعلقة بالمؤتمرات والندوات والاجتماعات المحلية والإقليميـة والدوليـة فـيـما يخـص نشـاط الـوزارة، بالتنسيـق مـع الوحـدات الإداريـة المعـنـيـة والجـهـات المخـتـصـة.
٢- الإعـداد لحضور المؤتمرات والاجـتـمـاعـات الـتـي تـعـقـدهـا المنـظـمـات والهيئات الإقليمية والدوليـة المختصة بقـضـايـا الوزارة، التي تشـارك الدولة في عـضـويـتـها .
٣- إعـداد التقاريـر والـردود التي تطلبـها المنظمات والهيئـات الإقليميـة والدوليـة فيما يخص نشاط الوزارة، بالتنسيـق مـع الوحدات الإداريـة المعـنيـة.
٤- دراسـة ومـتـابـعـة تنفيـذ تـوصـيـات المنظمـات والهيئـات الإقليميـة والدوليـة فيمـا يـخـص نشـاط الـوزارة، بالتنسيق مـع الجهـات المختصة.
٥- إعداد التقارير اللازمـة بشـأن إسهـامـات المنظمات والهيئات الإقليميـة والدوليـة في تنميـة وتـدعـيـم الأنشـطـة المـخـتـلفـة في مجـالات عـمـل الوزارة، بالتنسيـق مـع الجـهـات المخـتـصـة .
٦- متـابعـة سـداد المساهمات الماليـة للـدولـة في المنظمات المعـنيـة بمـجـالات عـمـل الـوزارة التي تكون الـدولـة عـضـواً فـيـها، بالتنسيـق مـع الجـهـات المختصة.
٧- دراسة السبل الكفيلة بتفعيل دور الدولـة في مجـال نـشـاط الـوزارة المنـظـمـات العـربـيـة والإقليميــة والـدوليـة .
٨- جـمـع المعلـومـات والبحـوث الصـادرة عـن المنظمات والهيئـات ذات الصلة بقـضـايـا الـوزارة مـحـلـيـاً وإقـلـيـمـيـاً ودولـيـاً، ونـشـرهـا بـغـرض الاسـتـفـادة مـنـها.
 

 

المادة 9


تختص إدارة العـلاقـات الطـبيـة والعـلاج بالخارج بمـا يـلـي :
١ -تـنـظـيـم عمليـة العـلاج بالخارج، وفقا للقـانـون والأحكام المنظمة لذلك.
۲ -الإشراف الإداري والفني على عمـل الملحقيات الطبيـة وتقييم أدائها .
٣ -إدارة ومـراقـبـة الإجـراءات الخـاصـة باخـتـيـار المسـتـشـفـيـات والمـرافـق الطـبيـة لعـلاج المواطنين بالخارج والتعـاقـد مـعـها، بالتنسيـق مـع الجهات المخـتـصــة والوحـدات الإداريـة المعـنيـة.
٤- اتخاذ الترتيبات اللازمة لسفر المرضى للعلاج بالخارج والمرافقين لهم.
٥- مـتـابـعـة إنجـاز وتنسيـق أعمـال اللجان الطبيـة المختصة بحـالات العلاج بالخـارج، والليـاقـة الطـبيـة للمـوظـفـين، وإصـابـات الـعـمـل، والإجـازات المرضيـة، والشـؤون الاجـتماعيـة، والإفراج الطـبـي عـن المسجونين، والحالات الأخـرى التي يحددهـا الـوزيـر .
٦- إبلاغ الجـهـات المخـتـصـة بـقـرارات اللجان الطبيـة المخـتـصـة بشـأن الحالات التي تنـظـرهـ
 

 

المادة 10


تختـص إدارة الـصـحـة الإلكتـرونيـة بما يلي :
١- وضـع الاستراتيجيـة الوطـنـيـة للصـحـة الإلـكـتـرونيـة والمخـطط التنفيـذي لـها، ومتـابعـة تنفيـذه، بالتنسيـق مـع الـوحـدة الإداريـة والجـهـات ذات العـلاقـة .
۲ -اقـتـراح مشـروعـات الأدوات التشريعيـة، بالتنسيق مع الإدارة المعنيـة، ووضـع المـعـايـيـر والـلـوائـح المنظمـة للاستخـدام الأمثـل للبيئـة الإلكتـرونيـة فـي تـقـديـم الخـدمـات الصـحيــة في الدولـة بمـا يـشـمـل سـريـة المعلـومـات والبيـانـات الصحيـة وطـريـقـة تـخـزيـنـهـا والـدخـول إليها وأنظمـة التـرميـز اللازمـة .
٣- التـنـسـيـق بـيـن مـقـدمـي الخـدمـات الصـحيـة العـامـة للتـأكـد مـن قـابـليـة التـوافـق بين الأنظمة الإلكتـرونيـة لـتـخـزيـن وانـتـقـال المعلـومـات بينـها بسلاسة.
٤- وضع الخطط اللازمـة لـتـدريـب مـقـدمـي الخـدمـات الصـحـيـة للتـوسـع في استخـدام الأنظمـة الإلـكـتـرونيـة فـي تقـديـم خـدمـاتـهـم.
٥- إنشاء وتحـديـث قـاعـدة بـيـانـات مـركـزيـة لـفـهـرسـة وحـفـظ بيـانـات المرضى مرتبـطـة بالأنظمـة الإلكتـرونيـة لمـقـدمـي الخـدمـة في القطـاع الصحي العـام والخـاص ونظـام التـأمـين الصـحـي، وإعـداد التقاريـر اللازمـة بشـأنـها.
٦- العـمـل عـلـى تـوفـيـر نـافـذة للمرضى، لتـوفـيـر مـعـلـومـاتـهم الطبيـة والخـدمـات الصـحـيـة مـن خـلالـها.
٧- مـتـابـعـة الـتـطـور الـتـكـنـولـوجـي فـي الخـدمـات الصـحيـة، وتـقـديـم التـقـاريـر والـتـوصـيـات بـشـأنـها.
 

 

المادة 11


يختـص مـركـز المـعـلـومـات الصـحـيـة بما يلي :
۱ -تـنـظـيـم قـيـد المـوالـيـد والـوفـيـات وفقاً للـقـانـون.
۲ -بناء وتـطـويـر قـاعـدة بـيـانـات مـركـزيـة للبيـانـات والمعلـومـات الصحيـة، وجـمـع البيـانـات وتـوفـيـر الإحصاءات الدقيقـة ومؤشرات الأداء الصحيـة لخـدمـة الوحدات الإداريـة المعـنيـة بتـحسـين الـوضـع الصـحي ورفع كفـاءة خـدمـات الـرعـايـة الصـحيـة.
۳- تـنـظـيـم السـجـلات الصـحـيـة الـوطـنـيـة الـتـي تُعـنـى بـرصـد وإحـصـاء الأوبئة والأمـراض ذات الأولـويـة لخـدمـة الوحـدات الإداريـة المعـنيـة بوضـع البـرامـج الـوقـائيـة والعـلاجـيـة ودراسـة مـدى انتشـارهـا في المجتمع.
٤ -إعـداد الإحـصـاءات الصـحـيـة الوطـنـيـة ونـشـرهـا، بالتنسيـق مـع الوحدات الإداريـة المعـنـيـة والجـهـات المخـتـصـة .
٥- تحـلـيـل البيانات والمعـلـومـات الصـحـيـة واستخلاص المؤشرات الداعـمـة لعمـليـة اتخاذ القرار فيما يتعلق بتخطيط وتطـويـر القطاع الصحي .
٦- تـطـويـر وتطبيـق أدوات ذكـاء الأعـمـال لخـدمـة مـراقـبـة الأداء وتقـديـم تـصـور شـامـل لصانعي القـرار لاتخاذ قرارات مستنيـرة .
۷- ضـمـان تـطـبـيـق مـتـطـلـبـات الخـصـوصيـة والسـريـة خـلال كـامـل دورة حـيـاة إدارة البيـانـات، وفقاً للسيـاسـات الـوطـنـيـة المعـتـمـدة .
۸ -العـمـل كـمـركـز لـتـوفـيـر وتبادل البيـانـات والمعـلـومـات الصـحيـة مع المؤسـسـات الإقـلـيـمـيـة والـدوليـة .
 

 

المادة 12


تختص إدارة حـمـايـة الصـحـة ومـكـافـحـة الأمـراض الانـتـقـاليـة بمـا يـلـي :
۱- اقـتـراح مشـروعـات الأدوات التشـريـعـيـة والسيـاسـات والـلـوائـح الخاصة بالـوقـايـة مـن الأمـراض الانتـقـالـيـة والـوبـائـيـة، والإشراف على تنفيـذهـا، بالتنسيـق مــع الوحـدات الإداريـة المعـنـيـة والجـهـات المختصة .
۲ -وضع خطط وبـرامـج الـوقـايـة مـن الأمـراض الانـتـقـالـيـة والـوبـائـيـة، والإشراف على تنفيـذهـا، بالتنـسـيـق مـع الـوحـدات الإداريـة المعـنيـة والجـهـات المخـتـصـة.
٣- وضـع البـرامـج الإرشـاديـة والتثقيـفـيـة والـتـوعـويـة للـوقـايـة مـن الأمـراض الانـتـقـالـيـة، والإشراف علـى تـنـفـيـذهـا، بالتنسيـق مـع الوحدات الإداريـة المعـنـيـة والجـهـات المخـتـصـة.
٤- تحـديـث بـرنـامـج التـطـعـيـم المـوسـع للأطـفـال وبرنامج تطـعـيـم البالغين، وتقييمهما على المسـتـوى الـوطـني.
٥- اقـتـراح النسـب المسـمـوح بـهـا لمـلـوثـات الـبـيـئـة والمـواد المـركبـة، والمشاركة في وضع الخطط اللازمة لتـقـلـيـل هـذه النسب، وتنـظيـم الإجـراءات الكفيلة بالحـفـاظ عـلـى سـلامـة البـيـئـة وتـحـسـينهـا، بالتنـسـيـق مـع الجـهـات المخـتـصـة .
٦- الإشـراف والـرقـابـة على الأعـمـال المتـعـلـقـة بالتـخـلـص مـن النـفـايـات الطبيـة، بالتنسيق مع الجهات المخـتـصـة .
٧- اسـتـحـداث اختبارات للفـحـوصـات الطـبيـة النادرة والمختصة في مجال الفحص المخبـري المعني بسلامة البيئة ومكافحـة ومـراقبـة الأمـراض.
۸- الـعـمـل عـلـى تـطـويـر مـفـهـوم الصـحـة الـواحـدة، وذلـك مـن خـلال بـنـاء الشـراكـات الاستراتيجيـة مع الجهات ذات الصلة
 

 

المادة 13


تختـص إدارة تـعـزيـز الصـحـة بما يلي :
۱- اقـتـــراح مشـروعــات الأدوات التشـريـعـيـة والسيـاسـات واللوائح الخـاصـة بـتـعـزيـز صـحـة جـمـيـع فـئـات المجتمع المخـتـلفـة، والإشـراف على تنفيـذهـا، بالتنسيـق مـع الوحدات الإداريـة المعـنيـة والجـهـات المختصة.
۲- وضع الخطط والبـرامـج والمبادئ والإرشـادات التـوجـيـهيـة الـتـي تهـدف إلى تعـزيـز الصـحـة المهـنيـة، والإشراف على تنفيـذهـا، بالتنسيق مع الوحدات الإداريـة المعـنـيـة والجـهـات المـخـتـصـة.
٣ -إعـداد الخطط والإجـراءات الـوقـائـيـة الخاصـة بالحـوادث وإصـابـات العمل، ومتـابـعـة تـنـفـيـذهـا، وتقييمها بشكل دوري، بالتـعـاون مـع الوحدات الإداريـة المعـنـيـة والجـهـات المختصة.
٤- وضـع الخطط والبرامج والمبادئ والإرشـادات التـوجـيـهيـة الوطنيـة الخاصـة بتعـزيـز الصـحـة في مجـال التغـذيـة والنـشـاط البـدني، والإشراف علـى تـنـفـيـذهـا، وتقييمها، بالتنـسيـق مـع الوحـدات الإداريـة المعـنـيـة والجـهـات المخـتـصـة.
٥ -الـرقـابـة عـلـى المـوظـفـين الـعـامـلـيـن فـي مـجـال الإشـعـاع، واتخـاذ إجراءات استصدار وتجـديـد التـرخـيـص الـدوري لهـم، بالتنـسيـق مـع الجـهـات المختصة .
٦- وضـع بـرامـج إرشـاديـة وتثـقـيـفـيـة لـتـعـزيـز الصـحـة، ووضع الخطط التـوعـويـة وخـطط الإعـلام الصـحي والإجـراءات اللازمة لتحقيـق أهـداف الصـحـة الـعـامـة، بالتنـسـيـق مـع الوحدات الإداريـة المعـنيـة والجـهـات المخـتـصـة.
٧- المشـاركـة فـي تنـظيـم الحـمـلات الـتـوعـويـة والتـثـقـيـفـيـة لتعـزيـز الصـحـة في المجـالات المختـلفـة، بالتنسيـق مـع الـوحـدات الإداريـة المعـنـيـة والجـهـات المخـتـصـة .
۸ -تطبيـق الـقـوانـين المتعلقة بالـرقـابـة علـى التـبـغ، ووضـع الاشتـراطـات الصـحـيـة الخاصـة بـه.
۹ -تـوفـيـر الخـبـرات والدعـم لـتـعـزيـز الصـحـة العـامـة والشراكة المجتمعية .
١٠- التنسيـق مـع مـقـدمـي خـدمـات الـرعـايـة الصـحـيـة لـدعـم الخـطـط الاستراتيجيـة لـتـعـزيـز الصـحـة وبـرامـجـها.
 

 

المادة 14


تختص إدارة بـرامـج الـوقـايـة مـن الأمـراض غـيـر الانـتـقـالـيـة بمـا يـلـي :
۱ -اقـتـــراح مشـروعــات الأدوات التشـريـعـيــة والسيـاسـات والـلـوائـح الخـاصـة بالأمراض المزمنـة غيـر الانتقاليـة ذات الأولـويـة، والإشـراف على تنفيـذهـا، بالتنسيـق مـع الـوحـدات الإداريـة المعـنـيـة والجهـات المختصة.
۲ -وضـع البـرامـج الإرشـاديـة والتـثـقـيـفـيـة والـتـوعـويـة للوقـايـة مـن الأمـراض المزمنـة غيـر الانـتـقـالـيـة ذات الأولـويـة، والإشراف على تنفيذها، بالتنسيق مع الوحدات الإداريـة المعـنـيـة والجهات المختصة.
۳- تنفيذ الخطط الاستراتيجيـة الخـاصـة بالأمـراض غيـر الانتقاليـة، بالتنسيـق مـع الوحدات الإداريـة المعـنـيـة والجهات المخـتـصـة.
٤ -التنسيق مـع مـقـدمـي خـدمـات الـرعـايـة الصـحـيـة لـدعـم الخـطـط الاستراتيجيـة للـوقـايـة مـن الأمـراض غير الانتـقـاليـة وبـرامـجـها.
٥- وضع إطـار العـمـل والخـطـة الخـاصـة بالخـدمـات الصـحـيـة مـع التـركـيـز على الأمراض المزمنـة ذات الأولـويـة، والسلوكيات المؤثرة في هذه الأمراض، بالتنسيق مع الوحدات الإدارية المعنية والجهات المختصة.
٦- مـتـابـعـة تطبيق القـانـون المنظم للصـحـة النـفـسيـة .
 

 

 

المادة 15


تختص إدارة القـومـسـيـون الـطـبـي بمـا يـلـي :
۱- تنفيـذ القـوانـين واللـوائـح والقـرارات ذات العـلاقـة بالقـومسـيـون الطبي.
۲ -فـحـص أفـراد العـمـالـة الـوافـدة إلى الدولـة بـغـرض الحـصـول علـى تصريح الإقـامـة، وفقاً للقـانـون والقرارات المنفذة له.
۳- فـحـص المرشحين للتعيـين، وتقـريـر لـيـاقـتـهم الصحيـة للتـعـيـين في الخـدمـة .
٤- الفحص الطبي لأيـة فـئـات أخـرى يـصـدر بـهـا قـرار مـن الـوزيـر 
 

 

المادة 16


تختص إدارة سـلامـة الـغـذاء بما يلي :
١- تطبيـق أحـكـام قـانـون تـنـظـيـم مـراقـبـة الأغـذيـة الآدمـيـة، وإعـداد القرارات والاشتـراطـات الفـنـيـة ذات الصلة .
۲ -اقـتـراح السيـاسـات ووضـع الخطط والقـواعـد الإرشـاديـة المتعلقـة بسـلامـة وصـلاحـيـة الغـذاء عبـر مـراحـل السـلسـلـة الغـذائـيـة.
٣- الـرقـابـة والتـفـتـيـش عـلـى الغـذاء وأمـاكـن تـداولـه ومـتـداوليـه عبـر مـراحـل السـلـسـلـة الغـذائـيـة، لضمــان سـلامتـه وصـلاحـيـتـه، ومـنـع تداول أي غـذاء غيـر مـطـابـق للمـواصـفـات والاشتـراطـات، بالتنسيـق مع الجهات المخـتـصــة.
٤ -وضع التـدابـيـر الصـحـيـة الـوقـائـيـة عبـر مـراحـل السـلـسـلـة الغـذائـيـة وفقا للمبادئ والمعـايـيـر والأدلـة العـلـمـيـة المبـنـيـة علـى مـبـدأ تحـليـل المخـاطـر، بما يحمي صـحـة المستهلك، وتنفيذ هذه التدابيـر بـشـفـافـيـة ومـوضـوعيـة واستقلاليـة.
٥ -المشاركة في إعـداد ومـراجـعـة وتـطـويـر مـواصـفـات الغـذاء القيـاسيـة ولـوائـحـه الفـنـيـة.
٦ -وضـع وتطبيـق أنظمـة تـتبـع فـعـالـة للغـذاء، وسـحـب الغـذاء غيـر المطابق للاشتراطات الفـنـيـة المعتمدة، والتنسيق مع الجهات المختصة للحصول على بيانات التتبع الخـاصـة بمـدخـلات الإنتـاج الـزراعـي.
۷ -وضع وتنفيذ الإجراءات المتعلقة بتسجيـل المـواد الغذائيـة والغـذاء ذي الاسـتـخـدام التغـذوي الخاص، ومنشآت تداول الغذاء، وتحـديـث البيـانـات الخـاصـة بـذلك.
۸ -وضع وتنفيذ الإجراءات المتعلقة بمـواجـهـة حـالات الطوارئ المرتبطـة بســلامـة وصـلاحـيـة الغـذاء، ابتـداءً مـن المـزرعـة ومـدخـلات الإنتـاج الزراعي ومـراحـل تـداول الغـذاء عبـر مـراحـل السـلـسـلـة الغـذائيـة، بالتنسيق مع الجهات ذات العلاقة والوحدات الإداريـة المعـنيـة .
٩- إصـدار الشـهـادات الصـحـيـة للمـواد الغذائيـة المحـلـيـة والمصدرة ومنـشـآت تـداول الغـذاء عبـر مـراحـل السـلـسـلـة الغذائيـة، واعتمـاد الشهادات الصحيـة للمـواد الغذائيـة المستـوردة، بنـاءً عـلـى الاشتـراطـات الفـنـيـة المعـتمـدة.
۱۰- التنسيـق مـع الجـهـات المختصـة بشـأن إصـدار تـراخـيـص مـنـشـآت تداول الـغـذاء .
١١- إصـدار تـصـاريـح مـزاولي أعـمـال تـداول الـغـذاء .
۱۲- التـنـسـيـق مـع الجـهـات المختصـة بشـأن مـنـع الإعـلان عـن الغـذاء الذي يتعارض مع أحـكـام قـانـون تـنـظـيـم مـراقـبـة الأغـذيـة الأدمـيـة.
۱۳ -اعتماد الهيئات الإسـلامـيـة والمسالخ والمقـاصـب والشـركـات المصدرة للغـذاء الحـلال، بالتنسيـق مـع الجهات المختصة.
١٤ -إدارة وتشغيـل وتـطـويـر مـخـتـبـرات لفحص وتحـلـيـل عـيـنـات المـواد الغذائيـة ومـا يتصل بها، وإصـدار تـقـاريـر مـخـبـريـة واضـحـة بشـأنـها.

 

المادة 17


تختـص إدارة الـطـوارئ الصـحـيـة بمـا يـلـي :
١- وضـع الـسيـاسـات والخطط اللازمة لإنشـاء نـظـام مـتـكـامـل للتـأهـب والاستـجـابـة للطـوارئ الصحيـة علـى الصـعـيـد الـوطني، بالتنسيق مـع الجـهـات المخـتـصـة.
۲ -اقـتـراح المعـايـيـر المتعلقة بالتـأهـب والاستـجـابـة للطوارئ الصـحـيـة، وتنفيـذهـا بـعـد اعـتـمـادهـا.
۳- رصـد وتـقـيـيـم المـوارد والقدرات المتاحة في قطاع الصـحـة للاستـجـابـة بفـعـالـيـة لحالات الطوارئ، وإعـداد التـقـاريـر بـشـأنـهـا .
٤ -تحـلـيـل وتقييـم المخـاطـر ووضـع الخطط المنـاسبـة مـع القـطـاعـات ذات الصلة.
٥- تحـديـد أدوار ومسؤوليـات المـرافـق الصحيـة للتـأهـب والاستـجـابـة للطوارئ، ومـراجـعـة الخطط المتعلقة بذلك.
٦- وضـع الخطط والآليـات المتعلقـة بالمخـزون الوطني الاستراتيـجـي مـن الأدويـة واللـقـاحـات والأجـهـزة الطبيـة.
٧- التنسيـق مـع الـوحـدات الإداريـة المعـنـيـة والجـهـات الحكوميـة وغيـر الحكومية لوضع آليـة مـشـتـركـة فـعـالـة للاستجـابـة للطوارئ الصحيـة.
٨- رفع مستـوى الـوعـي لـدى الجـمـهـور حـول أنظمـة إدارة الطوارئ الصـحـيـة بمـشـاركة القـطـاعـات ذات الصـلـة لـتـعـزيـز قـدرات ومـرونـة الاسـتـجـابـة في المجـتـمـع والقـطـاع الصحي.
۹ -تحـديـد السـيـاسـات التـدريبيـة والـدورات التعليميـة الخـاصـة بنـظام التـأهـب والاسـتـجـابـة للطوارئ الصـحـيـة لرفع كفاءة القطاع الصحي .
۱۰ -التنسيـق مـع الوحدات الإداريـة المعـنـيـة والمنـظـمـات الدوليـة وخـاصـة منظمـة الصـحـة العـالميـة في مجـال التـأهـب وإدارة الطوارئ وتنفيـذ اللوائح الصـحـيـة الـدوليـة .
 

 

المادة 18


تختـص إدارة التـخـصـصـات الصـحـيـة بمـا يـلـي :
۱- اقـتـراح مشـروعـات الأدوات التشـريـعـيـة والسياسات واللوائح والخطط والبـرامـج الخـاصـة بتنظيـم مـزاولـة المـهـن الصـحـيـة والطبيـة، بالتنسيق مع الوحـدات الإداريـة المعنيـة .
۲- وضـع المـواصـفـات والشـروط الخـاصـة باعتمـاد المستـشـفـيـات والمـراكـز الصـحـيـة والعيادات، مـراكـز لـتـدريـب وتـأهيـل الممارسين الصحيين .
٣- القيام بجـمـيـع الأعـمـال الإداريـة المتعلقة بإصـدار تـراخـيـص مـزاولـة المهن الصحيـة والطبيـة، وإصـدار الـتـراخـيـص بمزاولتها، وإلغائها. 
٤- وضع أسـس تـقـيـيـم شـهـادات الاخـتـصـاصـات الطبيـة المختـلفـة، وتحـديـد الشـروط اللازم تـوافـرهـا لإدراج اسـم الممـارس الصـحي في سجـلات الـوزارة، والتي تـؤهـلـه للـحـصـول على درجـة الاخـتـصـاص مستقبلاً، والإشراف عـلـيـها.
٥- وضـع مـعـايـيـر البـرامـج التـخـصـصيـة الصـحيـة، والإشراف عـلـيـها.
٦- وضـع مـعـايـيـر بـرامـج التـعـلـيـم الطـبي المستمر في جـمـيـع الـتـخـصـصـات الصـحـيـة.
۷- وضـع الشـروط الخاصة بأخلاقيات ممـارسـة المـهـن الـصـحيـة والطبيـة، واتخاذ الإجـراءات التي تهدف إلى التـأكـد مـن الالتزام بـها .
۸ -تـلـقـي الشكاوى بشـأن أي مـن الاخـتـصـاصـات الصـحـيـة والطبيـة وممـارسـيـها، واتخاذ مـا يـلـزم بشـأنـها .
٩ -وضع الخطط العـامـة لـتـدريـب وتـطـويـر المـوارد البشـريـة الصـحـيـة.
۱۰ -التنـسيـق والـتـعـاون مع الجهات المخـتـصـة في مجـال التخصصـات الصـحـيـة الإقليميـة والـدوليـة.
۱۱- التعـاون مـع المؤسسات التعليميـة والأكاديميـة والهيئات الطبيـة المختلفة لتنـظيـم وعـقـد الـدورات التـدريبيـة لجميـع مزاولـي المهـن الصحـيـة والـطـبـيـة لتـأهـيـلـهـم للاخـتـصـاص

 

المادة 19


تختـص إدارة المـنـشـآت الصـحـيـة بما يلي :
۱ -اقـتـراح مشـروعـات الأدوات التشـريـعـيـة واللوائح المنظمـة للمنشـآت الصحية، بالتنسيق مع الوحدات الإدارية المعنيـة والجهات المختصة .
٢ -إعداد وتـطـويـر الـسيـاسـات العـامـة والمعـايـيـر الخـاصـة بترخيص واعتماد منـشـآت الـرعـايـة الصـحـيـة بمـا يـتـوافـق مـع أفضل الممارسات الدوليـة .
٣- وضع المعايير والإجراءات المنظمـة لتسجيـل واعتماد الأجهزة الطبيـة، والـرقـابـة عـلـيـها، بالتنـسيـق مـع الجـهـات المختصة.
٤- الـرقـابـة والتـفـتـيـش الميـدانـي علـى منشـآت الـرعـايـة الصـحيـة، بالتنسيـق مـع الوحـدات الإداريـة ذات العـلاقـة، وإعـداد التقاريـر اللازمـة بـشـأنـها.
٥ -متابعة تنفيذ القرارات الخاصـة بـتـرخـيـص منشآت الـرعـايـة الصحيـة، بما في ذلك الجزاءات التي تـصـدر بشـأن المخـالـفـات.
٦- إعـداد وتـطـويـر بـرامـج التـدريـب التخصصية لإدارة المنشآت الصحيـة، بالتنسيق مع الوحدات الإدارية المعنية.
٧- القيام بجميع الأعمال الإداريـة المتعلقة بإصـدار تـراخـيـص المنشـآت الصحيـة، وإنشـاء السجـلات والملفـات الخـاصـة بـها، وحـفـظـها، وإصـدار النماذج اللازمة، وفقاً لأحكام القـانـون واللـوائـح والقـرارات المنظمة لذلك.
۸- تلقي الشكاوى الخـاصـة بأعـمـال التـرخـيـص والتسجيـل، واتخاذ الإجـراءات اللازمـة بشـأنـها
 

 

المادة 20


تختص إدارة جـودة الـرعـايـة الصـحـيـة بمـا يـلـي :
۱- تـطـويـر الـسيـاسـات والخطط الاستراتيجيــة والبـرامـج الخـاصـة بجـودة الـرعـايـة الصـحـيـة وفقاً لأفـضـل الممارسات الدوليـة، بالتنسيـق مع الوحدات الإداريـة المعـنـيـة والجهات المختصة، ومتـابعـة تـنـفـيـذهـا.
۲- متـابـعـة تـنـفـيـذ وتفـعـيـل آليـة الـرقـابـة والتـدقـيـق والتقييم المستمـر لجـودة أداء منـشـآت تـقـديـم خـدمـات الـرعـايـة الصـحـيـة، وإعـداد التـقـاريـر اللازمـة بشـأنـها.
۳- نشر ثـقـافـة ومـفـهـوم جـودة الـرعـايـة الصـحيـة بين مقـدمـي خـدمـات الرعاية الصحيـة وأفراد المجتمع، بالتنسيق مع الجـهـات المختصة.
٤- تشجيع الشراكة المجتمعية وتمكين أفراد المجتمـع ومـقـدمـي خـدمـات الـرعـايـة الصـحـيـة مـن المسـاهـمـة فـي تحـسـين جـودة خـدمـات الـرعـايـة الصـحـيـة وتطـويـرهـا بشكل مستمـر.
٥- الاسـتـخـدام الأمثـل والتحـلـيـل الفـعـال للبيـانـات والمعـلـومـات لإثـراء عـمـلـيـات وإجـراءات تحسين الجـودة .
٦ -قـيـاس رضـى المسـتـخـدمـين ومتلقـي خـدمـات الـرعـايـة الصـحـيـة، بالتنسيق مع الوحدات الإداريـة المختلفة.
۷- وضع أطر لتحسين الإجـراءات الطبيـة مـن الإرشادات الإكلينيكيـة والأدلة الإرشـاديـة العـلاجـيـة الـوطـنـيـة المبنـيـة على أفـضـل الممارسـات العـالميـة، بالتنسيـق مـع الـوحـدات الإداريـة والجـهـات ذات العـلاقـة .
۸ -متـابـعـة إنجـاز وتنسيـق أعـمـال اللجـان وفـرق العمـل المخـتـصـة بمجـالات الـوقـايـة مـن العـدوى ومـكـافـحـتها، ومقـاومـة مـضـادات الميكروبات، وإعـداد التـقـاريـر اللازمـة بـشـأنـها.
٩ -بنـاء الـكفاءات وتـطـويـر القـدرات في مختـلـف مـجـالات الجـودة وسلامة المرضى.
١٠- إدارة وتنفيـذ البـرامـج الوطنيـة للـوقـايـة مـن العـدوى ومـكـافـحـتـها ومقـاومـة مـضـادات الميكروبات، بالتنسيـق مـع مـخـتـلـف القـطـاعـات والجهات ذات العـلاقـة.
۱۱- التـعـاون مـع الـوحـدات الإداريـة المختصة لوضع نظام لضمـان جـودة خـدمـات الـرعـايـة الصـحـيـة، وربـطـه بنـظـام تـمويـل الخـدمـات الصحيـة، ومـتـابـعـتـه وتـطـويـره بشكل مـسـتـمـر.
 

 

المادة 21


تختص إدارة الصيـدلـة والـرقـابـة الـدوائـيـة بمـا يـلـي :
۱ -اقـتـراح السـيـاسـات الـدوائـيـة فـي الـدولـة .
۲ -تنـظيـم عـمـل الـصـيـدلـيـات الخـاصـة ومـخـازن ومصانع الأدويـة، بالتنسيق مع الجهات المخـتـصـة .
٣- إعـداد بـرامـج مـنـاوبـات صـيـدلـيـات القـطـاع الخـاص، بالتنسيـق مع الجهات المخـتـصـة.
٤- الإشراف على تحـديـد أسعار الأدويـة في القطاع الخاص، بالتنسيـق مع الجهات المخـتـصـة.
٥ -الـرقـابـة علـى جـودة وصـلاحـيـة الأدويـة والمستـحـضـرات الصيدليـة، والمواد الكيميـائيـة المسـتـخـدمـة في إعـدادهـا .
٦- التـفـتـيـش عـلـى جـمـيـع الصـيـدلـيـات الـعـامـة والخـاصـة ومـخـازن ومصانع الأدويـة، للتـحـقـق مـن تـنـفـيـذهـا للقوانين المعمـول بـها.
۷- التـعـاون مـع الجـهـات المختصة في مكافحـة غـش الأدويـة وممارسـة مهنة الصيـدلـة دون ترخيص .
۸- وضـع قـواعـد وضـوابـط صـرف الأدويـة المخدرة والمنـومـة والمنشـطـة، ومتـابـعـة تـقـاريـر التفتيش على الصيدليـات في شأنـها .
٩ -إصـدار تـصـاريح استيراد وتصدير الأدويـة والمستـحـضـرات الصيدليـة والمكملات الغذائيـة ومستـحـضـرات الـتـجـمـيـل العـلاجـيـة.
١٠- الإشراف على تسجيـل شـركـات الأدويـة ومـنـتـجـاتـها، بالتنسيـق مـع الوحدات الإداريـة المعنية، وإعـداد السجلات الخـاصـة بذلك.
١١- الـرقـابـة على اسـتـخـدام الأعـشـاب والأدويـة الشعبيـة، ومـن يـقـومـون بإعـدادهـا وبـيـعـها والتـعـامـل فيـها، وإصـدار الإرشـادات اللازمـة لاسـتـخـدامـها على النحو الصحيح
 

 

المادة 22


تختص إدارة تـنـظـيـم البـحـوث الصـحـيـة بمـا يـلـي :
۱- اقـتـراح مشـروعـات الأدوات التشريعيـة والسيـاسـات واللوائح الخـاصـة بالبحوث الصحـيـة والطبيـة لخـلـق بيئـة بحـوث آمنـة ومـتـطـورة، بالتنسيق مع الإدارة المعـنيـة.
۲- وضع السياسات والأطـر الـتـي تحـدد الاتجـاه الاستراتيجي الـوطـنـي للبحوث الصحيـة .
۳- تسجيـل لجـان البحـوث البشـريـة والحـيـوانـيـة والمعـمـليـة، واعتمـاد مؤسسات البحث، للقيام بأنشـطـة البحـوث الصـحـيـة فـي الـدولـة .
٤- تـفـعـيـل سـبـل المـراقبـة والتـدقيـق علـى سـيـر البحـوث الصحيـة لتحقيـق أعلـى مـعـايـيـر الجـودة العلميـة والنـزاهـة الأخـلاقـيـة، وحفظ سلامة المرضى وسـريـة مـعـلـومـاتـهـم الشـخـصـيـة، والصـحـيـة، والجـيـنـيـة، والوراثيـة.
٥- إنشـاء قـاعـدة بيـانـات لجـمـع وتحـلـيـل الأنشـطـة البحثيـة مـحـليـاً ودوليـاً، واسـتـخـدام الأدلـة البحـثيـة لـتـحـسـين الأداء العـمـلـي للمؤسسات والخـدمـات الصـحـيـة، بالتنسيـق مـع الـوحـدة الإداريـة
المعنيـة.
٦- دعم الأنشطة البحثيـة والباحثين والبـنـيـة الأسـاسيـة، وتـدبـيـر التمـويـل اللازم لتـلـبـيـة احـتـيـاجـات البحـوث الصـحـيـة بـالـدولـة.
٧ -العـمـل كـجـهـة اتـصـال تـعـنـى باستمرار عـمـلـيـات التنسيـق وتبسيط وتحسين السياسات والمتطلبات المتعلقة بـسـلـوكـيـات ومـراقبـة البحـوث الـتـي يـتـم إجراؤهـا علـى البـشـر والأنسـجـة والخلايا الجذعيـة، ومشـاريـع عـلـم الـوراثـة والجـيـنـوم والطب الشخصي.
۸ -تـفـعـيـل سبـل التـعـاون بـين مـؤسـسـات البحـث الصـحي لتـحـقـيـق الاسـتـخـدام الأمثـل والمشـاركة الفـعـلـيـة للمـعـلـومـات والابتكارات وكذلك الموارد الماليـة والعـمـلـيـة .
۹ -مـسـانـدة التطبيـق العمـلي للاكتـشـافـات والإنجـازات الطبيـة الـهـامـة التي مـن شـأنـها تحـسـيـن سـبـل الـرعـايـة الصـحـيـة ووسـائـل الـعـلاج .
۱۰- التنسيق بين الطلبـة البـاحـثـين والخـبـراء المخـتـصـين للحـصـول علـى التـدريـب والخـبـرة العـمـلـيـة في مجال البحـوث الصـحـيـة.
 

 

المادة 23


تختص إدارة الموارد البـشـريـة بمـا يـلـي :
۱ -تـنـفـيـذ القـوانـين واللوائح والنظم المتـعـلـقـة بشـؤون الموظفين .
۲- وضـع السياسات والخطط الاستراتيجيـة والمشـاريع لتـطـويـر خـدمـات وممارسات الموارد البشـريـة، بالتـعـاون مـع الوحدات الإداريـة المعـنـيـة والجـهـات المختصة، ومتـابـعـة تـنـفـيـذهـا.
٣- إعـداد الدراسـات الخـاصـة بتخطيط الموارد البشـريـة، وتحـديـد احـتـيـاجـات الـوزارة مـن الـوظـائـف والمـوظـفـين، بالتنسيـق مـع الجهـة المختـصـة والوحـدات الإداريـة المختـلفـة.
٤ -إعـداد مشـروع وصـف وتصنيــف وترتيـب الوظـائـف بالـوزارة، ومتـابعـة تنفيذه، وتطـويـره.
٥- إعـداد مشــروع مـوازنـة البـاب الأول، بالتنسـيـق مـع الوحـدات الإداريـة المعـنيـة .
٦ -تنفيذ إجراءات تقييـم الأداء للمـوظـفـين.
٧- القيـام بإجـراءات التـعـيـين والنـقـل والنـدب والإعـارة.
۸- دراسة استحـقـاق المـوظـفـين للإجـازات، وفقاً للـقـانـون.
٩ -إعداد الدراسـات الخـاصـة بالهيكل الوظيـفـي لـلـوزارة، وتقـديـم المقترحات اللازمـة لتطـويـر التنـظيـم الإداري بـها.
۱۰- تحـديـد الاحـتـيـاجـات التـدريبيـة لمـوظـفـي الـوزارة، بالتنسيـق مـع الوحـدات الإداريـة المـخـتـلفـة، ومتـابعـة تنفيـذهـا، وتقييـم مـدى الاستفادة منـها .
١١- إدارة بـرنـامـج الابتـعـاث والمنح الدراسيـة، ومتـابعـة شـؤون الدارسين والمبتعثين، وإعـداد التـقـاريـر المتـعـلـقـة، بالتنسيـق مـع الجـهـات المختصة.
۱۲- اتخاذ الإجراءات اللازمة لتخصيص السكن الحكومي وصـرف بـدل الأثـاث لموظفي الوزارة، بالتنسيق مع الجهات المختصة .
۱۳- وضع وتحـديـث قـواعـد البيانات الخاصـة بشـؤون مـوظـفـي الـوزارة.
١٤- مـتـابـعـة الـتـزام المـوظـفـيـن بقـواعـد السـلـوك وأخـلاقـيـات العـمـل واتخاذ الإجـراءات اللازمـة في حـال عـدم الامتثال، بالتنسيـق مـع الوحدات الإداريـة المعـنيـة.
 

 

المادة 24


تختص إدارة الـشـؤون الماليـة بمـا يـلـي :
۱- تنفيذ القـوانـين واللوائح والنظم الماليـة المتـعـلـقـة بـعـمـل الـوزارة .
٢- إصـدار المعايير والتعـلـيـمـات المـاليـة للقـطـاع الصـحـي ومـراجـعـتـها بشكل دوري وفـقـاً للقـرارات والتـوجـيـهـات الصـادرة مـن الجـهـات المختصة في الـدولة.
٣- التنسيق مع الوحدات الإداريـة المخـتـصـة لتنـظـيـم رصـد وحـسـاب تـكـالـيـف الخـدمـات والبـرامـج المقـدمـة مـن قـبـل الـوزارة.
٤- إعداد مشروع الموازنة السنـويـة للوزارة وللقطاع الصحي الحـكـومـي، والحـسـاب الخـتـامي، بالتنسيـق مـع الـوحـدات الإداريـة المعـنيـة والجهات المختصة.
٥ -الإشـراف علـى تـنـفـيـذ الاعتـمـادات، وتـدقـيـق حـسـابـات الإيرادات والمصروفـات، وإعـداد التـقـاريـر الـدوريـة اللازمـة، وذلـك لـدعـم اتخاذ القرار في الوقت المناسب .
٦- إعـداد مـسـتـنـدات الصـرف، وسـائـر المعـامـلات الماليـة الأخـرى.
٧- تحصيـل الإيرادات المستحـقـة والـرسـوم ومـقـابـل الخـدمـات الـتي تؤديها الوزارة، واتخاذ الإجـراءات اللازمـة بشـأنـها.
٨- إدارة الأصـول الثـابتـة وغيـر الثـابتـة والمستهلكـة للـوزارة، وفـقـاً للوائح والنظم والقرارات المعـمـول بـها.
٩ -الإشـراف علـى جـمـيـع المعـامـلات الماليـة والمصـروفـات الخـاصـة بالعلاج بالخارج، وتنـظـيـم السجلات المحاسبيـة لـهـا والـتـدقـيـق عـلـيـها
 

 

المادة 25


تختـص إدارة المشـتـريـات والمنـاقـصـات بمـا يـلـي :
۱ -تنفيذ القـوانين واللوائح والنظم الماليـة المرتبـطـة بالمشـتـريـات .
۲ -تـوفـيـر احـتـيـاجـات الـوزارة ووحداتهـا الإداريـة المختـلفـة مـن المستـلـزمـات والأجـهـزة اللازمـة لأداء مـهـامـها، بالتنسيـق مـع الـوحـدات الإداريـة المعـنيـة .
۳- مبـاشـرة جـمـيـع الأعمـال المـاليـة الخـاصـة بشـراء المـواد والمستـلـزمـات والأثـاث والأجـهـزة .
٤ -القيام بأعـمـال المشـتـريـات والمنـاقـصـات والمـزايـدات، وفـق الأنـظـمـة والقـواعـد المعـمـول بـها في الـدولـة.
 

 

المادة 26


تختص إدارة المـرافـق والخـدمـات الـعـامـة بمـا يـلـي :
۱ -تـوفـيـر احـتـيـاجـات الـوزارة ووحـداتـهـا الإداريـة المختـلفـة مـن المستـلـزمـات والأجـهـزة اللازمة لأداء مهـامـها، بالتنسيـق مـع الـوحـدات الإداريـة المعـنيـة .
۲ -تـوفـيـر خـدمـات النـظـافـة والضيـافـة والخـدمـات العـامـة الأخـرى، والإشراف عـلـيـها .
۳- صيـانـة المبـاني وشبــكـات الصـرف الصـحي ومعـداتـها وشـبـكـات المياه داخـل مـبـانـي الـوزارة، واتخاذ إجـراءات تأثيثـها، بالتنسيـق مع الوحدات الإداريـة المعـنيـة .
٤ -تسلـم وتـسـلـيـم وتصـنـيـف وقـيـد البـريـد الـوارد والصادر .
٥- تنـظـيـم أرشيـف الـوزارة، وحـفـظ الوثـائـق، وفقاً لأحـدث الـطـرق.
٦ -تـوفـيـر وسـائـل النـقـل والآليــات والمعـدات الخاصـة بالـوزارة، وصيانتها، وتـوفـيـر قـطـع الغـيـار والـوقـود.
۷- وضـع خـطط الأمـن والسـلامـة، ومتـابـعـة تـنـفـيـذهـا.
۸ -الإشراف على مخازن الوزارة .
٩ -إدارة الأصـول والإشراف علـى حـسـن اسـتـخـدامها واستـغـلالـها، وتحـديـث قـاعـدة بيـانـاتـها، وإعـداد التـقـاريـر الـدوريـة بشـأنـها.
 

 

المادة 27


تختص إدارة نـظـم المـعـلـومـات بمـا يـلـي :
١ -إعـداد خـطط وسـيـاسـات اسـتـخـدام الحاسـب الآلـي في أنشـطـة الوزارة، ومتـابـعـة تنفيـذهـا.
۲ -برمجـة وحـفـظ واستـرجـاع وتطـويـر نـظم المعلـومـات والبيـانـات اللازمة لأنشـطـة الـوزارة .
۳- تـأمـين الأنظمـة والبرمجيـات والشبكات والبنية التحتيـة لنـظم المعـلـومـات مـن مـخـاطـر الحـوادث الإلكتـرونيـة.
٤- إعداد خطط التعـافـي مـن آثـار الحـوادث الإلكتـرونيـة حـال وقـوعـها أو تقليـل حـدوثها أو تصعيـدهـا، وفـق الـسيـاسـات والمبادئ التـوجـيـهيـة لأمـن المعـلـومـات بالتنسـيـق مـع الجـهـة المختصة.
٥- تـوفـيـر وصـيـانـة الأجهـزة والبـرامـج والشبكات الإلكتـرونيــة اللازمـة لنـظم العمـل الآليـة بـالـوزارة، بالتنسـيـق مـع الـوحـدات الإداريـة المعنيـة.
٦- تـصـمـيـم وتشـغيـل وإدارة قـواعـد البـيـانـات والمعلـومـات بـأنـواعـها المختلفة.
۷- تقـديـم الـدعـم الفـنـي الـلازم لمسـتـخـدمي الشبكات الإلكتـرونيـة بالـــوزارة، وتـدريـبـهـم عـلـى استخـدام الأنظمـة والبـرامـج وأجـهـزة الحاسب الآلي وملحـقـاتـها .
٨- إنشاء ومتـابـعـة وتحديث موقع الوزارة على شبكة المعلـومـات الـدوليـة
 

 

المادة 28


يصدر بتحـديـد اختصـاصـات كـل مـن مـكـتـب الوزيـر، ومكتـب وكيـل الوزارة، ومكتـب وكيـل الـوزارة المسـاعـد، قـرار مـن الـوزيـر 

 

المادة 29


يجـوز بقـرار مـن مـجـلـس الـوزراء، بنـاءً عـلـى اقـتـراح الـوزيـر، تعـديـل تنظيم الوحدات الإداريـة الـتـي تتـألـف منهـا الـوزارة بالإضـافـة أو الحـذف أو الدمـج، وتعيين اختصـاصـاتـهـا وتـعـديـلـها.

 

المادة 30


يجـوز بقـرار مـن الوزيـر إنشاء ما تقتضيه المصلحـة العـامـة مـن أقسام في الوحدات الإداريـة التي تتـألـف منهـا الـوزارة، أو إلغاؤهـا أو دمجها، وتعيين اختصـاصـاتـهـا وتـعـديـلها، ولا يـكـون قـرار الوزيـر نـافـذاً إلا بعـد اعتماده من مجلس الوزراء.

 

المادة 31


يُلغى القـرار الأميري رقم (١٤) لسنة ٢٠١٩ المشـار إليـه.
