مـرسـوم رقـم (90) لـسـنـة 2025 بـالـتـصـديـق عـلـى مـذكـرة تـفـاهـم بـشـأن الـتـعـاون فـي مـجـال أنـشـطـة الـتـقـيـيـس بـين حـكـومـة دولـة قـطـر وحـكـومـة دولـة الـكـويـت 90 / 2025
عدد المواد: 2
فهرس الموضوعات

المواد (1-2)



نـحـن تمـيـم بـن حـمـد آل ثـانـي            أمـيـر دولـة قـطـر ،
بـعـد الاطـلاع عـلـى الـدسـتـور،
وعـلـى وثـيـقـة الـتـصـديـق الـصـادرة فـي الـثـالـث والـعـشـرين مـن شـهـر رمـضـان عـام 1446 هـجـريـة ، المـوافـق لـلـثـالـث والـعـشـرين مـن شـهـر مـارس عـام 2025 مـيـلاديـة ،
وعـلـى اقـتـراح رئـيـس مـجـلـس الـوزراء ووزيـر الـخـارجـيـة ،
وعـلـى مـشـروع المـرسـوم المـقـدم مـن مـجـلـس الـوزراء ،
رسـمـنـا بمـا هـو آت :
 

المواد

المادة 1


صُــودق عـلـى مـذكـرة تـفـاهـم بـشـأن الـتـعـاون فـي مـجـال أنـشـطـة الـتـقـييـس بـين حـكـومـة دولـة قـطـر وحـكـومـة دولـة الـكـويـت ، المـوقـعـة بمـديـنـة الـدوحـة بـتـاريـخ 2025/02/03 ، المـرفـق نـصـهـا بـهـذا المـرسـوم ، وتـكـون لـهـا قـوة الـقـانـون ، وفـقـاً لـلـمـادة (68) مـن الـدسـتـور .

 

المادة 2


عـلـى جـمـيـع الـجـهـات المـخـتـصـة ، كـل فـيـمـا يـخـصـه ، تـنـفـيـذ هـذا المـرسـوم . ويُـعـمـل بـه مـن تـاريـخ صـدوره . ويُـنـشـر فـي  الـجـريـدة الـرسـمـيـة .
