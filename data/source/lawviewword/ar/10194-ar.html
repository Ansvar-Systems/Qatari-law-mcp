قــرار أميــري رقـم (46) لسنـة 2025 بتـعــيــين ســفــيـــر فــــوق الــعـــادة مــفـــوض لــدى بـرونـاي دار الـسـلام 46 / 2025
عدد المواد: 2
فهرس الموضوعات

المواد (1-2)



نحـن تمـيــم بن حـمــد آل ثـاني                         أميــر دولــة قـطـــر ،
بـعـد الاطــلاع عـلى الـدستــور،
وعـلى الـقـانــون رقــم (17) لسـنـة 2021 بشــأن الـسـلـكـين الـدبـلــومــاسـي والـقـنــصـلي ،
وعــلى اقـتــراح رئيـس مجـلـس الـوزراء ووزيــر الخــارجــيــــة ،
قـررنـــا مــا يلــي :
 

المواد

المادة 1


يُعــين الـسـيـد/ عـبـدالله حـمـد عـلـي الأدهـم المـري ، ســفـيــراً فـــــوق الـعـــادة مـفــوضـــاً مــن لــدنـــا لـــدى بـرونـاي دار الـسـلام .

 

المادة 2


عـلى جــمـيــع الجـهــات المخـتـصــة ، كـل فيــمـا يخــصـه ، تنــفـيــذ هــذا الـقـــرار . ويُعـمـل بــه مـن تـاريــخ صــدوره . ويُنــشـــر في الجــريــدة الـرسـمـيـــة .
