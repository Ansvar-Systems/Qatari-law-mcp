مـرسـوم رقـم (91) لـسـنـة 2025 بـالـتـصـديـق عـلـى اتـفـاقـيـة تـعـاون بـشـأن المـسـاعـدات الإنـسـانـيـة بـين حـكـومـة دولـة قـطـر وحـكـومـة الجـمـهـوريـة الـتـركـيـة 91 / 2025
عدد المواد: 2
فهرس الموضوعات

المواد (1-2)



نـحـن تمـيـم بـن حـمـد آل ثـانـي            أمـيـر دولـة قـطـر ،
بـعـد الاطـلاع عـلـى الـدسـتـور،
وعـلـى وثـيـقـة الـتـصـديـق الـصـادرة فـي الـعـاشـر مـن شـهـر شـعـبـان عـام 1446 هـجـريـة ، المـوافـق للـتـاسـع مـن شـهـر فبـرايـر عـام 2025 مـيـلاديـة ،
وعـلـى اقـتـراح رئـيـس مـجـلـس الـوزراء ووزيـر الـخـارجـيـة ،
وعـلـى مـشـروع المـرسـوم المـقـدم مـن مـجـلـس الـوزراء ،
رسـمـنـا بمـا هـو آت :
 

المواد

المادة 1


 صُـودق على اتفاقيـة تعاون بشأن المساعـدات الإنـسـانـيـة بـين حـكـومـة دولـة قـطـر وحـكـومـة الجـمـهـوريـة الـتـركـيـة ، المـوقـعـة بمدينـة أنقـرة بـتـاريـخ 2024/11/14 ، المرفـق نصـهـا بـهـذا المـرسـوم ، وتكون لهـا قـوة الـقـانـون ، وفـقـاً لـلـمـادة (68) مـن الـدسـتـور .

 

المادة 2


عـلـى جـمـيـع الـجـهـات المـخـتـصـة ، كـل فـيـمـا يـخـصـه ، تـنـفـيـذ هـذا المـرسـوم . ويُـعـمـل بـه مـن تـاريـخ صـدوره . ويُـنـشـر فـي الـجـريـدة الـرسـمـيـة .
