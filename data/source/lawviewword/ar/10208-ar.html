مـــرســـوم رقـــم (97) لـسـنـــة 2025 بـالـتـصـديــق عـلـى اتـفـاقـيـة خـدمـات جـويـة بـين حـكـومـة دولـة قـطـر وجـمـهـوريـة تـريـنيـداد وتـوبـاغـو 97 / 2025
عدد المواد: 2
فهرس الموضوعات

المواد (1-2)



نـحـن تمـيـم بـن حـمــد آل ثـانـي           أمـيـر دولـة قـطـــر ،
بـعـد الاطــلاع عـلـى الـدسـتـور،
وعـلـى وثـيـقـة الـتـصـديـق الـصــادرة فـي الــثـالـث والـعـشـريـن مـن شـهــر صـفـر عـام 1446 هــجـريـة ، المـوافــق لـلـسـابـع والـعـشـريـن من شـهـر أغـسـطـس عـام 2024 مـيـلاديـة ،
وعـلـى اقـتـراح رئـيـس مـجـلـس الـوزراء ووزيـر الخـارجـيـة ،
وعـلـى مـشـروع المـرسـوم المـقـدم مـن مـجـلـس الـوزراء ،
رسمنـا بمـا هـو آت :
 

المواد

المادة 1


صُـودق عـلـى اتـفـاقـيـة خـدمـات جـويـة بـين حـكـومـة دولـة قـطـر وجـمـهـوريـة تـريـنيـداد وتـوبـاغـو ، المـوقـعـة بمـديـنـة الـدوحـة بــتـاريـخ 23 /05/ 2024 ، المـرفـق نـصـهـا بـهـذا المـرسـوم ، وتـكـون لـهـا قـوة الـقـانـون، وفـقـاً لـلـمـادة (68) مـن الـدسـتـور.

 

المادة 2


عـلـى جـمـيـع الجـهـات المـخـتـصـة ، كـل فـيـمـا يـخـصـه ، تـنـفـيـذ هـذا المـرســوم . ويُـعـمـل بـه مـن تـاريـخ صــدوره . ويُـنـشــر فـي الجـريــدة الـرسـمـيـة .
